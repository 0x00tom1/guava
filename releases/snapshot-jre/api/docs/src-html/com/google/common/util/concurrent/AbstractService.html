044/**
045 * Base class for implementing services that can handle {@link #doStart} and {@link #doStop}
046 * requests, responding to them with {@link #notifyStarted()} and {@link #notifyStopped()}
047 * callbacks. Its subclasses must manage threads manually; consider {@link
048 * AbstractExecutionThreadService} if you need only a single execution thread.
049 *
050 * @author Jesse Wilson
051 * @author Luke Sandberg
052 * @since 1.0
053 */
054@GwtIncompatible
055@J2ktIncompatible
056@ElementTypesAreNonnullByDefault
057public abstract class AbstractService implements Service {
058  private static final ListenerCallQueue.Event<Listener> STARTING_EVENT =
059      new ListenerCallQueue.Event<Listener>() {
060        @Override
061        public void call(Listener listener) {
062          listener.starting();

065        @Override
066        public String toString() {
067          return "starting()";
068        }
069      };

077        @Override

185  /**
186   * The current state of the service. This should be written with the lock held but can be read
187   * without it because it is an immutable object in a volatile field. This is desirable so that
188   * methods like {@link #state}, {@link #failureCause} and notably {@link #toString} can be run
189   * without grabbing the lock.
190   *
191   * <p>To update this field correctly the lock must be held to guarantee that the state is
192   * consistent.
193   */
194  private volatile StateSnapshot snapshot = new StateSnapshot(NEW);

199  /**
200   * This method is called by {@link #startAsync} to initiate service startup. The invocation of
201   * this method should cause a call to {@link #notifyStarted()}, either during this method's run,
202   * or after it has returned. If startup fails, the invocation should cause a call to {@link
203   * #notifyFailed(Throwable)} instead.
204   *
205   * <p>This method should return promptly; prefer to do work on a different thread where it is
206   * convenient. It is invoked exactly once on service startup, even when {@link #startAsync} is
207   * called multiple times.
208   */
209  @ForOverride
210  protected abstract void doStart();
211
212  /**
213   * This method should be used to initiate service shutdown. The invocation of this method should
214   * cause a call to {@link #notifyStopped()}, either during this method's run, or after it has
215   * returned. If shutdown fails, the invocation should cause a call to {@link
216   * #notifyFailed(Throwable)} instead.
217   *
218   * <p>This method should return promptly; prefer to do work on a different thread where it is
219   * convenient. It is invoked exactly once on service shutdown, even when {@link #stopAsync} is
220   * called multiple times.
221   *
222   * <p>If {@link #stopAsync} is called on a {@link State#STARTING} service, this method is not
223   * invoked immediately. Instead, it will be deferred until after the service is {@link
224   * State#RUNNING}. Services that need to cancel startup work can override {@link #doCancelStart}.
225   */
226  @ForOverride
227  protected abstract void doStop();
228
229  /**
230   * This method is called by {@link #stopAsync} when the service is still starting (i.e. {@link
231   * #startAsync} has been called but {@link #notifyStarted} has not). Subclasses can override the
232   * method to cancel pending work and then call {@link #notifyStopped} to stop the service.
233   *
234   * <p>This method should return promptly; prefer to do work on a different thread where it is
235   * convenient. It is invoked exactly once on service shutdown, even when {@link #stopAsync} is
236   * called multiple times.
237   *
238   * <p>When this method is called {@link #state()} will return {@link State#STOPPING}, which is the
239   * external state observable by the caller of {@link #stopAsync}.
240   *
241   * @since 27.0
242   */
243  @Beta
244  @ForOverride
245  protected void doCancelStart() {}
246
247  @CanIgnoreReturnValue
248  @Override
249  public final Service startAsync() {
250    if (monitor.enterIf(isStartable)) {
251      try {
252        snapshot = new StateSnapshot(STARTING);
253        enqueueStartingEvent();
254        doStart();
255      } catch (Throwable startupFailure) {
256        restoreInterruptIfIsInterruptedException(startupFailure);
257        notifyFailed(startupFailure);
258      } finally {
259        monitor.leave();
260        dispatchListenerEvents();
261      }
262    } else {
263      throw new IllegalStateException("Service " + this + " has already been started");
264    }
265    return this;
266  }
267
268  @CanIgnoreReturnValue
269  @Override
270  public final Service stopAsync() {
271    if (monitor.enterIf(isStoppable)) {
272      try {
273        State previous = state();
274        switch (previous) {
275          case NEW:
276            snapshot = new StateSnapshot(TERMINATED);
277            enqueueTerminatedEvent(NEW);
278            break;
279          case STARTING:
280            snapshot = new StateSnapshot(STARTING, true, null);
281            enqueueStoppingEvent(STARTING);
282            doCancelStart();
283            break;
284          case RUNNING:
285            snapshot = new StateSnapshot(STOPPING);
286            enqueueStoppingEvent(RUNNING);
287            doStop();
288            break;
289          case STOPPING:
290          case TERMINATED:
291          case FAILED:
292            // These cases are impossible due to the if statement above.
293            throw new AssertionError("isStoppable is incorrectly implemented, saw: " + previous);
294        }
295      } catch (Throwable shutdownFailure) {
296        restoreInterruptIfIsInterruptedException(shutdownFailure);
297        notifyFailed(shutdownFailure);
298      } finally {
299        monitor.leave();
300        dispatchListenerEvents();
301      }
302    }
303    return this;
304  }

306  @Override
307  public final void awaitRunning() {
308    monitor.enterWhenUninterruptibly(hasReachedRunning);
309    try {
310      checkCurrentState(RUNNING);
311    } finally {
312      monitor.leave();
313    }
314  }

322  @Override
323  public final void awaitRunning(long timeout, TimeUnit unit) throws TimeoutException {
324    if (monitor.enterWhenUninterruptibly(hasReachedRunning, timeout, unit)) {
325      try {
326        checkCurrentState(RUNNING);
327      } finally {
328        monitor.leave();
329      }
330    } else {
331      // It is possible due to races that we are currently in the expected state even though we
332      // timed out. e.g. if we weren't event able to grab the lock within the timeout we would never
333      // even check the guard. I don't think we care too much about this use case but it could lead
334      // to a confusing error message.
335      throw new TimeoutException("Timed out waiting for " + this + " to reach the RUNNING state.");
336    }
337  }
338
339  @Override
340  public final void awaitTerminated() {
341    monitor.enterWhenUninterruptibly(isStopped);
342    try {
343      checkCurrentState(TERMINATED);
344    } finally {
345      monitor.leave();
346    }
347  }

355  @Override
356  public final void awaitTerminated(long timeout, TimeUnit unit) throws TimeoutException {
357    if (monitor.enterWhenUninterruptibly(isStopped, timeout, unit)) {
358      try {
359        checkCurrentState(TERMINATED);
360      } finally {
361        monitor.leave();
362      }
363    } else {
364      // It is possible due to races that we are currently in the expected state even though we
365      // timed out. e.g. if we weren't event able to grab the lock within the timeout we would never
366      // even check the guard. I don't think we care too much about this use case but it could lead
367      // to a confusing error message.
368      throw new TimeoutException(
369          "Timed out waiting for "
370              + this
371              + " to reach a terminal state. "
372              + "Current state: "
373              + state());
374    }
375  }

377  /** Checks that the current state is equal to the expected state. */
378  @GuardedBy("monitor")
379  private void checkCurrentState(State expected) {
380    State actual = state();
381    if (actual != expected) {
382      if (actual == FAILED) {
383        // Handle this specially so that we can include the failureCause, if there is one.
384        throw new IllegalStateException(
385            "Expected the service " + this + " to be " + expected + ", but the service has FAILED",
386            failureCause());
387      }
388      throw new IllegalStateException(
389          "Expected the service " + this + " to be " + expected + ", but was " + actual);
390    }
391  }

393  /**
394   * Implementing classes should invoke this method once their service has started. It will cause
395   * the service to transition from {@link State#STARTING} to {@link State#RUNNING}.
396   *
397   * @throws IllegalStateException if the service is not {@link State#STARTING}.
398   */
399  protected final void notifyStarted() {
400    monitor.enter();
401    try {
402      // We have to examine the internal state of the snapshot here to properly handle the stop
403      // while starting case.
404      if (snapshot.state != STARTING) {
405        IllegalStateException failure =
406            new IllegalStateException(
407                "Cannot notifyStarted() when the service is " + snapshot.state);
408        notifyFailed(failure);
409        throw failure;
410      }
411
412      if (snapshot.shutdownWhenStartupFinishes) {
413        snapshot = new StateSnapshot(STOPPING);
414        // We don't call listeners here because we already did that when we set the
415        // shutdownWhenStartupFinishes flag.
416        doStop();
417      } else {
418        snapshot = new StateSnapshot(RUNNING);
419        enqueueRunningEvent();
420      }
421    } finally {
422      monitor.leave();
423      dispatchListenerEvents();
424    }
425  }
426
427  /**
428   * Implementing classes should invoke this method once their service has stopped. It will cause
429   * the service to transition from {@link State#STARTING} or {@link State#STOPPING} to {@link
430   * State#TERMINATED}.
431   *
432   * @throws IllegalStateException if the service is not one of {@link State#STOPPING}, {@link
433   *     State#STARTING}, or {@link State#RUNNING}.
434   */
435  protected final void notifyStopped() {
436    monitor.enter();
437    try {
438      State previous = state();
439      switch (previous) {
440        case NEW:
441        case TERMINATED:
442        case FAILED:
443          throw new IllegalStateException("Cannot notifyStopped() when the service is " + previous);
444        case RUNNING:
445        case STARTING:
446        case STOPPING:
447          snapshot = new StateSnapshot(TERMINATED);
448          enqueueTerminatedEvent(previous);
449          break;
450      }
451    } finally {
452      monitor.leave();
453      dispatchListenerEvents();
454    }
455  }
456
457  /**
458   * Invoke this method to transition the service to the {@link State#FAILED}. The service will
459   * <b>not be stopped</b> if it is running. Invoke this method when a service has failed critically
460   * or otherwise cannot be started nor stopped.
461   */
462  protected final void notifyFailed(Throwable cause) {
463    checkNotNull(cause);
464
465    monitor.enter();
466    try {
467      State previous = state();
468      switch (previous) {
469        case NEW:
470        case TERMINATED:
471          throw new IllegalStateException("Failed while in state:" + previous, cause);
472        case RUNNING:
473        case STARTING:
474        case STOPPING:
475          snapshot = new StateSnapshot(FAILED, false, cause);
476          enqueueFailedEvent(previous, cause);
477          break;
478        case FAILED:
479          // Do nothing
480          break;
481      }
482    } finally {
483      monitor.leave();
484      dispatchListenerEvents();
485    }
486  }

515  /**
516   * Attempts to execute all the listeners in {@link #listeners} while not holding the {@link
517   * #monitor}.
518   */
519  private void dispatchListenerEvents() {
520    if (!monitor.isOccupiedByCurrentThread()) {
521      listeners.dispatch();
522    }
523  }
524
525  private void enqueueStartingEvent() {
526    listeners.enqueue(STARTING_EVENT);
527  }
528
529  private void enqueueRunningEvent() {
530    listeners.enqueue(RUNNING_EVENT);
531  }
532
533  private void enqueueStoppingEvent(final State from) {
534    if (from == State.STARTING) {
535      listeners.enqueue(STOPPING_FROM_STARTING_EVENT);
536    } else if (from == State.RUNNING) {
537      listeners.enqueue(STOPPING_FROM_RUNNING_EVENT);
538    } else {
539      throw new AssertionError();
540    }
541  }
542
543  private void enqueueTerminatedEvent(final State from) {
544    switch (from) {
545      case NEW:
546        listeners.enqueue(TERMINATED_FROM_NEW_EVENT);
547        break;
548      case STARTING:
549        listeners.enqueue(TERMINATED_FROM_STARTING_EVENT);
550        break;
551      case RUNNING:
552        listeners.enqueue(TERMINATED_FROM_RUNNING_EVENT);
553        break;
554      case STOPPING:
555        listeners.enqueue(TERMINATED_FROM_STOPPING_EVENT);
556        break;
557      case TERMINATED:
558      case FAILED:
559        throw new AssertionError();
560    }
561  }
562
563  private void enqueueFailedEvent(final State from, final Throwable cause) {
564    // can't memoize this one due to the exception
565    listeners.enqueue(
566        new ListenerCallQueue.Event<Listener>() {
567          @Override
568          public void call(Listener listener) {
569            listener.failed(from, cause);
570          }
571
572          @Override
573          public String toString() {
574            return "failed({from = " + from + ", cause = " + cause + "})";
575          }
576        });
577  }

579  /**
580   * An immutable snapshot of the current state of the service. This class represents a consistent
581   * snapshot of the state and therefore it can be used to answer simple queries without needing to
582   * grab a lock.
583   */
584  // @Immutable except that Throwable is mutable (initCause(), setStackTrace(), mutable subclasses).
585  private static final class StateSnapshot {
586    /**
587     * The internal state, which equals external state unless shutdownWhenStartupFinishes is true.
588     */
589    final State state;
590
591    /** If true, the user requested a shutdown while the service was still starting up. */
592    final boolean shutdownWhenStartupFinishes;
593
594    /**
595     * The exception that caused this service to fail. This will be {@code null} unless the service
596     * has failed.
597     */
598    @CheckForNull final Throwable failure;
599
600    StateSnapshot(State internalState) {
601      this(internalState, false, null);
602    }
603
604    StateSnapshot(
605        State internalState, boolean shutdownWhenStartupFinishes, @CheckForNull Throwable failure) {
606      checkArgument(
607          !shutdownWhenStartupFinishes || internalState == STARTING,
608          "shutdownWhenStartupFinishes can only be set if state is STARTING. Got %s instead.",
609          internalState);
610      checkArgument(
611          (failure != null) == (internalState == FAILED),
612          "A failure cause should be set if and only if the state is failed.  Got %s and %s "
613              + "instead.",
614          internalState,
615          failure);
616      this.state = internalState;
617      this.shutdownWhenStartupFinishes = shutdownWhenStartupFinishes;
618      this.failure = failure;
619    }
620
621    /** @see Service#state() */
622    State externalState() {
623      if (shutdownWhenStartupFinishes && state == STARTING) {
624        return STOPPING;
625      } else {
626        return state;
627      }
628    }
629
630    /** @see Service#failureCause() */
631    Throwable failureCause() {
632      checkState(
633          state == FAILED,
634          "failureCause() is only valid if the service has failed, service is %s",
635          state);
636      // requireNonNull is safe because the constructor requires a non-null cause with state=FAILED.
637      return requireNonNull(failure);
638    }
639  }
