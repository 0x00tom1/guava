035/**
036 * An immutable well-formed internet domain name, such as {@code com} or {@code foo.co.uk}. Only
037 * syntactic analysis is performed; no DNS lookups or other network interactions take place. Thus
038 * there is no guarantee that the domain actually exists on the internet.
039 *
040 * <p>One common use of this class is to determine whether a given string is likely to represent an
041 * addressable domain on the web -- that is, for a candidate string {@code "xxx"}, might browsing to
042 * {@code "http://xxx/"} result in a webpage being displayed? In the past, this test was frequently
043 * done by determining whether the domain ended with a {@linkplain #isPublicSuffix() public suffix}
044 * but was not itself a public suffix. However, this test is no longer accurate. There are many
045 * domains which are both public suffixes and addressable as hosts; {@code "uk.com"} is one example.
046 * Using the subset of public suffixes that are {@linkplain #isRegistrySuffix() registry suffixes},
047 * one can get a better result, as only a few registry suffixes are addressable. However, the most
048 * useful test to determine if a domain is a plausible web host is {@link #hasPublicSuffix()}. This
049 * will return {@code true} for many domains which (currently) are not hosts, such as {@code "com"},
050 * but given that any public suffix may become a host without warning, it is better to err on the
051 * side of permissiveness and thus avoid spurious rejection of valid sites. Of course, to actually
052 * determine addressability of any host, clients of this class will need to perform their own DNS
053 * lookups.
054 *
055 * <p>During construction, names are normalized in two ways:
056 *
057 * <ol>
058 *   <li>ASCII uppercase characters are converted to lowercase.
059 *   <li>Unicode dot separators other than the ASCII period ({@code '.'}) are converted to the ASCII
060 *       period.
061 * </ol>
062 *
063 * <p>The normalized values will be returned from {@link #toString()} and {@link #parts()}, and will
064 * be reflected in the result of {@link #equals(Object)}.
065 *
066 * <p><a href="http://en.wikipedia.org/wiki/Internationalized_domain_name">Internationalized domain
067 * names</a> such as {@code 网络.cn} are supported, as are the equivalent <a
068 * href="http://en.wikipedia.org/wiki/Internationalized_domain_name">IDNA Punycode-encoded</a>
069 * versions.
070 *
071 * @author Catherine Berry
072 * @since 5.0
073 */
074@GwtCompatible(emulated = true)
075@Immutable
076@ElementTypesAreNonnullByDefault
077public final class InternetDomainName {
078
079  private static final CharMatcher DOTS_MATCHER = CharMatcher.anyOf(".\u3002\uFF0E\uFF61");
080  private static final Splitter DOT_SPLITTER = Splitter.on('.');
081  private static final Joiner DOT_JOINER = Joiner.on('.');
082
083  /**
084   * Value of {@link #publicSuffixIndex()} or {@link #registrySuffixIndex()} which indicates that no
085   * relevant suffix was found.
086   */
087  private static final int NO_SUFFIX_FOUND = -1;
088
089  /**
090   * Value of {@link #publicSuffixIndexCache} or {@link #registrySuffixIndexCache} which indicates
091   * that they were not initialized yet.
092   */
093  private static final int SUFFIX_NOT_INITIALIZED = -2;
094
095  /**
096   * Maximum parts (labels) in a domain name. This value arises from the 255-octet limit described
097   * in <a href="http://www.ietf.org/rfc/rfc2181.txt">RFC 2181</a> part 11 with the fact that the
098   * encoding of each part occupies at least two bytes (dot plus label externally, length byte plus
099   * label internally). Thus, if all labels have the minimum size of one byte, 127 of them will fit.
100   */
101  private static final int MAX_PARTS = 127;
102
103  /**
104   * Maximum length of a full domain name, including separators, and leaving room for the root
105   * label. See <a href="http://www.ietf.org/rfc/rfc2181.txt">RFC 2181</a> part 11.
106   */
107  private static final int MAX_LENGTH = 253;
108
109  /**
110   * Maximum size of a single part of a domain name. See <a
111   * href="http://www.ietf.org/rfc/rfc2181.txt">RFC 2181</a> part 11.
112   */
113  private static final int MAX_DOMAIN_PART_LENGTH = 63;
114
115  /** The full domain name, converted to lower case. */
116  private final String name;
117
118  /** The parts of the domain name, converted to lower case. */
119  private final ImmutableList<String> parts;
120
121  /**
122   * Cached value of #publicSuffixIndex(). Do not use directly.
123   *
124   * <p>Since this field isn't {@code volatile}, if an instance of this class is shared across
125   * threads before it is initialized, then each thread is likely to compute their own copy of the
126   * value.
127   */
128  @SuppressWarnings("Immutable")
129  private int publicSuffixIndexCache = SUFFIX_NOT_INITIALIZED;
130
131  /**
132   * Cached value of #registrySuffixIndex(). Do not use directly.
133   *
134   * <p>Since this field isn't {@code volatile}, if an instance of this class is shared across
135   * threads before it is initialized, then each thread is likely to compute their own copy of the
136   * value.
137   */
138  @SuppressWarnings("Immutable")
139  private int registrySuffixIndexCache = SUFFIX_NOT_INITIALIZED;
140
141  /** Constructor used to implement {@link #from(String)}, and from subclasses. */
142  InternetDomainName(String name) {
143    // Normalize:
144    // * ASCII characters to lowercase
145    // * All dot-like characters to '.'
146    // * Strip trailing '.'
147
148    name = Ascii.toLowerCase(DOTS_MATCHER.replaceFrom(name, '.'));
149
150    if (name.endsWith(".")) {
151      name = name.substring(0, name.length() - 1);
152    }
153
154    checkArgument(name.length() <= MAX_LENGTH, "Domain name too long: '%s':", name);
155    this.name = name;
156
157    this.parts = ImmutableList.copyOf(DOT_SPLITTER.split(name));
158    checkArgument(parts.size() <= MAX_PARTS, "Domain has too many parts: '%s'", name);
159    checkArgument(validateSyntax(parts), "Not a valid domain name: '%s'", name);
160  }
161
162  /**
163   * The index in the {@link #parts()} list at which the public suffix begins. For example, for the
164   * domain name {@code myblog.blogspot.co.uk}, the value would be 1 (the index of the {@code
165   * blogspot} part). The value is negative (specifically, {@link #NO_SUFFIX_FOUND}) if no public
166   * suffix was found.
167   */
168  private int publicSuffixIndex() {
169    if (publicSuffixIndexCache == SUFFIX_NOT_INITIALIZED) {
170      publicSuffixIndexCache = findSuffixOfType(Optional.<PublicSuffixType>absent());
171    }
172    return publicSuffixIndexCache;
173  }
174
175  /**
176   * The index in the {@link #parts()} list at which the registry suffix begins. For example, for
177   * the domain name {@code myblog.blogspot.co.uk}, the value would be 2 (the index of the {@code
178   * co} part). The value is negative (specifically, {@link #NO_SUFFIX_FOUND}) if no registry suffix
179   * was found.
180   */
181  private int registrySuffixIndex() {
182    if (registrySuffixIndexCache == SUFFIX_NOT_INITIALIZED) {
183      registrySuffixIndexCache = findSuffixOfType(Optional.of(PublicSuffixType.REGISTRY));
184    }
185    return registrySuffixIndexCache;
186  }
187
188  /**
189   * Returns the index of the leftmost part of the suffix, or -1 if not found. Note that the value
190   * defined as a suffix may not produce {@code true} results from {@link #isPublicSuffix()} or
191   * {@link #isRegistrySuffix()} if the domain ends with an excluded domain pattern such as {@code
192   * "nhs.uk"}.
193   *
194   * <p>If a {@code desiredType} is specified, this method only finds suffixes of the given type.
195   * Otherwise, it finds the first suffix of any type.
196   */
197  private int findSuffixOfType(Optional<PublicSuffixType> desiredType) {
198    int partsSize = parts.size();
199
200    for (int i = 0; i < partsSize; i++) {
201      String ancestorName = DOT_JOINER.join(parts.subList(i, partsSize));
202
203      if (i > 0
204          && matchesType(
205              desiredType, Optional.fromNullable(PublicSuffixPatterns.UNDER.get(ancestorName)))) {
206        return i - 1;
207      }
208
209      if (matchesType(
210          desiredType, Optional.fromNullable(PublicSuffixPatterns.EXACT.get(ancestorName)))) {
211        return i;
212      }
213
214      // Excluded domains (e.g. !nhs.uk) use the next highest
215      // domain as the effective public suffix (e.g. uk).
216
217      if (PublicSuffixPatterns.EXCLUDED.containsKey(ancestorName)) {
218        return i + 1;
219      }
220    }
221
222    return NO_SUFFIX_FOUND;
223  }

225  /**
226   * Returns an instance of {@link InternetDomainName} after lenient validation. Specifically,
227   * validation against <a href="http://www.ietf.org/rfc/rfc3490.txt">RFC 3490</a>
228   * ("Internationalizing Domain Names in Applications") is skipped, while validation against <a
229   * href="http://www.ietf.org/rfc/rfc1035.txt">RFC 1035</a> is relaxed in the following ways:
230   *
231   * <ul>
232   *   <li>Any part containing non-ASCII characters is considered valid.
233   *   <li>Underscores ('_') are permitted wherever dashes ('-') are permitted.
234   *   <li>Parts other than the final part may start with a digit, as mandated by <a
235   *       href="https://tools.ietf.org/html/rfc1123#section-2">RFC 1123</a>.
236   * </ul>
237   *
238   * @param domain A domain name (not IP address)
239   * @throws IllegalArgumentException if {@code domain} is not syntactically valid according to
240   *     {@link #isValid}
241   * @since 10.0 (previously named {@code fromLenient})
242   */
243  @CanIgnoreReturnValue // TODO(b/219820829): consider removing
244  public static InternetDomainName from(String domain) {
245    return new InternetDomainName(checkNotNull(domain));
246  }
247
248  /**
249   * Validation method used by {@code from} to ensure that the domain name is syntactically valid
250   * according to RFC 1035.
251   *
252   * @return Is the domain name syntactically valid?
253   */
254  private static boolean validateSyntax(List<String> parts) {
255    int lastIndex = parts.size() - 1;
256
257    // Validate the last part specially, as it has different syntax rules.
258
259    if (!validatePart(parts.get(lastIndex), true)) {
260      return false;
261    }
262
263    for (int i = 0; i < lastIndex; i++) {
264      String part = parts.get(i);
265      if (!validatePart(part, false)) {
266        return false;
267      }
268    }
269
270    return true;
271  }
272
273  private static final CharMatcher DASH_MATCHER = CharMatcher.anyOf("-_");
274
275  private static final CharMatcher DIGIT_MATCHER = CharMatcher.inRange('0', '9');
276
277  private static final CharMatcher LETTER_MATCHER =
278      CharMatcher.inRange('a', 'z').or(CharMatcher.inRange('A', 'Z'));
279
280  private static final CharMatcher PART_CHAR_MATCHER =
281      DIGIT_MATCHER.or(LETTER_MATCHER).or(DASH_MATCHER);
282
283  /**
284   * Helper method for {@link #validateSyntax(List)}. Validates that one part of a domain name is
285   * valid.
286   *
287   * @param part The domain name part to be validated
288   * @param isFinalPart Is this the final (rightmost) domain part?
289   * @return Whether the part is valid
290   */
291  private static boolean validatePart(String part, boolean isFinalPart) {
292
293    // These tests could be collapsed into one big boolean expression, but
294    // they have been left as independent tests for clarity.
295
296    if (part.length() < 1 || part.length() > MAX_DOMAIN_PART_LENGTH) {
297      return false;
298    }
299
300    /*
301     * GWT claims to support java.lang.Character's char-classification methods, but it actually only
302     * works for ASCII. So for now, assume any non-ASCII characters are valid. The only place this
303     * seems to be documented is here:
304     * https://groups.google.com/d/topic/google-web-toolkit-contributors/1UEzsryq1XI
305     *
306     * <p>ASCII characters in the part are expected to be valid per RFC 1035, with underscore also
307     * being allowed due to widespread practice.
308     */
309
310    String asciiChars = CharMatcher.ascii().retainFrom(part);
311
312    if (!PART_CHAR_MATCHER.matchesAllOf(asciiChars)) {
313      return false;
314    }
315
316    // No initial or final dashes or underscores.
317
318    if (DASH_MATCHER.matches(part.charAt(0))
319        || DASH_MATCHER.matches(part.charAt(part.length() - 1))) {
320      return false;
321    }
322
323    /*
324     * Note that we allow (in contravention of a strict interpretation of the relevant RFCs) domain
325     * parts other than the last may begin with a digit (for example, "3com.com"). It's important to
326     * disallow an initial digit in the last part; it's the only thing that stops an IPv4 numeric
327     * address like 127.0.0.1 from looking like a valid domain name.
328     */
329
330    if (isFinalPart && DIGIT_MATCHER.matches(part.charAt(0))) {
331      return false;
332    }
333
334    return true;
335  }

346  /**
347   * Indicates whether this domain name represents a <i>public suffix</i>, as defined by the Mozilla
348   * Foundation's <a href="http://publicsuffix.org/">Public Suffix List</a> (PSL). A public suffix
349   * is one under which Internet users can directly register names, such as {@code com}, {@code
350   * co.uk} or {@code pvt.k12.wy.us}. Examples of domain names that are <i>not</i> public suffixes
351   * include {@code google.com}, {@code foo.co.uk}, and {@code myblog.blogspot.com}.
352   *
353   * <p>Public suffixes are a proper superset of {@linkplain #isRegistrySuffix() registry suffixes}.
354   * The list of public suffixes additionally contains privately owned domain names under which
355   * Internet users can register subdomains. An example of a public suffix that is not a registry
356   * suffix is {@code blogspot.com}. Note that it is true that all public suffixes <i>have</i>
357   * registry suffixes, since domain name registries collectively control all internet domain names.
358   *
359   * <p>For considerations on whether the public suffix or registry suffix designation is more
360   * suitable for your application, see <a
361   * href="https://github.com/google/guava/wiki/InternetDomainNameExplained">this article</a>.
362   *
363   * @return {@code true} if this domain name appears exactly on the public suffix list
364   * @since 6.0
365   */
366  public boolean isPublicSuffix() {
367    return publicSuffixIndex() == 0;
368  }
369
370  /**
371   * Indicates whether this domain name ends in a {@linkplain #isPublicSuffix() public suffix},
372   * including if it is a public suffix itself. For example, returns {@code true} for {@code
373   * www.google.com}, {@code foo.co.uk} and {@code com}, but not for {@code invalid} or {@code
374   * google.invalid}. This is the recommended method for determining whether a domain is potentially
375   * an addressable host.
376   *
377   * <p>Note that this method is equivalent to {@link #hasRegistrySuffix()} because all registry
378   * suffixes are public suffixes <i>and</i> all public suffixes have registry suffixes.
379   *
380   * @since 6.0
381   */
382  public boolean hasPublicSuffix() {
383    return publicSuffixIndex() != NO_SUFFIX_FOUND;
384  }

386  /**
387   * Returns the {@linkplain #isPublicSuffix() public suffix} portion of the domain name, or {@code
388   * null} if no public suffix is present.
389   *
390   * @since 6.0
391   */
392  @CheckForNull
393  public InternetDomainName publicSuffix() {
394    return hasPublicSuffix() ? ancestor(publicSuffixIndex()) : null;
395  }

397  /**
398   * Indicates whether this domain name ends in a {@linkplain #isPublicSuffix() public suffix},
399   * while not being a public suffix itself. For example, returns {@code true} for {@code
400   * www.google.com}, {@code foo.co.uk} and {@code myblog.blogspot.com}, but not for {@code com},
401   * {@code co.uk}, {@code google.invalid}, or {@code blogspot.com}.
402   *
403   * <p>This method can be used to determine whether it will probably be possible to set cookies on
404   * the domain, though even that depends on individual browsers' implementations of cookie
405   * controls. See <a href="http://www.ietf.org/rfc/rfc2109.txt">RFC 2109</a> for details.
406   *
407   * @since 6.0
408   */
409  public boolean isUnderPublicSuffix() {
410    return publicSuffixIndex() > 0;
411  }

413  /**
414   * Indicates whether this domain name is composed of exactly one subdomain component followed by a
415   * {@linkplain #isPublicSuffix() public suffix}. For example, returns {@code true} for {@code

429  /**
430   * Returns the portion of this domain name that is one level beneath the {@linkplain
431   * #isPublicSuffix() public suffix}. For example, for {@code x.adwords.google.co.uk} it returns
432   * {@code google.co.uk}, since {@code co.uk} is a public suffix. Similarly, for {@code
433   * myblog.blogspot.com} it returns the same domain, {@code myblog.blogspot.com}, since {@code
434   * blogspot.com} is a public suffix.
435   *
436   * <p>If {@link #isTopPrivateDomain()} is true, the current domain name instance is returned.
437   *
438   * <p>This method can be used to determine the probable highest level parent domain for which
439   * cookies may be set, though even that depends on individual browsers' implementations of cookie
440   * controls.
441   *
442   * @throws IllegalStateException if this domain does not end with a public suffix
443   * @since 6.0
444   */
445  public InternetDomainName topPrivateDomain() {
446    if (isTopPrivateDomain()) {
447      return this;
448    }
449    checkState(isUnderPublicSuffix(), "Not under a public suffix: %s", name);
450    return ancestor(publicSuffixIndex() - 1);
451  }

574   * Returns the ancestor of the current domain at the given number of levels "higher" (rightward)
575   * in the subdomain list. The number of levels must be non-negative, and less than {@code N-1},
576   * where {@code N} is the number of parts in the domain.
577   *
578   * <p>TODO: Reasonable candidate for addition to public API.
579   */
580  private InternetDomainName ancestor(int levels) {
581    return from(DOT_JOINER.join(parts.subList(levels, parts.size())));
582  }

584  /**
585   * Creates and returns a new {@code InternetDomainName} by prepending the argument and a dot to
586   * the current name. For example, {@code InternetDomainName.from("foo.com").child("www.bar")}
587   * returns a new {@code InternetDomainName} with the value {@code www.bar.foo.com}. Only lenient
588   * validation is performed, as described {@link #from(String) here}.
589   *
590   * @throws NullPointerException if leftParts is null
591   * @throws IllegalArgumentException if the resulting name is not valid
592   */
593  public InternetDomainName child(String leftParts) {
594    return from(checkNotNull(leftParts) + "." + name);
595  }

597  /**
598   * Indicates whether the argument is a syntactically valid domain name using lenient validation.
599   * Specifically, validation against <a href="http://www.ietf.org/rfc/rfc3490.txt">RFC 3490</a>
600   * ("Internationalizing Domain Names in Applications") is skipped.
601   *
602   * <p>The following two code snippets are equivalent:
603   *
604   * <pre>{@code
605   * domainName = InternetDomainName.isValid(name)
606   *     ? InternetDomainName.from(name)
607   *     : DEFAULT_DOMAIN;
608   * }</pre>
609   *
610   * <pre>{@code
611   * try {
612   *   domainName = InternetDomainName.from(name);
613   * } catch (IllegalArgumentException e) {
614   *   domainName = DEFAULT_DOMAIN;
615   * }
616   * }</pre>
617   *
618   * @since 8.0 (previously named {@code isValidLenient})
619   */
620  public static boolean isValid(String name) {
621    try {
622      InternetDomainName unused = from(name);
623      return true;
624    } catch (IllegalArgumentException e) {
625      return false;
626    }
627  }

629  /**
630   * If a {@code desiredType} is specified, returns true only if the {@code actualType} is
631   * identical. Otherwise, returns true as long as {@code actualType} is present.
632   */
633  private static boolean matchesType(
634      Optional<PublicSuffixType> desiredType, Optional<PublicSuffixType> actualType) {
635    return desiredType.isPresent() ? desiredType.equals(actualType) : actualType.isPresent();
636  }
