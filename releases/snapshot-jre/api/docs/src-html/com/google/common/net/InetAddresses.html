036/**
037 * Static utility methods pertaining to {@link InetAddress} instances.
038 *
039 * <p><b>Important note:</b> Unlike {@code InetAddress.getByName()}, the methods of this class never
040 * cause DNS services to be accessed. For this reason, you should prefer these methods as much as
041 * possible over their JDK equivalents whenever you are expecting to handle only IP address string
042 * literals -- there is no blocking DNS penalty for a malformed string.
043 *
044 * <p>When dealing with {@link Inet4Address} and {@link Inet6Address} objects as byte arrays (vis.
045 * {@code InetAddress.getAddress()}) they are 4 and 16 bytes in length, respectively, and represent
046 * the address in network byte order.
047 *
048 * <p>Examples of IP addresses and their byte representations:
049 *
050 * <dl>
051 *   <dt>The IPv4 loopback address, {@code "127.0.0.1"}.
052 *   <dd>{@code 7f 00 00 01}
053 *   <dt>The IPv6 loopback address, {@code "::1"}.
054 *   <dd>{@code 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 01}
055 *   <dt>From the IPv6 reserved documentation prefix ({@code 2001:db8::/32}), {@code "2001:db8::1"}.
056 *   <dd>{@code 20 01 0d b8 00 00 00 00 00 00 00 00 00 00 00 01}
057 *   <dt>An IPv6 "IPv4 compatible" (or "compat") address, {@code "::192.168.0.1"}.
058 *   <dd>{@code 00 00 00 00 00 00 00 00 00 00 00 00 c0 a8 00 01}
059 *   <dt>An IPv6 "IPv4 mapped" address, {@code "::ffff:192.168.0.1"}.
060 *   <dd>{@code 00 00 00 00 00 00 00 00 00 00 ff ff c0 a8 00 01}
061 * </dl>
062 *
063 * <p>A few notes about IPv6 "IPv4 mapped" addresses and their observed use in Java.
064 *
065 * <p>"IPv4 mapped" addresses were originally a representation of IPv4 addresses for use on an IPv6
066 * socket that could receive both IPv4 and IPv6 connections (by disabling the {@code IPV6_V6ONLY}
067 * socket option on an IPv6 socket). Yes, it's confusing. Nevertheless, these "mapped" addresses
068 * were never supposed to be seen on the wire. That assumption was dropped, some say mistakenly, in
069 * later RFCs with the apparent aim of making IPv4-to-IPv6 transition simpler.
070 *
071 * <p>Technically one <i>can</i> create a 128bit IPv6 address with the wire format of a "mapped"
072 * address, as shown above, and transmit it in an IPv6 packet header. However, Java's InetAddress
073 * creation methods appear to adhere doggedly to the original intent of the "mapped" address: all
074 * "mapped" addresses return {@link Inet4Address} objects.
075 *
076 * <p>For added safety, it is common for IPv6 network operators to filter all packets where either
077 * the source or destination address appears to be a "compat" or "mapped" address. Filtering
078 * suggestions usually recommend discarding any packets with source or destination addresses in the
079 * invalid range {@code ::/3}, which includes both of these bizarre address formats. For more
080 * information on "bogons", including lists of IPv6 bogon space, see:
081 *
082 * <ul>
083 *   <li><a target="_parent"
084 *       href="http://en.wikipedia.org/wiki/Bogon_filtering">http://en.wikipedia.
085 *       org/wiki/Bogon_filtering</a>
086 *   <li><a target="_parent"
087 *       href="http://www.cymru.com/Bogons/ipv6.txt">http://www.cymru.com/Bogons/ ipv6.txt</a>
088 *   <li><a target="_parent" href="http://www.cymru.com/Bogons/v6bogon.html">http://www.cymru.com/
089 *       Bogons/v6bogon.html</a>
090 *   <li><a target="_parent" href="http://www.space.net/~gert/RIPE/ipv6-filters.html">http://www.
091 *       space.net/~gert/RIPE/ipv6-filters.html</a>
092 * </ul>
093 *
094 * @author Erik Kline
095 * @since 5.0
096 */

111  /**
112   * Returns an {@link Inet4Address}, given a byte array representation of the IPv4 address.
113   *
114   * @param bytes byte array representing an IPv4 address (should be of length 4)
115   * @return {@link Inet4Address} corresponding to the supplied byte array
116   * @throws IllegalArgumentException if a valid {@link Inet4Address} can not be created
117   */
118  private static Inet4Address getInet4Address(byte[] bytes) {
119    checkArgument(
120        bytes.length == 4,
121        "Byte array has invalid length for an IPv4 address: %s != 4.",
122        bytes.length);
123
124    // Given a 4-byte array, this cast should always succeed.
125    return (Inet4Address) bytesToInetAddress(bytes);
126  }

128  /**
129   * Returns the {@link InetAddress} having the given string representation.
130   *
131   * <p>This deliberately avoids all nameservice lookups (e.g. no DNS).
132   *
133   * <p>Anything after a {@code %} in an IPv6 address is ignored (assumed to be a Scope ID).
134   *
135   * <p>This method accepts non-ASCII digits, for example {@code "１９２.１６８.０.１"} (those are fullwidth
136   * characters). That is consistent with {@link InetAddress}, but not with various RFCs. If you
137   * want to accept ASCII digits only, you can use something like {@code
138   * CharMatcher.ascii().matchesAllOf(ipString)}.
139   *
140   * @param ipString {@code String} containing an IPv4 or IPv6 string literal, e.g. {@code
141   *     "192.168.0.1"} or {@code "2001:db8::1"}
142   * @return {@link InetAddress} representing the argument
143   * @throws IllegalArgumentException if the argument is not a valid IP string literal
144   */
145  public static InetAddress forString(String ipString) {
146    byte[] addr = ipStringToBytes(ipString);
147
148    // The argument was malformed, i.e. not an IP string literal.
149    if (addr == null) {
150      throw formatIllegalArgumentException("'%s' is not an IP string literal.", ipString);
151    }
152
153    return bytesToInetAddress(addr);
154  }

172  /** Returns {@code null} if unable to parse into a {@code byte[]}. */
173  @CheckForNull
174  private static byte[] ipStringToBytes(String ipStringParam) {
175    String ipString = ipStringParam;
176    // Make a first pass to categorize the characters in this string.
177    boolean hasColon = false;
178    boolean hasDot = false;
179    int percentIndex = -1;
180    for (int i = 0; i < ipString.length(); i++) {
181      char c = ipString.charAt(i);
182      if (c == '.') {
183        hasDot = true;
184      } else if (c == ':') {
185        if (hasDot) {
186          return null; // Colons must not appear after dots.
187        }
188        hasColon = true;
189      } else if (c == '%') {
190        percentIndex = i;
191        break; // everything after a '%' is ignored (it's a Scope ID): http://superuser.com/a/99753
192      } else if (Character.digit(c, 16) == -1) {
193        return null; // Everything else must be a decimal or hex digit.
194      }
195    }
196
197    // Now decide which address family to parse.
198    if (hasColon) {
199      if (hasDot) {
200        ipString = convertDottedQuadToHex(ipString);
201        if (ipString == null) {
202          return null;
203        }
204      }
205      if (percentIndex != -1) {
206        ipString = ipString.substring(0, percentIndex);
207      }
208      return textToNumericFormatV6(ipString);
209    } else if (hasDot) {
210      if (percentIndex != -1) {
211        return null; // Scope IDs are not supported for IPV4
212      }
213      return textToNumericFormatV4(ipString);
214    }
215    return null;
216  }

218  @CheckForNull
219  private static byte[] textToNumericFormatV4(String ipString) {
220    if (IPV4_DELIMITER_MATCHER.countIn(ipString) + 1 != IPV4_PART_COUNT) {
221      return null; // Wrong number of parts
222    }
223
224    byte[] bytes = new byte[IPV4_PART_COUNT];
225    int start = 0;
226    // Iterate through the parts of the ip string.
227    // Invariant: start is always the beginning of an octet.
228    for (int i = 0; i < IPV4_PART_COUNT; i++) {
229      int end = ipString.indexOf(IPV4_DELIMITER, start);
230      if (end == -1) {
231        end = ipString.length();
232      }
233      try {
234        bytes[i] = parseOctet(ipString, start, end);
235      } catch (NumberFormatException ex) {
236        return null;
237      }
238      start = end + 1;
239    }
240
241    return bytes;
242  }

244  @CheckForNull
245  private static byte[] textToNumericFormatV6(String ipString) {
246    // An address can have [2..8] colons.
247    int delimiterCount = IPV6_DELIMITER_MATCHER.countIn(ipString);
248    if (delimiterCount < 2 || delimiterCount > IPV6_PART_COUNT) {
249      return null;
250    }
251    int partsSkipped = IPV6_PART_COUNT - (delimiterCount + 1); // estimate; may be modified later
252    boolean hasSkip = false;
253    // Scan for the appearance of ::, to mark a skip-format IPV6 string and adjust the partsSkipped
254    // estimate.
255    for (int i = 0; i < ipString.length() - 1; i++) {
256      if (ipString.charAt(i) == IPV6_DELIMITER && ipString.charAt(i + 1) == IPV6_DELIMITER) {
257        if (hasSkip) {
258          return null; // Can't have more than one ::
259        }
260        hasSkip = true;
261        partsSkipped++; // :: means we skipped an extra part in between the two delimiters.
262        if (i == 0) {
263          partsSkipped++; // Begins with ::, so we skipped the part preceding the first :
264        }
265        if (i == ipString.length() - 2) {
266          partsSkipped++; // Ends with ::, so we skipped the part after the last :
267        }
268      }
269    }
270    if (ipString.charAt(0) == IPV6_DELIMITER && ipString.charAt(1) != IPV6_DELIMITER) {
271      return null; // ^: requires ^::
272    }
273    if (ipString.charAt(ipString.length() - 1) == IPV6_DELIMITER
274        && ipString.charAt(ipString.length() - 2) != IPV6_DELIMITER) {
275      return null; // :$ requires ::$
276    }
277    if (hasSkip && partsSkipped <= 0) {
278      return null; // :: must expand to at least one '0'
279    }
280    if (!hasSkip && delimiterCount + 1 != IPV6_PART_COUNT) {
281      return null; // Incorrect number of parts
282    }
283
284    ByteBuffer rawBytes = ByteBuffer.allocate(2 * IPV6_PART_COUNT);
285    try {
286      // Iterate through the parts of the ip string.
287      // Invariant: start is always the beginning of a hextet, or the second ':' of the skip
288      // sequence "::"
289      int start = 0;
290      if (ipString.charAt(0) == IPV6_DELIMITER) {
291        start = 1;
292      }
293      while (start < ipString.length()) {
294        int end = ipString.indexOf(IPV6_DELIMITER, start);
295        if (end == -1) {
296          end = ipString.length();
297        }
298        if (ipString.charAt(start) == IPV6_DELIMITER) {
299          // expand zeroes
300          for (int i = 0; i < partsSkipped; i++) {
301            rawBytes.putShort((short) 0);
302          }
303
304        } else {
305          rawBytes.putShort(parseHextet(ipString, start, end));
306        }
307        start = end + 1;
308      }
309    } catch (NumberFormatException ex) {
310      return null;
311    }
312    return rawBytes.array();
313  }

315  @CheckForNull
316  private static String convertDottedQuadToHex(String ipString) {
317    int lastColon = ipString.lastIndexOf(':');
318    String initialPart = ipString.substring(0, lastColon + 1);
319    String dottedQuad = ipString.substring(lastColon + 1);
320    byte[] quad = textToNumericFormatV4(dottedQuad);
321    if (quad == null) {
322      return null;
323    }
324    String penultimate = Integer.toHexString(((quad[0] & 0xff) << 8) | (quad[1] & 0xff));
325    String ultimate = Integer.toHexString(((quad[2] & 0xff) << 8) | (quad[3] & 0xff));
326    return initialPart + penultimate + ":" + ultimate;
327  }

329  private static byte parseOctet(String ipString, int start, int end) {
330    // Note: we already verified that this string contains only hex digits, but the string may still
331    // contain non-decimal characters.
332    int length = end - start;
333    if (length <= 0 || length > 3) {
334      throw new NumberFormatException();
335    }
336    // Disallow leading zeroes, because no clear standard exists on
337    // whether these should be interpreted as decimal or octal.
338    if (length > 1 && ipString.charAt(start) == '0') {
339      throw new NumberFormatException();
340    }
341    int octet = 0;
342    for (int i = start; i < end; i++) {
343      octet *= 10;
344      int digit = Character.digit(ipString.charAt(i), 10);
345      if (digit < 0) {
346        throw new NumberFormatException();
347      }
348      octet += digit;
349    }
350    if (octet > 255) {
351      throw new NumberFormatException();
352    }
353    return (byte) octet;
354  }

356  // Parse a hextet out of the ipString from start (inclusive) to end (exclusive)
357  private static short parseHextet(String ipString, int start, int end) {
358    // Note: we already verified that this string contains only hex digits.
359    int length = end - start;
360    if (length <= 0 || length > 4) {
361      throw new NumberFormatException();
362    }
363    int hextet = 0;
364    for (int i = start; i < end; i++) {
365      hextet = hextet << 4;
366      hextet |= Character.digit(ipString.charAt(i), 16);
367    }
368    return (short) hextet;
369  }

372   * Convert a byte array into an InetAddress.
373   *
374   * <p>{@link InetAddress#getByAddress} is documented as throwing a checked exception "if IP
375   * address is of illegal length." We replace it with an unchecked exception, for use by callers
376   * who already know that addr is an array of length 4 or 16.
377   *
378   * @param addr the raw 4-byte or 16-byte IP address in big-endian order
379   * @return an InetAddress object created from the raw IP address
380   */
381  private static InetAddress bytesToInetAddress(byte[] addr) {
382    try {
383      return InetAddress.getByAddress(addr);
384    } catch (UnknownHostException e) {
385      throw new AssertionError(e);
386    }
387  }

389  /**
390   * Returns the string representation of an {@link InetAddress}.
391   *
392   * <p>For IPv4 addresses, this is identical to {@link InetAddress#getHostAddress()}, but for IPv6
393   * addresses, the output follows <a href="http://tools.ietf.org/html/rfc5952">RFC 5952</a> section
394   * 4. The main difference is that this method uses "::" for zero compression, while Java's version
395   * uses the uncompressed form.
396   *
397   * <p>This method uses hexadecimal for all IPv6 addresses, including IPv4-mapped IPv6 addresses
398   * such as "::c000:201". The output does not include a Scope ID.
399   *
400   * @param ip {@link InetAddress} to be converted to an address string
401   * @return {@code String} containing the text-formatted IP address
402   * @since 10.0
403   */
404  public static String toAddrString(InetAddress ip) {
405    checkNotNull(ip);
406    if (ip instanceof Inet4Address) {
407      // For IPv4, Java's formatting is good enough.
408      return ip.getHostAddress();
409    }
410    checkArgument(ip instanceof Inet6Address);
411    byte[] bytes = ip.getAddress();
412    int[] hextets = new int[IPV6_PART_COUNT];
413    for (int i = 0; i < hextets.length; i++) {
414      hextets[i] = Ints.fromBytes((byte) 0, (byte) 0, bytes[2 * i], bytes[2 * i + 1]);
415    }
416    compressLongestRunOfZeroes(hextets);
417    return hextetsToIPv6String(hextets);
418  }

420  /**
421   * Identify and mark the longest run of zeroes in an IPv6 address.
422   *
423   * <p>Only runs of two or more hextets are considered. In case of a tie, the leftmost run wins. If
424   * a qualifying run is found, its hextets are replaced by the sentinel value -1.
425   *
426   * @param hextets {@code int[]} mutable array of eight 16-bit hextets
427   */
428  private static void compressLongestRunOfZeroes(int[] hextets) {
429    int bestRunStart = -1;
430    int bestRunLength = -1;
431    int runStart = -1;
432    for (int i = 0; i < hextets.length + 1; i++) {
433      if (i < hextets.length && hextets[i] == 0) {
434        if (runStart < 0) {
435          runStart = i;
436        }
437      } else if (runStart >= 0) {
438        int runLength = i - runStart;
439        if (runLength > bestRunLength) {
440          bestRunStart = runStart;
441          bestRunLength = runLength;
442        }
443        runStart = -1;
444      }
445    }
446    if (bestRunLength >= 2) {
447      Arrays.fill(hextets, bestRunStart, bestRunStart + bestRunLength, -1);
448    }
449  }

451  /**
452   * Convert a list of hextets into a human-readable IPv6 address.
453   *
454   * <p>In order for "::" compression to work, the input should contain negative sentinel values in
455   * place of the elided zeroes.
456   *
457   * @param hextets {@code int[]} array of eight 16-bit hextets, or -1s
458   */
459  private static String hextetsToIPv6String(int[] hextets) {
460    // While scanning the array, handle these state transitions:
461    //   start->num => "num"     start->gap => "::"
462    //   num->num   => ":num"    num->gap   => "::"
463    //   gap->num   => "num"     gap->gap   => ""
464    StringBuilder buf = new StringBuilder(39);
465    boolean lastWasNumber = false;
466    for (int i = 0; i < hextets.length; i++) {
467      boolean thisIsNumber = hextets[i] >= 0;
468      if (thisIsNumber) {
469        if (lastWasNumber) {
470          buf.append(':');
471        }
472        buf.append(Integer.toHexString(hextets[i]));
473      } else {
474        if (i == 0 || lastWasNumber) {
475          buf.append("::");
476        }
477      }
478      lastWasNumber = thisIsNumber;
479    }
480    return buf.toString();
481  }

483  /**
484   * Returns the string representation of an {@link InetAddress} suitable for inclusion in a URI.
485   *
486   * <p>For IPv4 addresses, this is identical to {@link InetAddress#getHostAddress()}, but for IPv6
487   * addresses it compresses zeroes and surrounds the text with square brackets; for example {@code
488   * "[2001:db8::1]"}.
489   *
490   * <p>Per section 3.2.2 of <a target="_parent"
491   * href="http://tools.ietf.org/html/rfc3986#section-3.2.2">RFC 3986</a>, a URI containing an IPv6
492   * string literal is of the form {@code "http://[2001:db8::1]:8888/index.html"}.
493   *
494   * <p>Use of either {@link InetAddresses#toAddrString}, {@link InetAddress#getHostAddress()}, or
495   * this method is recommended over {@link InetAddress#toString()} when an IP address string
496   * literal is desired. This is because {@link InetAddress#toString()} prints the hostname and the
497   * IP address string joined by a "/".
498   *
499   * @param ip {@link InetAddress} to be converted to URI string literal
500   * @return {@code String} containing URI-safe string literal
501   */
502  public static String toUriString(InetAddress ip) {
503    if (ip instanceof Inet6Address) {
504      return "[" + toAddrString(ip) + "]";
505    }
506    return toAddrString(ip);
507  }

509  /**
510   * Returns an InetAddress representing the literal IPv4 or IPv6 host portion of a URL, encoded in
511   * the format specified by RFC 3986 section 3.2.2.
512   *
513   * <p>This method is similar to {@link InetAddresses#forString(String)}, however, it requires that
514   * IPv6 addresses are surrounded by square brackets.
515   *
516   * <p>This method is the inverse of {@link InetAddresses#toUriString(java.net.InetAddress)}.
517   *
518   * <p>This method accepts non-ASCII digits, for example {@code "１９２.１６８.０.１"} (those are fullwidth
519   * characters). That is consistent with {@link InetAddress}, but not with various RFCs. If you
520   * want to accept ASCII digits only, you can use something like {@code
521   * CharMatcher.ascii().matchesAllOf(ipString)}.
522   *
523   * @param hostAddr A RFC 3986 section 3.2.2 encoded IPv4 or IPv6 address
524   * @return an InetAddress representing the address in {@code hostAddr}
525   * @throws IllegalArgumentException if {@code hostAddr} is not a valid IPv4 address, or IPv6
526   *     address surrounded by square brackets
527   */
528  public static InetAddress forUriString(String hostAddr) {
529    InetAddress addr = forUriStringNoThrow(hostAddr);
530    if (addr == null) {
531      throw formatIllegalArgumentException("Not a valid URI IP literal: '%s'", hostAddr);
532    }
533
534    return addr;
535  }

537  @CheckForNull
538  private static InetAddress forUriStringNoThrow(String hostAddr) {
539    checkNotNull(hostAddr);
540
541    // Decide if this should be an IPv6 or IPv4 address.
542    String ipString;
543    int expectBytes;
544    if (hostAddr.startsWith("[") && hostAddr.endsWith("]")) {
545      ipString = hostAddr.substring(1, hostAddr.length() - 1);
546      expectBytes = 16;
547    } else {
548      ipString = hostAddr;
549      expectBytes = 4;
550    }
551
552    // Parse the address, and make sure the length/version is correct.
553    byte[] addr = ipStringToBytes(ipString);
554    if (addr == null || addr.length != expectBytes) {
555      return null;
556    }
557
558    return bytesToInetAddress(addr);
559  }

561  /**
562   * Returns {@code true} if the supplied string is a valid URI IP string literal, {@code false}
563   * otherwise.
564   *
565   * <p>This method accepts non-ASCII digits, for example {@code "１９２.１６８.０.１"} (those are fullwidth
566   * characters). That is consistent with {@link InetAddress}, but not with various RFCs. If you
567   * want to accept ASCII digits only, you can use something like {@code
568   * CharMatcher.ascii().matchesAllOf(ipString)}.
569   *
570   * @param ipString {@code String} to evaluated as an IP URI host string literal
571   * @return {@code true} if the argument is a valid IP URI host
572   */

577  /**
578   * Evaluates whether the argument is an IPv6 "compat" address.
579   *
580   * <p>An "IPv4 compatible", or "compat", address is one with 96 leading bits of zero, with the
581   * remaining 32 bits interpreted as an IPv4 address. These are conventionally represented in
582   * string literals as {@code "::192.168.0.1"}, though {@code "::c0a8:1"} is also considered an
583   * IPv4 compatible address (and equivalent to {@code "::192.168.0.1"}).
584   *
585   * <p>For more on IPv4 compatible addresses see section 2.5.5.1 of <a target="_parent"
586   * href="http://tools.ietf.org/html/rfc4291#section-2.5.5.1">RFC 4291</a>.
587   *
588   * <p>NOTE: This method is different from {@link Inet6Address#isIPv4CompatibleAddress} in that it
589   * more correctly classifies {@code "::"} and {@code "::1"} as proper IPv6 addresses (which they
590   * are), NOT IPv4 compatible addresses (which they are generally NOT considered to be).
591   *
592   * @param ip {@link Inet6Address} to be examined for embedded IPv4 compatible address format
593   * @return {@code true} if the argument is a valid "compat" address
594   */
595  public static boolean isCompatIPv4Address(Inet6Address ip) {
596    if (!ip.isIPv4CompatibleAddress()) {
597      return false;
598    }
599
600    byte[] bytes = ip.getAddress();
601    if ((bytes[12] == 0)
602        && (bytes[13] == 0)
603        && (bytes[14] == 0)
604        && ((bytes[15] == 0) || (bytes[15] == 1))) {
605      return false;
606    }
607
608    return true;
609  }

616   * @throws IllegalArgumentException if the argument is not a valid IPv4 compatible address

618  public static Inet4Address getCompatIPv4Address(Inet6Address ip) {

655  /**
656   * A simple immutable data class to encapsulate the information to be found in a Teredo address.
657   *
658   * <p>All of the fields in this class are encoded in various portions of the IPv6 address as part
659   * of the protocol. More protocols details can be found at: <a target="_parent"
660   * href="http://en.wikipedia.org/wiki/Teredo_tunneling">http://en.wikipedia.
661   * org/wiki/Teredo_tunneling</a>.
662   *
663   * <p>The RFC can be found here: <a target="_parent" href="http://tools.ietf.org/html/rfc4380">RFC
664   * 4380</a>.
665   *
666   * @since 5.0
667   */
668  public static final class TeredoInfo {
669    private final Inet4Address server;
670    private final Inet4Address client;
671    private final int port;
672    private final int flags;
673
674    /**
675     * Constructs a TeredoInfo instance.
676     *
677     * <p>Both server and client can be {@code null}, in which case the value {@code "0.0.0.0"} will
678     * be assumed.
679     *
680     * @throws IllegalArgumentException if either of the {@code port} or the {@code flags} arguments
681     *     are out of range of an unsigned short
682     */
683    // TODO: why is this public?
684    public TeredoInfo(
685        @CheckForNull Inet4Address server, @CheckForNull Inet4Address client, int port, int flags) {
686      checkArgument(
687          (port >= 0) && (port <= 0xffff), "port '%s' is out of range (0 <= port <= 0xffff)", port);
688      checkArgument(
689          (flags >= 0) && (flags <= 0xffff),
690          "flags '%s' is out of range (0 <= flags <= 0xffff)",
691          flags);
692
693      this.server = MoreObjects.firstNonNull(server, ANY4);
694      this.client = MoreObjects.firstNonNull(client, ANY4);
695      this.port = port;
696      this.flags = flags;
697    }
698
699    public Inet4Address getServer() {
700      return server;
701    }
702
703    public Inet4Address getClient() {
704      return client;
705    }
706
707    public int getPort() {
708      return port;
709    }
710
711    public int getFlags() {
712      return flags;
713    }
714  }

716  /**
717   * Evaluates whether the argument is a Teredo address.
718   *
719   * <p>Teredo addresses begin with the {@code "2001::/32"} prefix.
720   *
721   * @param ip {@link Inet6Address} to be examined for Teredo address format
722   * @return {@code true} if the argument is a Teredo address
723   */
724  public static boolean isTeredoAddress(Inet6Address ip) {
725    byte[] bytes = ip.getAddress();
726    return (bytes[0] == (byte) 0x20)
727        && (bytes[1] == (byte) 0x01)
728        && (bytes[2] == 0)
729        && (bytes[3] == 0);
730  }

732  /**
733   * Returns the Teredo information embedded in a Teredo address.
734   *
735   * @param ip {@link Inet6Address} to be examined for embedded Teredo information
736   * @return extracted {@code TeredoInfo}
737   * @throws IllegalArgumentException if the argument is not a valid IPv6 Teredo address
738   */
739  public static TeredoInfo getTeredoInfo(Inet6Address ip) {
740    checkArgument(isTeredoAddress(ip), "Address '%s' is not a Teredo address.", toAddrString(ip));
741
742    byte[] bytes = ip.getAddress();
743    Inet4Address server = getInet4Address(Arrays.copyOfRange(bytes, 4, 8));
744
745    int flags = ByteStreams.newDataInput(bytes, 8).readShort() & 0xffff;
746
747    // Teredo obfuscates the mapped client port, per section 4 of the RFC.
748    int port = ~ByteStreams.newDataInput(bytes, 10).readShort() & 0xffff;
749
750    byte[] clientBytes = Arrays.copyOfRange(bytes, 12, 16);
751    for (int i = 0; i < clientBytes.length; i++) {
752      // Teredo obfuscates the mapped client IP, per section 4 of the RFC.
753      clientBytes[i] = (byte) ~clientBytes[i];
754    }
755    Inet4Address client = getInet4Address(clientBytes);
756
757    return new TeredoInfo(server, client, port, flags);
758  }

760  /**
761   * Evaluates whether the argument is an ISATAP address.
762   *
763   * <p>From RFC 5214: "ISATAP interface identifiers are constructed in Modified EUI-64 format [...]
764   * by concatenating the 24-bit IANA OUI (00-00-5E), the 8-bit hexadecimal value 0xFE, and a 32-bit
765   * IPv4 address in network byte order [...]"
766   *
767   * <p>For more on ISATAP addresses see section 6.1 of <a target="_parent"
768   * href="http://tools.ietf.org/html/rfc5214#section-6.1">RFC 5214</a>.
769   *
770   * @param ip {@link Inet6Address} to be examined for ISATAP address format
771   * @return {@code true} if the argument is an ISATAP address
772   */
773  public static boolean isIsatapAddress(Inet6Address ip) {
774
775    // If it's a Teredo address with the right port (41217, or 0xa101)
776    // which would be encoded as 0x5efe then it can't be an ISATAP address.
777    if (isTeredoAddress(ip)) {
778      return false;
779    }
780
781    byte[] bytes = ip.getAddress();
782
783    if ((bytes[8] | (byte) 0x03) != (byte) 0x03) {
784
785      // Verify that high byte of the 64 bit identifier is zero, modulo
786      // the U/L and G bits, with which we are not concerned.
787      return false;
788    }
789
790    return (bytes[9] == (byte) 0x00) && (bytes[10] == (byte) 0x5e) && (bytes[11] == (byte) 0xfe);
791  }

850  /**
851   * Evaluates whether the argument is an "IPv4 mapped" IPv6 address.
852   *
853   * <p>An "IPv4 mapped" address is anything in the range ::ffff:0:0/96 (sometimes written as
854   * ::ffff:0.0.0.0/96), with the last 32 bits interpreted as an IPv4 address.
855   *
856   * <p>For more on IPv4 mapped addresses see section 2.5.5.2 of <a target="_parent"
857   * href="http://tools.ietf.org/html/rfc4291#section-2.5.5.2">RFC 4291</a>.
858   *
859   * <p>Note: This method takes a {@code String} argument because {@link InetAddress} automatically
860   * collapses mapped addresses to IPv4. (It is actually possible to avoid this using one of the
861   * obscure {@link Inet6Address} methods, but it would be unwise to depend on such a
862   * poorly-documented feature.)
863   *
864   * <p>This method accepts non-ASCII digits. That is consistent with {@link InetAddress}, but not
865   * with various RFCs. If you want to accept ASCII digits only, you can use something like {@code
866   * CharMatcher.ascii().matchesAllOf(ipString)}.
867   *
868   * @param ipString {@code String} to be examined for embedded IPv4-mapped IPv6 address format
869   * @return {@code true} if the argument is a valid "mapped" address
870   * @since 10.0
871   */
872  public static boolean isMappedIPv4Address(String ipString) {
873    byte[] bytes = ipStringToBytes(ipString);
874    if (bytes != null && bytes.length == 16) {
875      for (int i = 0; i < 10; i++) {
876        if (bytes[i] != 0) {
877          return false;
878        }
879      }
880      for (int i = 10; i < 12; i++) {
881        if (bytes[i] != (byte) 0xff) {
882          return false;
883        }
884      }
885      return true;
886    }
887    return false;
888  }

890  /**
891   * Coerces an IPv6 address into an IPv4 address.
892   *
893   * <p>HACK: As long as applications continue to use IPv4 addresses for indexing into tables,
894   * accounting, et cetera, it may be necessary to <b>coerce</b> IPv6 addresses into IPv4 addresses.
895   * This method does so by hashing 64 bits of the IPv6 address into {@code 224.0.0.0/3} (64 bits
896   * into 29 bits):
897   *
898   * <ul>
899   *   <li>If the IPv6 address contains an embedded IPv4 address, the function hashes that.
900   *   <li>Otherwise, it hashes the upper 64 bits of the IPv6 address.
901   * </ul>
902   *
903   * <p>A "coerced" IPv4 address is equivalent to itself.
904   *
905   * <p>NOTE: This method is failsafe for security purposes: ALL IPv6 addresses (except localhost
906   * (::1)) are hashed to avoid the security risk associated with extracting an embedded IPv4
907   * address that might permit elevated privileges.
908   *
909   * @param ip {@link InetAddress} to "coerce"
910   * @return {@link Inet4Address} represented "coerced" address
911   * @since 7.0
912   */
913  public static Inet4Address getCoercedIPv4Address(InetAddress ip) {
914    if (ip instanceof Inet4Address) {
915      return (Inet4Address) ip;
916    }
917
918    // Special cases:
919    byte[] bytes = ip.getAddress();
920    boolean leadingBytesOfZero = true;
921    for (int i = 0; i < 15; ++i) {
922      if (bytes[i] != 0) {
923        leadingBytesOfZero = false;
924        break;
925      }
926    }
927    if (leadingBytesOfZero && (bytes[15] == 1)) {
928      return LOOPBACK4; // ::1
929    } else if (leadingBytesOfZero && (bytes[15] == 0)) {
930      return ANY4; // ::0
931    }
932
933    Inet6Address ip6 = (Inet6Address) ip;
934    long addressAsLong = 0;
935    if (hasEmbeddedIPv4ClientAddress(ip6)) {
936      addressAsLong = getEmbeddedIPv4ClientAddress(ip6).hashCode();
937    } else {
938      // Just extract the high 64 bits (assuming the rest is user-modifiable).
939      addressAsLong = ByteBuffer.wrap(ip6.getAddress(), 0, 8).getLong();
940    }
941
942    // Many strategies for hashing are possible. This might suffice for now.
943    int coercedHash = Hashing.murmur3_32_fixed().hashLong(addressAsLong).asInt();
944
945    // Squash into 224/4 Multicast and 240/4 Reserved space (i.e. 224/3).
946    coercedHash |= 0xe0000000;
947
948    // Fixup to avoid some "illegal" values. Currently the only potential
949    // illegal value is 255.255.255.255.
950    if (coercedHash == 0xffffffff) {
951      coercedHash = 0xfffffffe;
952    }
953
954    return getInet4Address(Ints.toByteArray(coercedHash));
955  }

1025  /**
1026   * Converts a BigInteger to either an IPv4 or IPv6 address. If the IP is IPv4, it must be
1027   * constrainted to 32 bits, otherwise it is constrained to 128 bits.
1028   *
1029   * @param address the address represented as a big integer
1030   * @param isIpv6 whether the created address should be IPv4 or IPv6
1031   * @return the BigInteger converted to an address
1032   * @throws IllegalArgumentException if the BigInteger is not between 0 and maximum value for IPv4
1033   *     or IPv6 respectively
1034   */
1035  private static InetAddress fromBigInteger(BigInteger address, boolean isIpv6) {
1036    checkArgument(address.signum() >= 0, "BigInteger must be greater than or equal to 0");
1037
1038    int numBytes = isIpv6 ? 16 : 4;
1039
1040    byte[] addressBytes = address.toByteArray();
1041    byte[] targetCopyArray = new byte[numBytes];
1042
1043    int srcPos = Math.max(0, addressBytes.length - numBytes);
1044    int copyLength = addressBytes.length - srcPos;
1045    int destPos = numBytes - copyLength;
1046
1047    // Check the extra bytes in the BigInteger are all zero.
1048    for (int i = 0; i < srcPos; i++) {
1049      if (addressBytes[i] != 0x00) {
1050        throw formatIllegalArgumentException(
1051            "BigInteger cannot be converted to InetAddress because it has more than %d"
1052                + " bytes: %s",
1053            numBytes, address);
1054      }
1055    }
1056
1057    // Copy the bytes into the least significant positions.
1058    System.arraycopy(addressBytes, srcPos, targetCopyArray, destPos, copyLength);
1059
1060    try {
1061      return InetAddress.getByAddress(targetCopyArray);
1062    } catch (UnknownHostException impossible) {
1063      throw new AssertionError(impossible);
1064    }
1065  }

1087   * both IPv4 and IPv6 addresses.
1088   *
1089   * @param address the InetAddress to decrement

1098      addr[i] = (byte) 0xff;
1099      i--;
1100    }
