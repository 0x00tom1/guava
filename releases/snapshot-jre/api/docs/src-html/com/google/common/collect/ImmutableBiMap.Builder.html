037/**
038 * A {@link BiMap} whose contents will never change, with many other important properties detailed
039 * at {@link ImmutableCollection}.
040 *
041 * @author Jared Levy
042 * @since 2.0
043 */
044@GwtCompatible(serializable = true, emulated = true)
045@ElementTypesAreNonnullByDefault
046public abstract class ImmutableBiMap<K, V> extends ImmutableBiMapFauxverideShim<K, V>
047    implements BiMap<K, V> {
048
049  /**
050   * Returns a {@link Collector} that accumulates elements into an {@code ImmutableBiMap} whose keys
051   * and values are the result of applying the provided mapping functions to the input elements.
052   * Entries appear in the result {@code ImmutableBiMap} in encounter order.
053   *
054   * <p>If the mapped keys or values contain duplicates (according to {@link Object#equals(Object)},
055   * an {@code IllegalArgumentException} is thrown when the collection operation is performed. (This
056   * differs from the {@code Collector} returned by {@link Collectors#toMap(Function, Function)},
057   * which throws an {@code IllegalStateException}.)
058   *
059   * @since 21.0
060   */
061  public static <T extends @Nullable Object, K, V>
062      Collector<T, ?, ImmutableBiMap<K, V>> toImmutableBiMap(
063          Function<? super T, ? extends K> keyFunction,
064          Function<? super T, ? extends V> valueFunction) {
065    return CollectCollectors.toImmutableBiMap(keyFunction, valueFunction);
066  }

151  /**
152   * A builder for creating immutable bimap instances, especially {@code public static final} bimaps
153   * ("constant bimaps"). Example:
154   *
155   * <pre>{@code
156   * static final ImmutableBiMap<String, Integer> WORD_TO_INT =
157   *     new ImmutableBiMap.Builder<String, Integer>()
158   *         .put("one", 1)
159   *         .put("two", 2)
160   *         .put("three", 3)
161   *         .build();
162   * }</pre>
163   *
164   * <p>For <i>small</i> immutable bimaps, the {@code ImmutableBiMap.of()} methods are even more
165   * convenient.
166   *
167   * <p>By default, a {@code Builder} will generate bimaps that iterate over entries in the order
168   * they were inserted into the builder. For example, in the above example, {@code
169   * WORD_TO_INT.entrySet()} is guaranteed to iterate over the entries in the order {@code "one"=1,
170   * "two"=2, "three"=3}, and {@code keySet()} and {@code values()} respect the same order. If you
171   * want a different order, consider using {@link #orderEntriesByValue(Comparator)}, which changes
172   * this builder to sort entries by value.
173   *
174   * <p>Builder instances can be reused - it is safe to call {@link #build} multiple times to build
175   * multiple bimaps in series. Each bimap is a superset of the bimaps created before it.
176   *
177   * @since 2.0
178   */
179  public static final class Builder<K, V> extends ImmutableMap.Builder<K, V> {
180
181    /**
182     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
183     * ImmutableBiMap#builder}.
184     */
185    public Builder() {}
186
187    Builder(int size) {
188      super(size);
189    }
190
191    /**
192     * Associates {@code key} with {@code value} in the built bimap. Duplicate keys or values are
193     * not allowed, and will cause {@link #build} to fail.
194     */
195    @CanIgnoreReturnValue
196    @Override
197    public Builder<K, V> put(K key, V value) {
198      super.put(key, value);
199      return this;
200    }
201
202    /**
203     * Adds the given {@code entry} to the bimap. Duplicate keys or values are not allowed, and will
204     * cause {@link #build} to fail.
205     *
206     * @since 19.0
207     */
208    @CanIgnoreReturnValue
209    @Override
210    public Builder<K, V> put(Entry<? extends K, ? extends V> entry) {
211      super.put(entry);
212      return this;
213    }
214
215    /**
216     * Associates all of the given map's keys and values in the built bimap. Duplicate keys or
217     * values are not allowed, and will cause {@link #build} to fail.
218     *
219     * @throws NullPointerException if any key or value in {@code map} is null
220     */
221    @CanIgnoreReturnValue
222    @Override
223    public Builder<K, V> putAll(Map<? extends K, ? extends V> map) {
224      super.putAll(map);
225      return this;
226    }
227
228    /**
229     * Adds all of the given entries to the built bimap. Duplicate keys or values are not allowed,
230     * and will cause {@link #build} to fail.
231     *
232     * @throws NullPointerException if any key, value, or entry is null
233     * @since 19.0
234     */
235    @CanIgnoreReturnValue
236    @Beta
237    @Override
238    public Builder<K, V> putAll(Iterable<? extends Entry<? extends K, ? extends V>> entries) {
239      super.putAll(entries);
240      return this;
241    }
242
243    /**
244     * Configures this {@code Builder} to order entries by value according to the specified
245     * comparator.
246     *
247     * <p>The sort order is stable, that is, if two entries have values that compare as equivalent,
248     * the entry that was inserted first will be first in the built map's iteration order.
249     *
250     * @throws IllegalStateException if this method was already called
251     * @since 19.0
252     */
253    @CanIgnoreReturnValue
254    @Beta
255    @Override
256    public Builder<K, V> orderEntriesByValue(Comparator<? super V> valueComparator) {
257      super.orderEntriesByValue(valueComparator);
258      return this;
259    }
260
261    @Override
262    @CanIgnoreReturnValue
263    Builder<K, V> combine(ImmutableMap.Builder<K, V> builder) {
264      super.combine(builder);
265      return this;
266    }
267
268    /**
269     * Returns a newly-created immutable bimap. The iteration order of the returned bimap is the
270     * order in which entries were inserted into the builder, unless {@link #orderEntriesByValue}
271     * was called, in which case entries are sorted by value.
272     *
273     * @throws IllegalArgumentException if duplicate keys or values were added
274     */
275    @Override
276    public ImmutableBiMap<K, V> build() {
277      switch (size) {
278        case 0:
279          return of();
280        case 1:
281          // requireNonNull is safe because the first `size` elements have been filled in.
282          Entry<K, V> onlyEntry = requireNonNull(entries[0]);
283          return of(onlyEntry.getKey(), onlyEntry.getValue());
284        default:
285          /*
286           * If entries is full, or if hash flooding is detected, then this implementation may end
287           * up using the entries array directly and writing over the entry objects with
288           * non-terminal entries, but this is safe; if this Builder is used further, it will grow
289           * the entries array (so it can't affect the original array), and future build() calls
290           * will always copy any entry objects that cannot be safely reused.
291           */
292          if (valueComparator != null) {
293            if (entriesUsed) {
294              entries = Arrays.copyOf(entries, size);
295            }
296            Arrays.sort(
297                entries,
298                0,
299                size,
300                Ordering.from(valueComparator).onResultOf(Maps.<V>valueFunction()));
301          }
302          entriesUsed = true;
303          return RegularImmutableBiMap.fromEntryArray(size, entries);
304      }
305    }
306
307    @Override
308    @VisibleForTesting
309    ImmutableBiMap<K, V> buildJdkBacked() {
310      checkState(
311          valueComparator == null,
312          "buildJdkBacked is for tests only, doesn't support orderEntriesByValue");
313      switch (size) {
314        case 0:
315          return of();
316        case 1:
317          // requireNonNull is safe because the first `size` elements have been filled in.
318          Entry<K, V> onlyEntry = requireNonNull(entries[0]);
319          return of(onlyEntry.getKey(), onlyEntry.getValue());
320        default:
321          entriesUsed = true;
322          return RegularImmutableBiMap.fromEntryArray(size, entries);
323      }
324    }
325  }

327  /**
328   * Returns an immutable bimap containing the same entries as {@code map}. If {@code map} somehow
329   * contains entries with duplicate keys (for example, if it is a {@code SortedMap} whose
330   * comparator is not <i>consistent with equals</i>), the results of this method are undefined.
331   *
332   * <p>The returned {@code BiMap} iterates over entries in the same order as the {@code entrySet}
333   * of the original map.
334   *
335   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
336   * safe to do so. The exact circumstances under which a copy will or will not be performed are
337   * undocumented and subject to change.
338   *
339   * @throws IllegalArgumentException if two keys have the same value or two values have the same
340   *     key
341   * @throws NullPointerException if any key or value in {@code map} is null
342   */
343  public static <K, V> ImmutableBiMap<K, V> copyOf(Map<? extends K, ? extends V> map) {
344    if (map instanceof ImmutableBiMap) {
345      @SuppressWarnings("unchecked") // safe since map is not writable
346      ImmutableBiMap<K, V> bimap = (ImmutableBiMap<K, V>) map;
347      // TODO(lowasser): if we need to make a copy of a BiMap because the
348      // forward map is a view, don't make a copy of the non-view delegate map
349      if (!bimap.isPartialView()) {
350        return bimap;
351      }
352    }
353    return copyOf(map.entrySet());
354  }
355
356  /**
357   * Returns an immutable bimap containing the given entries. The returned bimap iterates over
358   * entries in the same order as the original iterable.
359   *
360   * @throws IllegalArgumentException if two keys have the same value or two values have the same
361   *     key
362   * @throws NullPointerException if any key, value, or entry is null
363   * @since 19.0
364   */
365  @Beta
366  public static <K, V> ImmutableBiMap<K, V> copyOf(
367      Iterable<? extends Entry<? extends K, ? extends V>> entries) {
368    @SuppressWarnings("unchecked") // we'll only be using getKey and getValue, which are covariant
369    Entry<K, V>[] entryArray = (Entry<K, V>[]) Iterables.toArray(entries, EMPTY_ENTRY_ARRAY);
370    switch (entryArray.length) {
371      case 0:
372        return of();
373      case 1:
374        Entry<K, V> entry = entryArray[0];
375        return of(entry.getKey(), entry.getValue());
376      default:
377        /*
378         * The current implementation will end up using entryArray directly, though it will write
379         * over the (arbitrary, potentially mutable) Entry objects actually stored in entryArray.
380         */
381        return RegularImmutableBiMap.fromEntries(entryArray);
382    }
383  }

424  /**
425   * Serialized type for all ImmutableBiMap instances. It captures the logical contents and they are
426   * reconstructed using public factory methods. This ensures that the implementation types remain
427   * as implementation details.
428   *
429   * <p>Since the bimap is immutable, ImmutableBiMap doesn't require special logic for keeping the
430   * bimap and its inverse in sync during serialization, the way AbstractBiMap does.
431   */
432  private static class SerializedForm<K, V> extends ImmutableMap.SerializedForm<K, V> {
433    SerializedForm(ImmutableBiMap<K, V> bimap) {
434      super(bimap);
435    }
436
437    @Override
438    Builder<K, V> makeBuilder(int size) {
439      return new Builder<>(size);
440    }
441
442    private static final long serialVersionUID = 0;
443  }
444
445  @Override
446  Object writeReplace() {
447    return new SerializedForm<>(this);
448  }
