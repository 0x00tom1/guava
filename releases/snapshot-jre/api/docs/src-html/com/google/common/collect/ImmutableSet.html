131  /**
132   * Returns an immutable set containing the given elements, minus duplicates, in the order each was
133   * first specified. That is, if multiple elements are {@linkplain Object#equals equal}, all except
134   * the first are ignored.
135   *
136   * <p>The array {@code others} must not be longer than {@code Integer.MAX_VALUE - 6}.
137   *
138   * @since 3.0 (source-compatible since 2.0)
139   */
140  @SafeVarargs // For Eclipse. For internal javac we have disabled this pointless type of warning.
141  public static <E> ImmutableSet<E> of(E e1, E e2, E e3, E e4, E e5, E e6, E... others) {
142    checkArgument(
143        others.length <= Integer.MAX_VALUE - 6, "the total number of elements must fit in an int");
144    final int paramCount = 6;
145    Object[] elements = new Object[paramCount + others.length];
146    elements[0] = e1;
147    elements[1] = e2;
148    elements[2] = e3;
149    elements[3] = e4;
150    elements[4] = e5;
151    elements[5] = e6;
152    System.arraycopy(others, 0, elements, paramCount, others.length);
153    return construct(elements.length, elements.length, elements);
154  }

156  /**
157   * Constructs an {@code ImmutableSet} from the first {@code n} elements of the specified array,
158   * which we have no particular reason to believe does or does not contain duplicates. If {@code k}
159   * is the size of the returned {@code ImmutableSet}, then the unique elements of {@code elements}
160   * will be in the first {@code k} positions, and {@code elements[i] == null} for {@code k <= i <
161   * n}.
162   *
163   * <p>This may modify {@code elements}. Additionally, if {@code n == elements.length} and {@code
164   * elements} contains no duplicates, {@code elements} may be used without copying in the returned
165   * {@code ImmutableSet}, in which case the caller must not modify it.
166   *
167   * <p>{@code elements} may contain only values of type {@code E}.
168   *
169   * @throws NullPointerException if any of the first {@code n} elements of {@code elements} is null
170   */
171  private static <E> ImmutableSet<E> constructUnknownDuplication(int n, Object... elements) {
172    // Guess the size is "halfway between" all duplicates and no duplicates, on a log scale.
173    return construct(
174        n,
175        Math.max(
176            ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY,
177            IntMath.sqrt(n, RoundingMode.CEILING)),
178        elements);
179  }

181  /**
182   * Constructs an {@code ImmutableSet} from the first {@code n} elements of the specified array. If
183   * {@code k} is the size of the returned {@code ImmutableSet}, then the unique elements of {@code
184   * elements} will be in the first {@code k} positions, and {@code elements[i] == null} for {@code
185   * k <= i < n}.
186   *
187   * <p>This may modify {@code elements}. Additionally, if {@code n == elements.length} and {@code
188   * elements} contains no duplicates, {@code elements} may be used without copying in the returned
189   * {@code ImmutableSet}, in which case it may no longer be modified.
190   *
191   * <p>{@code elements} may contain only values of type {@code E}.
192   *
193   * @throws NullPointerException if any of the first {@code n} elements of {@code elements} is null
194   */
195  private static <E> ImmutableSet<E> construct(int n, int expectedSize, Object... elements) {
196    switch (n) {
197      case 0:
198        return of();
199      case 1:
200        @SuppressWarnings("unchecked") // safe; elements contains only E's
201        E elem = (E) elements[0];
202        return of(elem);
203      default:
204        SetBuilderImpl<E> builder = new RegularSetBuilderImpl<E>(expectedSize);
205        for (int i = 0; i < n; i++) {
206          @SuppressWarnings("unchecked")
207          E e = (E) checkNotNull(elements[i]);
208          builder = builder.add(e);
209        }
210        return builder.review().build();
211    }
212  }

214  /**
215   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
216   * each appears first in the source collection.
217   *
218   * <p><b>Performance note:</b> This method will sometimes recognize that the actual copy operation
219   * is unnecessary; for example, {@code copyOf(copyOf(anArrayList))} will copy the data only once.
220   * This reduces the expense of habitually making defensive copies at API boundaries. However, the
221   * precise conditions for skipping the copy operation are undefined.
222   *
223   * @throws NullPointerException if any of {@code elements} is null
224   * @since 7.0 (source-compatible since 2.0)
225   */
226  public static <E> ImmutableSet<E> copyOf(Collection<? extends E> elements) {
227    /*
228     * TODO(lowasser): consider checking for ImmutableAsList here
229     * TODO(lowasser): consider checking for Multiset here
230     */
231    // Don't refer to ImmutableSortedSet by name so it won't pull in all that code
232    if (elements instanceof ImmutableSet && !(elements instanceof SortedSet)) {
233      @SuppressWarnings("unchecked") // all supported methods are covariant
234      ImmutableSet<E> set = (ImmutableSet<E>) elements;
235      if (!set.isPartialView()) {
236        return set;
237      }
238    } else if (elements instanceof EnumSet) {
239      return copyOfEnumSet((EnumSet) elements);
240    }
241    Object[] array = elements.toArray();
242    if (elements instanceof Set) {
243      // assume probably no duplicates (though it might be using different equality semantics)
244      return construct(array.length, array.length, array);
245    } else {
246      return constructUnknownDuplication(array.length, array);
247    }
248  }

250  /**
251   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
252   * each appears first in the source iterable. This method iterates over {@code elements} only
253   * once.
254   *
255   * <p><b>Performance note:</b> This method will sometimes recognize that the actual copy operation
256   * is unnecessary; for example, {@code copyOf(copyOf(anArrayList))} should copy the data only
257   * once. This reduces the expense of habitually making defensive copies at API boundaries.
258   * However, the precise conditions for skipping the copy operation are undefined.
259   *
260   * @throws NullPointerException if any of {@code elements} is null
261   */
262  public static <E> ImmutableSet<E> copyOf(Iterable<? extends E> elements) {
263    return (elements instanceof Collection)
264        ? copyOf((Collection<? extends E>) elements)
265        : copyOf(elements.iterator());
266  }

268  /**
269   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
270   * each appears first in the source iterator.
271   *
272   * @throws NullPointerException if any of {@code elements} is null
273   */
274  public static <E> ImmutableSet<E> copyOf(Iterator<? extends E> elements) {
275    // We special-case for 0 or 1 elements, but anything further is madness.
276    if (!elements.hasNext()) {
277      return of();
278    }
279    E first = elements.next();
280    if (!elements.hasNext()) {
281      return of(first);
282    } else {
283      return new ImmutableSet.Builder<E>().add(first).addAll(elements).build();
284    }
285  }

287  /**
288   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
289   * each appears first in the source array.
290   *
291   * @throws NullPointerException if any of {@code elements} is null
292   * @since 3.0
293   */
294  public static <E> ImmutableSet<E> copyOf(E[] elements) {
295    switch (elements.length) {
296      case 0:
297        return of();
298      case 1:
299        return of(elements[0]);
300      default:
301        return constructUnknownDuplication(elements.length, elements.clone());
302    }
303  }

317  @Override
318  public boolean equals(@CheckForNull Object object) {
319    if (object == this) {
320      return true;
321    }
322    if (object instanceof ImmutableSet
323        && isHashCodeFast()
324        && ((ImmutableSet<?>) object).isHashCodeFast()
325        && hashCode() != object.hashCode()) {
326      return false;
327    }
328    return Sets.equalsImpl(this, object);
329  }

462  /**
463   * A builder for creating {@code ImmutableSet} instances. Example:
464   *
465   * <pre>{@code
466   * static final ImmutableSet<Color> GOOGLE_COLORS =
467   *     ImmutableSet.<Color>builder()
468   *         .addAll(WEBSAFE_COLORS)
469   *         .add(new Color(0, 191, 255))
470   *         .build();
471   * }</pre>
472   *
473   * <p>Elements appear in the resulting set in the same order they were first added to the builder.
474   *
475   * <p>Building does not change the state of the builder, so it is still possible to add more
476   * elements and to build again.
477   *
478   * @since 2.0
479   */
480  public static class Builder<E> extends ImmutableCollection.Builder<E> {
481    /*
482     * `impl` is null only for instances of the subclass, ImmutableSortedSet.Builder. That subclass
483     * overrides all the methods that access it here. Thus, all the methods here can safely assume
484     * that this field is non-null.
485     */
486    @CheckForNull private SetBuilderImpl<E> impl;
487    boolean forceCopy;
488
489    public Builder() {
490      this(0);
491    }
492
493    Builder(int capacity) {
494      if (capacity > 0) {
495        impl = new RegularSetBuilderImpl<E>(capacity);
496      } else {
497        impl = EmptySetBuilderImpl.instance();
498      }
499    }
500
501    Builder(@SuppressWarnings("unused") boolean subclass) {
502      this.impl = null; // unused
503    }
504
505    @VisibleForTesting
506    void forceJdk() {
507      requireNonNull(impl); // see the comment on the field
508      this.impl = new JdkBackedSetBuilderImpl<E>(impl);
509    }
510
511    final void copyIfNecessary() {
512      if (forceCopy) {
513        copy();
514        forceCopy = false;
515      }
516    }
517
518    void copy() {
519      requireNonNull(impl); // see the comment on the field
520      impl = impl.copy();
521    }
522
523    @Override
524    @CanIgnoreReturnValue
525    public Builder<E> add(E element) {
526      requireNonNull(impl); // see the comment on the field
527      checkNotNull(element);
528      copyIfNecessary();
529      impl = impl.add(element);
530      return this;
531    }
532
533    @Override
534    @CanIgnoreReturnValue
535    public Builder<E> add(E... elements) {
536      super.add(elements);
537      return this;
538    }
539
540    /**
541     * Adds each element of {@code elements} to the {@code ImmutableSet}, ignoring duplicate
542     * elements (only the first duplicate element is added).
543     *
544     * @param elements the elements to add
545     * @return this {@code Builder} object
546     * @throws NullPointerException if {@code elements} is null or contains a null element
547     */
548    @Override
549    @CanIgnoreReturnValue
550    public Builder<E> addAll(Iterable<? extends E> elements) {
551      super.addAll(elements);
552      return this;
553    }
554
555    @Override
556    @CanIgnoreReturnValue
557    public Builder<E> addAll(Iterator<? extends E> elements) {
558      super.addAll(elements);
559      return this;
560    }
561
562    @CanIgnoreReturnValue
563    Builder<E> combine(Builder<E> other) {
564      requireNonNull(impl);
565      requireNonNull(other.impl);
566      /*
567       * For discussion of requireNonNull, see the comment on the field.
568       *
569       * (And I don't believe there's any situation in which we call x.combine(y) when x is a plain
570       * ImmutableSet.Builder but y is an ImmutableSortedSet.Builder (or vice versa). Certainly
571       * ImmutableSortedSet.Builder.combine() is written as if its argument will never be a plain
572       * ImmutableSet.Builder: It casts immediately to ImmutableSortedSet.Builder.)
573       */
574      copyIfNecessary();
575      this.impl = this.impl.combine(other.impl);
576      return this;
577    }
578
579    @Override
580    public ImmutableSet<E> build() {
581      requireNonNull(impl); // see the comment on the field
582      forceCopy = true;
583      impl = impl.review();
584      return impl.build();
585    }
586  }

588  /** Swappable internal implementation of an ImmutableSet.Builder. */
589  private abstract static class SetBuilderImpl<E> {
590    // The first `distinct` elements are non-null.
591    // Since we can never access null elements, we don't mark this nullable.
592    E[] dedupedElements;
593    int distinct;
594
595    @SuppressWarnings("unchecked")
596    SetBuilderImpl(int expectedCapacity) {
597      this.dedupedElements = (E[]) new Object[expectedCapacity];
598      this.distinct = 0;
599    }
600
601    /** Initializes this SetBuilderImpl with a copy of the deduped elements array from toCopy. */
602    SetBuilderImpl(SetBuilderImpl<E> toCopy) {
603      this.dedupedElements = Arrays.copyOf(toCopy.dedupedElements, toCopy.dedupedElements.length);
604      this.distinct = toCopy.distinct;
605    }
606
607    /**
608     * Resizes internal data structures if necessary to store the specified number of distinct
609     * elements.
610     */
611    private void ensureCapacity(int minCapacity) {
612      if (minCapacity > dedupedElements.length) {
613        int newCapacity =
614            ImmutableCollection.Builder.expandedCapacity(dedupedElements.length, minCapacity);
615        dedupedElements = Arrays.copyOf(dedupedElements, newCapacity);
616      }
617    }
618
619    /** Adds e to the insertion-order array of deduplicated elements. Calls ensureCapacity. */
620    final void addDedupedElement(E e) {
621      ensureCapacity(distinct + 1);
622      dedupedElements[distinct++] = e;
623    }
624
625    /**
626     * Adds e to this SetBuilderImpl, returning the updated result. Only use the returned
627     * SetBuilderImpl, since we may switch implementations if e.g. hash flooding is detected.
628     */
629    abstract SetBuilderImpl<E> add(E e);
630
631    /** Adds all the elements from the specified SetBuilderImpl to this SetBuilderImpl. */
632    final SetBuilderImpl<E> combine(SetBuilderImpl<E> other) {
633      SetBuilderImpl<E> result = this;
634      for (int i = 0; i < other.distinct; i++) {
635        /*
636         * requireNonNull is safe because we ensure that the first `distinct` elements have been
637         * populated.
638         */
639        result = result.add(requireNonNull(other.dedupedElements[i]));
640      }
641      return result;
642    }
643
644    /**
645     * Creates a new copy of this SetBuilderImpl. Modifications to that SetBuilderImpl will not
646     * affect this SetBuilderImpl or sets constructed from this SetBuilderImpl via build().
647     */
648    abstract SetBuilderImpl<E> copy();
649
650    /**
651     * Call this before build(). Does a final check on the internal data structures, e.g. shrinking
652     * unnecessarily large structures or detecting previously unnoticed hash flooding.
653     */
654    SetBuilderImpl<E> review() {
655      return this;
656    }
657
658    abstract ImmutableSet<E> build();
659  }

689  // We use power-of-2 tables, and this is the highest int that's a power of 2
690  static final int MAX_TABLE_SIZE = Ints.MAX_POWER_OF_TWO;
691
692  // Represents how tightly we can pack things, as a maximum.
693  private static final double DESIRED_LOAD_FACTOR = 0.7;
694
695  // If the set has this many elements, it will "max out" the table size
696  private static final int CUTOFF = (int) (MAX_TABLE_SIZE * DESIRED_LOAD_FACTOR);
697
698  /**
699   * Returns an array size suitable for the backing array of a hash table that uses open addressing
700   * with linear probing in its implementation. The returned size is the smallest power of two that
701   * can hold setSize elements with the desired load factor. Always returns at least setSize + 2.
702   */
703  // TODO(cpovirk): Move to Hashing or something, since it's used elsewhere in the Android version.
704  static int chooseTableSize(int setSize) {
705    setSize = Math.max(setSize, 2);
706    // Correct the size for open addressing to match desired load factor.
707    if (setSize < CUTOFF) {
708      // Round up to the next highest power of 2.
709      int tableSize = Integer.highestOneBit(setSize - 1) << 1;
710      while (tableSize * DESIRED_LOAD_FACTOR < setSize) {
711        tableSize <<= 1;
712      }
713      return tableSize;
714    }
715
716    // The table can't be completely full or we'll get infinite reprobes
717    checkArgument(setSize < MAX_TABLE_SIZE, "collection too large");
718    return MAX_TABLE_SIZE;
719  }

721  /**
722   * Default implementation of the guts of ImmutableSet.Builder, creating an open-addressed hash
723   * table and deduplicating elements as they come, so it only allocates O(max(distinct,
724   * expectedCapacity)) rather than O(calls to add).
725   *
726   * <p>This implementation attempts to detect hash flooding, and if it's identified, falls back to
727   * JdkBackedSetBuilderImpl.
728   */
729  private static final class RegularSetBuilderImpl<E> extends SetBuilderImpl<E> {
730    // null until at least two elements are present
731    @CheckForNull private @Nullable Object[] hashTable;
732    private int maxRunBeforeFallback;
733    private int expandTableThreshold;
734    private int hashCode;
735
736    RegularSetBuilderImpl(int expectedCapacity) {
737      super(expectedCapacity);
738      this.hashTable = null;
739      this.maxRunBeforeFallback = 0;
740      this.expandTableThreshold = 0;
741    }
742
743    RegularSetBuilderImpl(RegularSetBuilderImpl<E> toCopy) {
744      super(toCopy);
745      this.hashTable = (toCopy.hashTable == null) ? null : toCopy.hashTable.clone();
746      this.maxRunBeforeFallback = toCopy.maxRunBeforeFallback;
747      this.expandTableThreshold = toCopy.expandTableThreshold;
748      this.hashCode = toCopy.hashCode;
749    }
750
751    @Override
752    SetBuilderImpl<E> add(E e) {
753      checkNotNull(e);
754      if (hashTable == null) {
755        if (distinct == 0) {
756          addDedupedElement(e);
757          return this;
758        } else {
759          ensureTableCapacity(dedupedElements.length);
760          E elem = dedupedElements[0];
761          distinct--;
762          return insertInHashTable(elem).add(e);
763        }
764      }
765      return insertInHashTable(e);
766    }
767
768    private SetBuilderImpl<E> insertInHashTable(E e) {
769      requireNonNull(hashTable);
770      int eHash = e.hashCode();
771      int i0 = Hashing.smear(eHash);
772      int mask = hashTable.length - 1;
773      for (int i = i0; i - i0 < maxRunBeforeFallback; i++) {
774        int index = i & mask;
775        Object tableEntry = hashTable[index];
776        if (tableEntry == null) {
777          addDedupedElement(e);
778          hashTable[index] = e;
779          hashCode += eHash;
780          ensureTableCapacity(distinct); // rebuilds table if necessary
781          return this;
782        } else if (tableEntry.equals(e)) { // not a new element, ignore
783          return this;
784        }
785      }
786      // we fell out of the loop due to a long run; fall back to JDK impl
787      return new JdkBackedSetBuilderImpl<E>(this).add(e);
788    }
789
790    @Override
791    SetBuilderImpl<E> copy() {
792      return new RegularSetBuilderImpl<E>(this);
793    }
794
795    @Override
796    SetBuilderImpl<E> review() {
797      if (hashTable == null) {
798        return this;
799      }
800      int targetTableSize = chooseTableSize(distinct);
801      if (targetTableSize * 2 < hashTable.length) {
802        hashTable = rebuildHashTable(targetTableSize, dedupedElements, distinct);
803        maxRunBeforeFallback = maxRunBeforeFallback(targetTableSize);
804        expandTableThreshold = (int) (DESIRED_LOAD_FACTOR * targetTableSize);
805      }
806      return hashFloodingDetected(hashTable) ? new JdkBackedSetBuilderImpl<E>(this) : this;
807    }
808
809    @Override
810    ImmutableSet<E> build() {
811      switch (distinct) {
812        case 0:
813          return of();
814        case 1:
815          /*
816           * requireNonNull is safe because we ensure that the first `distinct` elements have been
817           * populated.
818           */
819          return of(requireNonNull(dedupedElements[0]));
820        default:
821          /*
822           * The suppression is safe because we ensure that the first `distinct` elements have been
823           * populated.
824           */
825          @SuppressWarnings("nullness")
826          Object[] elements =
827              (distinct == dedupedElements.length)
828                  ? dedupedElements
829                  : Arrays.copyOf(dedupedElements, distinct);
830          return new RegularImmutableSet<E>(
831              elements, hashCode, requireNonNull(hashTable), hashTable.length - 1);
832      }
833    }
834
835    /** Builds a new open-addressed hash table from the first n objects in elements. */
836    static @Nullable Object[] rebuildHashTable(int newTableSize, Object[] elements, int n) {
837      @Nullable Object[] hashTable = new @Nullable Object[newTableSize];
838      int mask = hashTable.length - 1;
839      for (int i = 0; i < n; i++) {
840        // requireNonNull is safe because we ensure that the first n elements have been populated.
841        Object e = requireNonNull(elements[i]);
842        int j0 = Hashing.smear(e.hashCode());
843        for (int j = j0; ; j++) {
844          int index = j & mask;
845          if (hashTable[index] == null) {
846            hashTable[index] = e;
847            break;
848          }
849        }
850      }
851      return hashTable;
852    }
853
854    void ensureTableCapacity(int minCapacity) {
855      int newTableSize;
856      if (hashTable == null) {
857        newTableSize = chooseTableSize(minCapacity);
858        hashTable = new Object[newTableSize];
859      } else if (minCapacity > expandTableThreshold && hashTable.length < MAX_TABLE_SIZE) {
860        newTableSize = hashTable.length * 2;
861        hashTable = rebuildHashTable(newTableSize, dedupedElements, distinct);
862      } else {
863        return;
864      }
865      maxRunBeforeFallback = maxRunBeforeFallback(newTableSize);
866      expandTableThreshold = (int) (DESIRED_LOAD_FACTOR * newTableSize);
867    }
868
869    /**
870     * We attempt to detect deliberate hash flooding attempts. If one is detected, we fall back to a
871     * wrapper around j.u.HashSet, which has built-in flooding protection. MAX_RUN_MULTIPLIER was
872     * determined experimentally to match our desired probability of false positives.
873     */
874    // NB: yes, this is surprisingly high, but that's what the experiments said was necessary
875    // Raising this number slows the worst-case contains behavior, speeds up hashFloodingDetected,
876    // and reduces the false-positive probability.
877    static final int MAX_RUN_MULTIPLIER = 13;
878
879    /**
880     * Checks the whole hash table for poor hash distribution. Takes O(n) in the worst case, O(n /
881     * log n) on average.
882     *
883     * <p>The online hash flooding detecting in RegularSetBuilderImpl.add can detect e.g. many
884     * exactly matching hash codes, which would cause construction to take O(n^2), but can't detect
885     * e.g. hash codes adversarially designed to go into ascending table locations, which keeps
886     * construction O(n) (as desired) but then can have O(n) queries later.
887     *
888     * <p>If this returns false, then no query can take more than O(log n).
889     *
890     * <p>Note that for a RegularImmutableSet with elements with truly random hash codes, contains
891     * operations take expected O(1) time but with high probability take O(log n) for at least some
892     * element. (https://en.wikipedia.org/wiki/Linear_probing#Analysis)
893     *
894     * <p>This method may return {@code true} even on truly random input, but {@code
895     * ImmutableSetTest} tests that the probability of that is low.
896     */
897    static boolean hashFloodingDetected(@Nullable Object[] hashTable) {
898      int maxRunBeforeFallback = maxRunBeforeFallback(hashTable.length);
899      int mask = hashTable.length - 1;
900
901      // Invariant: all elements at indices in [knownRunStart, knownRunEnd) are nonnull.
902      // If knownRunStart == knownRunEnd, this is vacuously true.
903      // When knownRunEnd exceeds hashTable.length, it "wraps", detecting runs around the end
904      // of the table.
905      int knownRunStart = 0;
906      int knownRunEnd = 0;
907
908      outerLoop:
909      while (knownRunStart < hashTable.length) {
910        if (knownRunStart == knownRunEnd && hashTable[knownRunStart] == null) {
911          if (hashTable[(knownRunStart + maxRunBeforeFallback - 1) & mask] == null) {
912            // There are only maxRunBeforeFallback - 1 elements between here and there,
913            // so even if they were all nonnull, we wouldn't detect a hash flood.  Therefore,
914            // we can skip them all.
915            knownRunStart += maxRunBeforeFallback;
916          } else {
917            knownRunStart++; // the only case in which maxRunEnd doesn't increase by mRBF
918            // happens about f * (1-f) for f = DESIRED_LOAD_FACTOR, so around 21% of the time
919          }
920          knownRunEnd = knownRunStart;
921        } else {
922          for (int j = knownRunStart + maxRunBeforeFallback - 1; j >= knownRunEnd; j--) {
923            if (hashTable[j & mask] == null) {
924              knownRunEnd = knownRunStart + maxRunBeforeFallback;
925              knownRunStart = j + 1;
926              continue outerLoop;
927            }
928          }
929          return true;
930        }
931      }
932      return false;
933    }
934
935    /**
936     * If more than this many consecutive positions are filled in a table of the specified size,
937     * report probable hash flooding. ({@link #hashFloodingDetected} may also report hash flooding
938     * if fewer consecutive positions are filled; see that method for details.)
939     */
940    static int maxRunBeforeFallback(int tableSize) {
941      return MAX_RUN_MULTIPLIER * IntMath.log2(tableSize, RoundingMode.UNNECESSARY);
942    }
943  }

945  /**
946   * SetBuilderImpl version that uses a JDK HashSet, which has built in hash flooding protection.
947   */
948  private static final class JdkBackedSetBuilderImpl<E> extends SetBuilderImpl<E> {
949    private final Set<Object> delegate;
950
951    JdkBackedSetBuilderImpl(SetBuilderImpl<E> toCopy) {
952      super(toCopy); // initializes dedupedElements and distinct
953      delegate = Sets.newHashSetWithExpectedSize(distinct);
954      for (int i = 0; i < distinct; i++) {
955        /*
956         * requireNonNull is safe because we ensure that the first `distinct` elements have been
957         * populated.
958         */
959        delegate.add(requireNonNull(dedupedElements[i]));
960      }
961    }
962
963    @Override
964    SetBuilderImpl<E> add(E e) {
965      checkNotNull(e);
966      if (delegate.add(e)) {
967        addDedupedElement(e);
968      }
969      return this;
970    }
971
972    @Override
973    SetBuilderImpl<E> copy() {
974      return new JdkBackedSetBuilderImpl<>(this);
975    }
976
977    @Override
978    ImmutableSet<E> build() {
979      switch (distinct) {
980        case 0:
981          return of();
982        case 1:
983          /*
984           * requireNonNull is safe because we ensure that the first `distinct` elements have been
985           * populated.
986           */
987          return of(requireNonNull(dedupedElements[0]));
988        default:
989          return new JdkBackedImmutableSet<E>(
990              delegate, ImmutableList.asImmutableList(dedupedElements, distinct));
991      }
992    }
