032/**
033 * Serializes execution of a set of operations. This class guarantees that a submitted callable will
034 * not be called before previously submitted callables (and any {@code Future}s returned from them)
035 * have completed.
036 *
037 * <p>This class implements a superset of the behavior of {@link
038 * MoreExecutors#newSequentialExecutor}. If your tasks all run on the same underlying executor and
039 * don't need to wait for {@code Future}s returned from {@code AsyncCallable}s, use it instead.
040 *
041 * @since 26.0
042 */
043@Beta
044public final class ExecutionSequencer {
045
046  private ExecutionSequencer() {}
047
048  /** Creates a new instance. */
049  public static ExecutionSequencer create() {
050    return new ExecutionSequencer();
051  }
052
053  /** This reference acts as a pointer tracking the head of a linked list of ListenableFutures. */
054  private final AtomicReference<ListenableFuture<Void>> ref =
055      new AtomicReference<>(immediateVoidFuture());
056
057  private ThreadConfinedTaskQueue latestTaskQueue = new ThreadConfinedTaskQueue();
058
059  /**
060   * This object is unsafely published, but avoids problematic races by relying exclusively on the
061   * identity equality of its Thread field so that the task field is only accessed by a single
062   * thread.
063   */
064  private static final class ThreadConfinedTaskQueue {
065    /**
066     * This field is only used for identity comparisons with the current thread. Field assignments
067     * are atomic, but do not provide happens-before ordering; however:
068     *
069     * <ul>
070     *   <li>If this field's value == currentThread, we know that it's up to date, because write
071     *       operations in a thread always happen-before subsequent read operations in the same
072     *       thread
073     *   <li>If this field's value == null because of unsafe publication, we know that it isn't the
074     *       object associated with our thread, because if it was the publication wouldn't have been
075     *       unsafe and we'd have seen our thread as the value. This state is also why a new
076     *       ThreadConfinedTaskQueue object must be created for each inline execution, because
077     *       observing a null thread does not mean the object is safe to reuse.
078     *   <li>If this field's value is some other thread object, we know that it's not our thread.
079     *   <li>If this field's value == null because it originally belonged to another thread and that
080     *       thread cleared it, we still know that it's not associated with our thread
081     *   <li>If this field's value == null because it was associated with our thread and was
082     *       cleared, we know that we're not executing inline any more
083     * </ul>
084     *
085     * All the states where thread != currentThread are identical for our purposes, and so even
086     * though it's racy, we don't care which of those values we get, so no need to synchronize.
087     */
088    Thread thread;
089    /** Only used by the thread associated with this object */
090    Runnable nextTask;
091    /** Only used by the thread associated with this object */
092    Executor nextExecutor;
093  }
094
095  /**
096   * Enqueues a task to run when the previous task (if any) completes.
097   *
098   * <p>Cancellation does not propagate from the output future to a callable that has begun to
099   * execute, but if the output future is cancelled before {@link Callable#call()} is invoked,
100   * {@link Callable#call()} will not be invoked.
101   */
102  public <T> ListenableFuture<T> submit(final Callable<T> callable, Executor executor) {
103    checkNotNull(callable);
104    checkNotNull(executor);
105    return submitAsync(
106        new AsyncCallable<T>() {
107          @Override
108          public ListenableFuture<T> call() throws Exception {
109            return immediateFuture(callable.call());
110          }
111
112          @Override
113          public String toString() {
114            return callable.toString();
115          }
116        },
117        executor);
118  }
119
120  /**
121   * Enqueues a task to run when the previous task (if any) completes.
122   *
123   * <p>Cancellation does not propagate from the output future to the future returned from {@code
124   * callable} or a callable that has begun to execute, but if the output future is cancelled before
125   * {@link AsyncCallable#call()} is invoked, {@link AsyncCallable#call()} will not be invoked.
126   */
127  public <T> ListenableFuture<T> submitAsync(
128      final AsyncCallable<T> callable, final Executor executor) {
129    checkNotNull(callable);
130    checkNotNull(executor);
131    final TaskNonReentrantExecutor taskExecutor = new TaskNonReentrantExecutor(executor, this);
132    final AsyncCallable<T> task =
133        new AsyncCallable<T>() {
134          @Override
135          public ListenableFuture<T> call() throws Exception {
136            if (!taskExecutor.trySetStarted()) {
137              return immediateCancelledFuture();
138            }
139            return callable.call();
140          }
141
142          @Override
143          public String toString() {
144            return callable.toString();
145          }
146        };
147    /*
148     * Four futures are at play here:
149     * taskFuture is the future tracking the result of the callable.
150     * newFuture is a future that completes after this and all prior tasks are done.
151     * oldFuture is the previous task's newFuture.
152     * outputFuture is the future we return to the caller, a nonCancellationPropagating taskFuture.
153     *
154     * newFuture is guaranteed to only complete once all tasks previously submitted to this instance
155     * have completed - namely after oldFuture is done, and taskFuture has either completed or been
156     * cancelled before the callable started execution.
157     */
158    final SettableFuture<Void> newFuture = SettableFuture.create();
159
160    final ListenableFuture<Void> oldFuture = ref.getAndSet(newFuture);
161
162    // Invoke our task once the previous future completes.
163    final TrustedListenableFutureTask<T> taskFuture = TrustedListenableFutureTask.create(task);
164    oldFuture.addListener(taskFuture, taskExecutor);
165
166    final ListenableFuture<T> outputFuture = Futures.nonCancellationPropagating(taskFuture);
167
168    // newFuture's lifetime is determined by taskFuture, which can't complete before oldFuture
169    // unless taskFuture is cancelled, in which case it falls back to oldFuture. This ensures that
170    // if the future we return is cancelled, we don't begin execution of the next task until after
171    // oldFuture completes.
172    Runnable listener =
173        new Runnable() {
174          @Override
175          public void run() {
176            if (taskFuture.isDone()) {
177              // Since the value of oldFuture can only ever be immediateFuture(null) or setFuture of
178              // a future that eventually came from immediateFuture(null), this doesn't leak
179              // throwables or completion values.
180              newFuture.setFuture(oldFuture);
181            } else if (outputFuture.isCancelled() && taskExecutor.trySetCancelled()) {
182              // If this CAS succeeds, we know that the provided callable will never be invoked,
183              // so when oldFuture completes it is safe to allow the next submitted task to
184              // proceed. Doing this immediately here lets the next task run without waiting for
185              // the cancelled task's executor to run the noop AsyncCallable.
186              taskFuture.cancel(false);
187            }
188          }
189        };
190    // Adding the listener to both futures guarantees that newFuture will aways be set. Adding to
191    // taskFuture guarantees completion if the callable is invoked, and adding to outputFuture
192    // propagates cancellation if the callable has not yet been invoked.
193    outputFuture.addListener(listener, directExecutor());
194    taskFuture.addListener(listener, directExecutor());
195
196    return outputFuture;
197  }
198
199  enum RunningState {
200    NOT_RUN,
201    CANCELLED,
202    STARTED,
203  }
204
205  /**
206   * This class helps avoid a StackOverflowError when large numbers of tasks are submitted with
207   * {@link MoreExecutors#directExecutor}. Normally, when the first future completes, all the other
208   * tasks would be called recursively. Here, we detect that the delegate executor is executing
209   * inline, and maintain a queue to dispatch tasks iteratively. There is one instance of this class
210   * per call to submit() or submitAsync(), and each instance supports only one call to execute().
211   *
212   * <p>This class would certainly be simpler and easier to reason about if it were built with
213   * ThreadLocal; however, ThreadLocal is not well optimized for the case where the ThreadLocal is
214   * non-static, and is initialized/removed frequently - this causes churn in the Thread specific
215   * hashmaps. Using a static ThreadLocal to avoid that overhead would mean that different
216   * ExecutionSequencer objects interfere with each other, which would be undesirable, in addition
217   * to increasing the memory footprint of every thread that interacted with it. In order to release
218   * entries in thread-specific maps when the ThreadLocal object itself is no longer referenced,
219   * ThreadLocal is usually implemented with a WeakReference, which can have negative performance
220   * properties; for example, calling WeakReference.get() on Android will block during an
221   * otherwise-concurrent GC cycle.
222   */
223  @SuppressWarnings("ShouldNotSubclass") // Saving an allocation here is worth it
224  private static final class TaskNonReentrantExecutor extends AtomicReference<RunningState>
225      implements Executor, Runnable {
226
227    /**
228     * Used to update and read the latestTaskQueue field. Set to null once the runnable has been run
229     * or queued.
230     */
231    ExecutionSequencer sequencer;
232
233    /**
234     * Executor the task was set to run on. Set to null when the task has been queued, run, or
235     * cancelled.
236     */
237    Executor delegate;
238
239    /**
240     * Set before calling delegate.execute(); set to null once run, so that it can be GCed; this
241     * object may live on after, if submitAsync returns an incomplete future.
242     */
243    Runnable task;
244
245    /** Thread that called execute(). Set in execute, cleared when delegate.execute() returns. */
246    Thread submitting;
247
248    private TaskNonReentrantExecutor(Executor delegate, ExecutionSequencer sequencer) {
249      super(NOT_RUN);
250      this.delegate = delegate;
251      this.sequencer = sequencer;
252    }
253
254    @Override
255    public void execute(Runnable task) {
256      // If this operation was successfully cancelled already, calling the runnable will be a noop.
257      // This also avoids a race where if outputFuture is cancelled, it will call taskFuture.cancel,
258      // which will call newFuture.setFuture(oldFuture), to allow the next task in the queue to run
259      // without waiting for the user's executor to run our submitted Runnable. However, this can
260      // interact poorly with the reentrancy-avoiding behavior of this executor - when the operation
261      // before the cancelled future completes, it will synchronously complete both the newFuture
262      // from the cancelled operation and its own. This can cause one runnable to queue two tasks,
263      // breaking the invariant this method relies on to iteratively run the next task after the
264      // previous one completes.
265      if (get() == RunningState.CANCELLED) {
266        delegate = null;
267        sequencer = null;
268        return;
269      }
270      submitting = Thread.currentThread();
271      try {
272        ThreadConfinedTaskQueue submittingTaskQueue = sequencer.latestTaskQueue;
273        if (submittingTaskQueue.thread == submitting) {
274          sequencer = null;
275          // Submit from inside a reentrant submit. We don't know if this one will be reentrant (and
276          // can't know without submitting something to the executor) so queue to run iteratively.
277          // Task must be null, since each execution on this executor can only produce one more
278          // execution.
279          checkState(submittingTaskQueue.nextTask == null);
280          submittingTaskQueue.nextTask = task;
281          submittingTaskQueue.nextExecutor = delegate;
282          delegate = null;
283        } else {
284          Executor localDelegate = delegate;
285          delegate = null;
286          this.task = task;
287          localDelegate.execute(this);
288        }
289      } finally {
290        // Important to null this out here - if we did *not* execute inline, we might still
291        // run() on the same thread that called execute() - such as in a thread pool, and think
292        // that it was happening inline. As a side benefit, avoids holding on to the Thread object
293        // longer than necessary.
294        submitting = null;
295      }
296    }
297
298    @SuppressWarnings("ShortCircuitBoolean")
299    @Override
300    public void run() {
301      Thread currentThread = Thread.currentThread();
302      if (currentThread != submitting) {
303        Runnable localTask = task;
304        task = null;
305        localTask.run();
306        return;
307      }
308      // Executor called reentrantly! Make sure that further calls don't overflow stack. Further
309      // reentrant calls will see that their current thread is the same as the one set in
310      // latestTaskQueue, and queue rather than calling execute() directly.
311      ThreadConfinedTaskQueue executingTaskQueue = new ThreadConfinedTaskQueue();
312      executingTaskQueue.thread = currentThread;
313      // Unconditionally set; there is no risk of throwing away a queued task from another thread,
314      // because in order for the current task to run on this executor the previous task must have
315      // already started execution. Because each task on a TaskNonReentrantExecutor can only produce
316      // one execute() call to another instance from the same ExecutionSequencer, we know by
317      // induction that the task that launched this one must not have added any other runnables to
318      // that thread's queue, and thus we cannot be replacing a TaskAndThread object that would
319      // otherwise have another task queued on to it. Note the exception to this, cancellation, is
320      // specially handled in execute() - execute() calls triggered by cancellation are no-ops, and
321      // thus don't count.
322      sequencer.latestTaskQueue = executingTaskQueue;
323      sequencer = null;
324      try {
325        Runnable localTask = task;
326        task = null;
327        localTask.run();
328        // Now check if our task attempted to reentrantly execute the next task.
329        Runnable queuedTask;
330        Executor queuedExecutor;
331        // Intentionally using non-short-circuit operator
332        while ((queuedTask = executingTaskQueue.nextTask) != null
333            & (queuedExecutor = executingTaskQueue.nextExecutor) != null) {
334          executingTaskQueue.nextTask = null;
335          executingTaskQueue.nextExecutor = null;
336          queuedExecutor.execute(queuedTask);
337        }
338      } finally {
339        // Null out the thread field, so that we don't leak a reference to Thread, and so that
340        // future `thread == currentThread()` calls from this thread don't incorrectly queue instead
341        // of executing. Don't null out the latestTaskQueue field, because the work done here
342        // may have scheduled more operations on another thread, and if those operations then
343        // trigger reentrant calls that thread will have updated the latestTaskQueue field, and
344        // we'd be interfering with their operation.
345        executingTaskQueue.thread = null;
346      }
347    }
348
349    private boolean trySetStarted() {
350      return compareAndSet(NOT_RUN, STARTED);
351    }
352
353    private boolean trySetCancelled() {
354      return compareAndSet(NOT_RUN, CANCELLED);
355    }
356  }
