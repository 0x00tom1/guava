147  /**
148   * Returns the start position of the first occurrence of the specified {@code target} within
149   * {@code array}, or {@code -1} if there is no such occurrence.
150   *
151   * <p>More formally, returns the lowest index {@code i} such that {@code Arrays.copyOfRange(array,
152   * i, i + target.length)} contains exactly the same elements as {@code target}.
153   *
154   * <p>Note that this always returns {@code -1} when {@code target} contains {@code NaN}.
155   *
156   * @param array the array to search for the sequence {@code target}
157   * @param target the array to search for as a sub-sequence of {@code array}
158   */
159  public static int indexOf(float[] array, float[] target) {
160    checkNotNull(array, "array");
161    checkNotNull(target, "target");
162    if (target.length == 0) {
163      return 0;
164    }
165
166    outer:
167    for (int i = 0; i < array.length - target.length + 1; i++) {
168      for (int j = 0; j < target.length; j++) {
169        if (array[i + j] != target[j]) {
170          continue outer;
171        }
172      }
173      return i;
174    }
175    return -1;
176  }

241  /**
242   * Returns the value nearest to {@code value} which is within the closed range {@code [min..max]}.
243   *
244   * <p>If {@code value} is within the range {@code [min..max]}, {@code value} is returned
245   * unchanged. If {@code value} is less than {@code min}, {@code min} is returned, and if {@code
246   * value} is greater than {@code max}, {@code max} is returned.
247   *
248   * @param value the {@code float} value to constrain
249   * @param min the lower bound (inclusive) of the range to constrain {@code value} to
250   * @param max the upper bound (inclusive) of the range to constrain {@code value} to
251   * @throws IllegalArgumentException if {@code min > max}
252   * @since 21.0
253   */
254  @Beta
255  public static float constrainToRange(float value, float min, float max) {
256    // avoid auto-boxing by not using Preconditions.checkArgument(); see Guava issue 3984
257    // Reject NaN by testing for the good case (min <= max) instead of the bad (min > max).
258    if (min <= max) {
259      return Math.min(Math.max(value, min), max);
260    }
261    throw new IllegalArgumentException(
262        lenientFormat("min (%s) must be less than or equal to max (%s)", min, max));
263  }

265  /**
266   * Returns the values from each provided array combined into a single array. For example, {@code
267   * concat(new float[] {a, b}, new float[] {}, new float[] {c}} returns the array {@code {a, b,
268   * c}}.
269   *
270   * @param arrays zero or more {@code float} arrays
271   * @return a single array containing all the values from the source arrays, in order
272   */
273  public static float[] concat(float[]... arrays) {
274    int length = 0;
275    for (float[] array : arrays) {
276      length += array.length;
277    }
278    float[] result = new float[length];
279    int pos = 0;
280    for (float[] array : arrays) {
281      System.arraycopy(array, 0, result, pos, array.length);
282      pos += array.length;
283    }
284    return result;
285  }

325   * Returns an array containing the same values as {@code array}, but guaranteed to be of a
326   * specified minimum length. If {@code array} already has a length of at least {@code minLength},
327   * it is returned directly. Otherwise, a new array of size {@code minLength + padding} is
328   * returned, containing the values of {@code array}, and zeroes in the remaining places.
329   *
330   * @param array the source array
331   * @param minLength the minimum length the returned array must guarantee
332   * @param padding an extra amount to "grow" the array by if growth is necessary
333   * @throws IllegalArgumentException if {@code minLength} or {@code padding} is negative
334   * @return an array containing the values of {@code array}, with guaranteed minimum length {@code

338    checkArgument(minLength >= 0, "Invalid minLength: %s", minLength);
339    checkArgument(padding >= 0, "Invalid padding: %s", padding);
340    return (array.length < minLength) ? Arrays.copyOf(array, minLength + padding) : array;
341  }

343  /**
344   * Returns a string containing the supplied {@code float} values, converted to strings as
345   * specified by {@link Float#toString(float)}, and separated by {@code separator}. For example,
346   * {@code join("-", 1.0f, 2.0f, 3.0f)} returns the string {@code "1.0-2.0-3.0"}.
347   *
348   * <p>Note that {@link Float#toString(float)} formats {@code float} differently in GWT. In the
349   * previous example, it returns the string {@code "1-2-3"}.
350   *
351   * @param separator the text that should appear between consecutive values in the resulting string
352   *     (but not at the start or end)
353   * @param array an array of {@code float} values, possibly empty
354   */
355  public static String join(String separator, float... array) {
356    checkNotNull(separator);
357    if (array.length == 0) {
358      return "";
359    }
360
361    // For pre-sizing a builder, just get the right order of magnitude
362    StringBuilder builder = new StringBuilder(array.length * 12);
363    builder.append(array[0]);
364    for (int i = 1; i < array.length; i++) {
365      builder.append(separator).append(array[i]);
366    }
367    return builder.toString();
368  }

370  /**
371   * Returns a comparator that compares two {@code float} arrays <a
372   * href="http://en.wikipedia.org/wiki/Lexicographical_order">lexicographically</a>. That is, it
373   * compares, using {@link #compare(float, float)}), the first pair of values that follow any
374   * common prefix, or when one array is a prefix of the other, treats the shorter array as the
375   * lesser. For example, {@code [] < [1.0f] < [1.0f, 2.0f] < [2.0f]}.
376   *
377   * <p>The returned comparator is inconsistent with {@link Object#equals(Object)} (since arrays
378   * support only identity equality), but it is consistent with {@link Arrays#equals(float[],
379   * float[])}.
380   *
381   * @since 2.0
382   */
383  public static Comparator<float[]> lexicographicalComparator() {
384    return LexicographicalComparator.INSTANCE;
385  }
386
387  private enum LexicographicalComparator implements Comparator<float[]> {
388    INSTANCE;
389
390    @Override
391    public int compare(float[] left, float[] right) {
392      int minLength = Math.min(left.length, right.length);
393      for (int i = 0; i < minLength; i++) {
394        int result = Float.compare(left[i], right[i]);
395        if (result != 0) {
396          return result;
397        }
398      }
399      return left.length - right.length;
400    }
401
402    @Override
403    public String toString() {
404      return "Floats.lexicographicalComparator()";
405    }
406  }

448  /**
449   * Reverses the elements of {@code array} between {@code fromIndex} inclusive and {@code toIndex}
450   * exclusive. This is equivalent to {@code
451   * Collections.reverse(Floats.asList(array).subList(fromIndex, toIndex))}, but is likely to be
452   * more efficient.
453   *
454   * @throws IndexOutOfBoundsException if {@code fromIndex < 0}, {@code toIndex > array.length}, or
455   *     {@code toIndex > fromIndex}
456   * @since 23.1
457   */
458  public static void reverse(float[] array, int fromIndex, int toIndex) {
459    checkNotNull(array);
460    checkPositionIndexes(fromIndex, toIndex, array.length);
461    for (int i = fromIndex, j = toIndex - 1; i < j; i++, j--) {
462      float tmp = array[i];
463      array[i] = array[j];
464      array[j] = tmp;
465    }
466  }

468  /**
469   * Performs a right rotation of {@code array} of "distance" places, so that the first element is
470   * moved to index "distance", and the element at index {@code i} ends up at index {@code (distance
471   * + i) mod array.length}. This is equivalent to {@code Collections.rotate(Floats.asList(array),
472   * distance)}, but is considerably faster and avoids allocation and garbage collection.
473   *
474   * <p>The provided "distance" may be negative, which will rotate left.
475   *
476   * @since NEXT
477   */
478  public static void rotate(float[] array, int distance) {
479    rotate(array, distance, 0, array.length);
480  }

482  /**
483   * Performs a right rotation of {@code array} between {@code fromIndex} inclusive and {@code
484   * toIndex} exclusive. This is equivalent to {@code
485   * Collections.rotate(Floats.asList(array).subList(fromIndex, toIndex), distance)}, but is
486   * considerably faster and avoids allocations and garbage collection.
487   *
488   * <p>The provided "distance" may be negative, which will rotate left.
489   *
490   * @throws IndexOutOfBoundsException if {@code fromIndex < 0}, {@code toIndex > array.length}, or
491   *     {@code toIndex > fromIndex}
492   * @since NEXT
493   */
494  public static void rotate(float[] array, int distance, int fromIndex, int toIndex) {
495    // See Ints.rotate for more details about possible algorithms here.
496    checkNotNull(array);
497    checkPositionIndexes(fromIndex, toIndex, array.length);
498    if (array.length <= 1) {
499      return;
500    }
501
502    int length = toIndex - fromIndex;
503    // Obtain m = (-distance mod length), a non-negative value less than "length". This is how many
504    // places left to rotate.
505    int m = -distance % length;
506    m = (m < 0) ? m + length : m;
507    // The current index of what will become the first element of the rotated section.
508    int newFirstIndex = m + fromIndex;
509    if (newFirstIndex == fromIndex) {
510      return;
511    }
512
513    reverse(array, fromIndex, newFirstIndex);
514    reverse(array, newFirstIndex, toIndex);
515    reverse(array, fromIndex, toIndex);
516  }

519   * Returns an array containing each value of {@code collection}, converted to a {@code float}
520   * value in the manner of {@link Number#floatValue}.
521   *
522   * <p>Elements are copied from the argument collection as if by {@code collection.toArray()}.
523   * Calling this method is as thread-safe as calling that method.
524   *
525   * @param collection a collection of {@code Number} instances
526   * @return an array containing the same values as {@code collection}, in the same order, converted
527   *     to primitives
528   * @throws NullPointerException if {@code collection} or any of its elements is null
529   * @since 1.0 (parameter was {@code Collection<Float>} before 12.0)

531  public static float[] toArray(Collection<? extends Number> collection) {
532    if (collection instanceof FloatArrayAsList) {
533      return ((FloatArrayAsList) collection).toFloatArray();
534    }
535
536    Object[] boxedArray = collection.toArray();
537    int len = boxedArray.length;
538    float[] array = new float[len];
539    for (int i = 0; i < len; i++) {
540      // checkNotNull for GWT (do not optimize)
541      array[i] = ((Number) checkNotNull(boxedArray[i])).floatValue();
542    }
543    return array;
544  }

570  @GwtCompatible
571  private static class FloatArrayAsList extends AbstractList<Float>
572      implements RandomAccess, Serializable {
573    final float[] array;
574    final int start;
575    final int end;
576
577    FloatArrayAsList(float[] array) {
578      this(array, 0, array.length);
579    }
580
581    FloatArrayAsList(float[] array, int start, int end) {
582      this.array = array;
583      this.start = start;
584      this.end = end;
585    }
586
587    @Override
588    public int size() {
589      return end - start;
590    }
591
592    @Override
593    public boolean isEmpty() {
594      return false;
595    }
596
597    @Override
598    public Float get(int index) {
599      checkElementIndex(index, size());
600      return array[start + index];
601    }
602
603    @Override
604    public boolean contains(@CheckForNull Object target) {
605      // Overridden to prevent a ton of boxing
606      return (target instanceof Float) && Floats.indexOf(array, (Float) target, start, end) != -1;
607    }
608
609    @Override
610    public int indexOf(@CheckForNull Object target) {
611      // Overridden to prevent a ton of boxing
612      if (target instanceof Float) {
613        int i = Floats.indexOf(array, (Float) target, start, end);
614        if (i >= 0) {
615          return i - start;
616        }
617      }
618      return -1;
619    }
620
621    @Override
622    public int lastIndexOf(@CheckForNull Object target) {
623      // Overridden to prevent a ton of boxing
624      if (target instanceof Float) {
625        int i = Floats.lastIndexOf(array, (Float) target, start, end);
626        if (i >= 0) {
627          return i - start;
628        }
629      }
630      return -1;
631    }
632
633    @Override
634    public Float set(int index, Float element) {
635      checkElementIndex(index, size());
636      float oldValue = array[start + index];
637      // checkNotNull for GWT (do not optimize)
638      array[start + index] = checkNotNull(element);
639      return oldValue;
640    }
641
642    @Override
643    public List<Float> subList(int fromIndex, int toIndex) {
644      int size = size();
645      checkPositionIndexes(fromIndex, toIndex, size);
646      if (fromIndex == toIndex) {
647        return Collections.emptyList();
648      }
649      return new FloatArrayAsList(array, start + fromIndex, start + toIndex);
650    }
651
652    @Override
653    public boolean equals(@CheckForNull Object object) {
654      if (object == this) {
655        return true;
656      }
657      if (object instanceof FloatArrayAsList) {
658        FloatArrayAsList that = (FloatArrayAsList) object;
659        int size = size();
660        if (that.size() != size) {
661          return false;
662        }
663        for (int i = 0; i < size; i++) {
664          if (array[start + i] != that.array[that.start + i]) {
665            return false;
666          }
667        }
668        return true;
669      }
670      return super.equals(object);
671    }

673    @Override
674    public int hashCode() {
675      int result = 1;
676      for (int i = start; i < end; i++) {
677        result = 31 * result + Floats.hashCode(array[i]);
678      }
679      return result;
680    }

682    @Override
683    public String toString() {
684      StringBuilder builder = new StringBuilder(size() * 12);
685      builder.append('[').append(array[start]);
686      for (int i = start + 1; i < end; i++) {
687        builder.append(", ").append(array[i]);
688      }
689      return builder.append(']').toString();
690    }

692    float[] toFloatArray() {
693      return Arrays.copyOfRange(array, start, end);
694    }

700   * Parses the specified string as a single-precision floating point value. The ASCII character
701   * {@code '-'} (<code>'&#92;u002D'</code>) is recognized as the minus sign.
702   *
703   * <p>Unlike {@link Float#parseFloat(String)}, this method returns {@code null} instead of
704   * throwing an exception if parsing fails. Valid inputs are exactly those accepted by {@link
705   * Float#valueOf(String)}, except that leading and trailing whitespace is not permitted.
706   *
707   * <p>This implementation is likely to be faster than {@code Float.parseFloat} if many failures
708   * are expected.
709   *
710   * @param string the string representation of a {@code float} value
711   * @return the floating point value represented by {@code string}, or {@code null} if {@code
712   *     string} has a length of zero or cannot be parsed as a {@code float} value
713   * @throws NullPointerException if {@code string} is {@code null}
714   * @since 14.0
715   */
716  @J2ktIncompatible
717  @GwtIncompatible // regular expressions
718  @CheckForNull
719  public static Float tryParse(String string) {
720    if (Doubles.FLOATING_POINT_PATTERN.matcher(string).matches()) {
721      // TODO(lowasser): could be potentially optimized, but only with
722      // extensive testing
723      try {
724        return Float.parseFloat(string);
725      } catch (NumberFormatException e) {
726        // Float.parseFloat has changed specs several times, so fall through
727        // gracefully
728      }
729    }
730    return null;
731  }
