037/**
038 * Wrapper around either a {@link Method} or a {@link Constructor}. Convenience API is provided to
039 * make common reflective operation easier to deal with, such as {@link #isPublic}, {@link
040 * #getParameters} etc.
041 *
042 * <p>In addition to convenience methods, {@link TypeToken#method} and {@link TypeToken#constructor}
043 * will resolve the type parameters of the method or constructor in the context of the owner type,
044 * which may be a subtype of the declaring class. For example:
045 *
046 * <pre>{@code
047 * Method getMethod = List.class.getMethod("get", int.class);
048 * Invokable<List<String>, ?> invokable = new TypeToken<List<String>>() {}.method(getMethod);
049 * assertEquals(TypeToken.of(String.class), invokable.getReturnType()); // Not Object.class!
050 * assertEquals(new TypeToken<List<String>>() {}, invokable.getOwnerType());
051 * }</pre>
052 *
053 * @param <T> the type that owns this method or constructor.
054 * @param <R> the return type of (or supertype thereof) the method or the declaring type of the
055 *     constructor.
056 * @author Ben Yu
057 * @since 14.0
058 */
059@Beta
060@ElementTypesAreNonnullByDefault
061public abstract class Invokable<T, R> extends Element implements GenericDeclaration {
062
063  <M extends AccessibleObject & Member> Invokable(M member) {
064    super(member);
065  }
066
067  /** Returns {@link Invokable} of {@code method}. */
068  public static Invokable<?, Object> from(Method method) {
069    return new MethodInvokable<>(method);
070  }
071
072  /** Returns {@link Invokable} of {@code constructor}. */
073  public static <T> Invokable<T, T> from(Constructor<T> constructor) {
074    return new ConstructorInvokable<T>(constructor);
075  }
076
077  /**
078   * Returns {@code true} if this is an overridable method. Constructors, private, static or final
079   * methods, or methods declared by final classes are not overridable.
080   */
081  public abstract boolean isOverridable();
082
083  /** Returns {@code true} if this was declared to take a variable number of arguments. */
084  public abstract boolean isVarArgs();
085
086  /**
087   * Invokes with {@code receiver} as 'this' and {@code args} passed to the underlying method and
088   * returns the return value; or calls the underlying constructor with {@code args} and returns the
089   * constructed instance.
090   *
091   * @throws IllegalAccessException if this {@code Constructor} object enforces Java language access
092   *     control and the underlying method or constructor is inaccessible.
093   * @throws IllegalArgumentException if the number of actual and formal parameters differ; if an
094   *     unwrapping conversion for primitive arguments fails; or if, after possible unwrapping, a
095   *     parameter value cannot be converted to the corresponding formal parameter type by a method
096   *     invocation conversion.
097   * @throws InvocationTargetException if the underlying method or constructor throws an exception.
098   */
099  // All subclasses are owned by us and we'll make sure to get the R type right, including nullness.
100  @SuppressWarnings({"unchecked", "nullness"})
101  @CanIgnoreReturnValue
102  @CheckForNull
103  public final R invoke(@CheckForNull T receiver, @Nullable Object... args)
104      throws InvocationTargetException, IllegalAccessException {
105    return (R) invokeInternal(receiver, checkNotNull(args));
106  }
107
108  /** Returns the return type of this {@code Invokable}. */
109  // All subclasses are owned by us and we'll make sure to get the R type right.
110  @SuppressWarnings("unchecked")
111  public final TypeToken<? extends R> getReturnType() {
112    return (TypeToken<? extends R>) TypeToken.of(getGenericReturnType());
113  }
114
115  /**
116   * Returns all declared parameters of this {@code Invokable}. Note that if this is a constructor
117   * of a non-static inner class, unlike {@link Constructor#getParameterTypes}, the hidden {@code
118   * this} parameter of the enclosing class is excluded from the returned parameters.
119   */
120  public final ImmutableList<Parameter> getParameters() {
121    Type[] parameterTypes = getGenericParameterTypes();
122    Annotation[][] annotations = getParameterAnnotations();
123    AnnotatedType[] annotatedTypes = getAnnotatedParameterTypes();
124    ImmutableList.Builder<Parameter> builder = ImmutableList.builder();
125    for (int i = 0; i < parameterTypes.length; i++) {
126      builder.add(
127          new Parameter(
128              this, i, TypeToken.of(parameterTypes[i]), annotations[i], annotatedTypes[i]));
129    }
130    return builder.build();
131  }
132
133  /** Returns all declared exception types of this {@code Invokable}. */
134  public final ImmutableList<TypeToken<? extends Throwable>> getExceptionTypes() {
135    ImmutableList.Builder<TypeToken<? extends Throwable>> builder = ImmutableList.builder();
136    for (Type type : getGenericExceptionTypes()) {
137      // getGenericExceptionTypes() will never return a type that's not exception
138      @SuppressWarnings("unchecked")
139      TypeToken<? extends Throwable> exceptionType =
140          (TypeToken<? extends Throwable>) TypeToken.of(type);
141      builder.add(exceptionType);
142    }
143    return builder.build();
144  }
145
146  /**
147   * Explicitly specifies the return type of this {@code Invokable}. For example:
148   *
149   * <pre>{@code
150   * Method factoryMethod = Person.class.getMethod("create");
151   * Invokable<?, Person> factory = Invokable.of(getNameMethod).returning(Person.class);
152   * }</pre>
153   */
154  public final <R1 extends R> Invokable<T, R1> returning(Class<R1> returnType) {
155    return returning(TypeToken.of(returnType));
156  }
157
158  /** Explicitly specifies the return type of this {@code Invokable}. */
159  public final <R1 extends R> Invokable<T, R1> returning(TypeToken<R1> returnType) {
160    if (!returnType.isSupertypeOf(getReturnType())) {
161      throw new IllegalArgumentException(
162          "Invokable is known to return " + getReturnType() + ", not " + returnType);
163    }
164    @SuppressWarnings("unchecked") // guarded by previous check
165    Invokable<T, R1> specialized = (Invokable<T, R1>) this;
166    return specialized;
167  }
168
169  @SuppressWarnings("unchecked") // The declaring class is T's raw class, or one of its supertypes.
170  @Override
171  public final Class<? super T> getDeclaringClass() {
172    return (Class<? super T>) super.getDeclaringClass();
173  }
174
175  /** Returns the type of {@code T}. */
176  // Overridden in TypeToken#method() and TypeToken#constructor()
177  @SuppressWarnings("unchecked") // The declaring class is T.
178  @Override
179  public TypeToken<T> getOwnerType() {
180    return (TypeToken<T>) TypeToken.of(getDeclaringClass());
181  }
182
183  @CheckForNull
184  abstract Object invokeInternal(@CheckForNull Object receiver, @Nullable Object[] args)
185      throws InvocationTargetException, IllegalAccessException;
186
187  abstract Type[] getGenericParameterTypes();
188
189  abstract AnnotatedType[] getAnnotatedParameterTypes();
190
191  /** This should never return a type that's not a subtype of Throwable. */
192  abstract Type[] getGenericExceptionTypes();
193
194  abstract Annotation[][] getParameterAnnotations();
195
196  abstract Type getGenericReturnType();
197
198  public abstract AnnotatedType getAnnotatedReturnType();

200  static class MethodInvokable<T> extends Invokable<T, Object> {
201
202    final Method method;
203
204    MethodInvokable(Method method) {
205      super(method);
206      this.method = method;
207    }
208
209    @Override
210    @CheckForNull
211    final Object invokeInternal(@CheckForNull Object receiver, @Nullable Object[] args)
212        throws InvocationTargetException, IllegalAccessException {
213      return method.invoke(receiver, args);
214    }
215
216    @Override
217    Type getGenericReturnType() {
218      return method.getGenericReturnType();
219    }
220
221    @Override
222    Type[] getGenericParameterTypes() {
223      return method.getGenericParameterTypes();
224    }
225
226    @Override
227    AnnotatedType[] getAnnotatedParameterTypes() {
228      return method.getAnnotatedParameterTypes();
229    }
230
231    @Override
232    public AnnotatedType getAnnotatedReturnType() {
233      return method.getAnnotatedReturnType();
234    }
235
236    @Override
237    Type[] getGenericExceptionTypes() {
238      return method.getGenericExceptionTypes();
239    }
240
241    @Override
242    final Annotation[][] getParameterAnnotations() {
243      return method.getParameterAnnotations();
244    }
245
246    @Override
247    public final TypeVariable<?>[] getTypeParameters() {
248      return method.getTypeParameters();
249    }
250
251    @Override
252    public final boolean isOverridable() {
253      return !(isFinal()
254          || isPrivate()
255          || isStatic()
256          || Modifier.isFinal(getDeclaringClass().getModifiers()));
257    }
258
259    @Override
260    public final boolean isVarArgs() {
261      return method.isVarArgs();
262    }
263  }
264
265  static class ConstructorInvokable<T> extends Invokable<T, T> {
266
267    final Constructor<?> constructor;
268
269    ConstructorInvokable(Constructor<?> constructor) {
270      super(constructor);
271      this.constructor = constructor;
272    }
273
274    @Override
275    final Object invokeInternal(@CheckForNull Object receiver, @Nullable Object[] args)
276        throws InvocationTargetException, IllegalAccessException {
277      try {
278        return constructor.newInstance(args);
279      } catch (InstantiationException e) {
280        throw new RuntimeException(constructor + " failed.", e);
281      }
282    }
283
284    /**
285     * If the class is parameterized, such as {@link java.util.ArrayList ArrayList}, this returns
286     * {@code ArrayList<E>}.
287     */
288    @Override
289    Type getGenericReturnType() {
290      Class<?> declaringClass = getDeclaringClass();
291      TypeVariable<?>[] typeParams = declaringClass.getTypeParameters();
292      if (typeParams.length > 0) {
293        return Types.newParameterizedType(declaringClass, typeParams);
294      } else {
295        return declaringClass;
296      }
297    }
298
299    @Override
300    Type[] getGenericParameterTypes() {
301      Type[] types = constructor.getGenericParameterTypes();
302      if (types.length > 0 && mayNeedHiddenThis()) {
303        Class<?>[] rawParamTypes = constructor.getParameterTypes();
304        if (types.length == rawParamTypes.length
305            && rawParamTypes[0] == getDeclaringClass().getEnclosingClass()) {
306          // first parameter is the hidden 'this'
307          return Arrays.copyOfRange(types, 1, types.length);
308        }
309      }
310      return types;
311    }
312
313    @Override
314    AnnotatedType[] getAnnotatedParameterTypes() {
315      return constructor.getAnnotatedParameterTypes();
316    }
317
318    @Override
319    public AnnotatedType getAnnotatedReturnType() {
320      return constructor.getAnnotatedReturnType();
321    }
322
323    @Override
324    Type[] getGenericExceptionTypes() {
325      return constructor.getGenericExceptionTypes();
326    }
327
328    @Override
329    final Annotation[][] getParameterAnnotations() {
330      return constructor.getParameterAnnotations();
331    }
332
333    /**
334     * {@inheritDoc}
335     *
336     * <p>{@code [<E>]} will be returned for ArrayList's constructor. When both the class and the
337     * constructor have type parameters, the class parameters are prepended before those of the
338     * constructor's. This is an arbitrary rule since no existing language spec mandates one way or
339     * the other. From the declaration syntax, the class type parameter appears first, but the call
340     * syntax may show up in opposite order such as {@code new <A>Foo<B>()}.
341     */
342    @Override
343    public final TypeVariable<?>[] getTypeParameters() {
344      TypeVariable<?>[] declaredByClass = getDeclaringClass().getTypeParameters();
345      TypeVariable<?>[] declaredByConstructor = constructor.getTypeParameters();
346      TypeVariable<?>[] result =
347          new TypeVariable<?>[declaredByClass.length + declaredByConstructor.length];
348      System.arraycopy(declaredByClass, 0, result, 0, declaredByClass.length);
349      System.arraycopy(
350          declaredByConstructor, 0, result, declaredByClass.length, declaredByConstructor.length);
351      return result;
352    }
353
354    @Override
355    public final boolean isOverridable() {
356      return false;
357    }
358
359    @Override
360    public final boolean isVarArgs() {
361      return constructor.isVarArgs();
362    }
363
364    private boolean mayNeedHiddenThis() {
365      Class<?> declaringClass = constructor.getDeclaringClass();
366      if (declaringClass.getEnclosingConstructor() != null) {
367        // Enclosed in a constructor, needs hidden this
368        return true;
369      }
370      Method enclosingMethod = declaringClass.getEnclosingMethod();
371      if (enclosingMethod != null) {
372        // Enclosed in a method, if it's not static, must need hidden this.
373        return !Modifier.isStatic(enclosingMethod.getModifiers());
374      } else {
375        // Strictly, this doesn't necessarily indicate a hidden 'this' in the case of
376        // static initializer. But there seems no way to tell in that case. :(
377        // This may cause issues when an anonymous class is created inside a static initializer,
378        // and the class's constructor's first parameter happens to be the enclosing class.
379        // In such case, we may mistakenly think that the class is within a non-static context
380        // and the first parameter is the hidden 'this'.
381        return declaringClass.getEnclosingClass() != null
382            && !Modifier.isStatic(declaringClass.getModifiers());
383      }
384    }
385  }
