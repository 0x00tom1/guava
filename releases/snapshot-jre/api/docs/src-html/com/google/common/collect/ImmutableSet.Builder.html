128  /**
129   * Returns an immutable set containing the given elements, minus duplicates, in the order each was
130   * first specified. That is, if multiple elements are {@linkplain Object#equals equal}, all except
131   * the first are ignored.
132   *
133   * <p>The array {@code others} must not be longer than {@code Integer.MAX_VALUE - 6}.
134   *
135   * @since 3.0 (source-compatible since 2.0)
136   */
137  @SafeVarargs // For Eclipse. For internal javac we have disabled this pointless type of warning.
138  public static <E> ImmutableSet<E> of(E e1, E e2, E e3, E e4, E e5, E e6, E... others) {
139    checkArgument(
140        others.length <= Integer.MAX_VALUE - 6, "the total number of elements must fit in an int");
141    final int paramCount = 6;
142    Object[] elements = new Object[paramCount + others.length];
143    elements[0] = e1;
144    elements[1] = e2;
145    elements[2] = e3;
146    elements[3] = e4;
147    elements[4] = e5;
148    elements[5] = e6;
149    System.arraycopy(others, 0, elements, paramCount, others.length);
150    return construct(elements.length, elements.length, elements);
151  }

153  /**
154   * Constructs an {@code ImmutableSet} from the first {@code n} elements of the specified array,
155   * which we have no particular reason to believe does or does not contain duplicates. If {@code k}
156   * is the size of the returned {@code ImmutableSet}, then the unique elements of {@code elements}
157   * will be in the first {@code k} positions, and {@code elements[i] == null} for {@code k <= i <
158   * n}.
159   *
160   * <p>This may modify {@code elements}. Additionally, if {@code n == elements.length} and {@code
161   * elements} contains no duplicates, {@code elements} may be used without copying in the returned
162   * {@code ImmutableSet}, in which case the caller must not modify it.
163   *
164   * <p>{@code elements} may contain only values of type {@code E}.
165   *
166   * @throws NullPointerException if any of the first {@code n} elements of {@code elements} is null
167   */
168  private static <E> ImmutableSet<E> constructUnknownDuplication(int n, Object... elements) {
169    // Guess the size is "halfway between" all duplicates and no duplicates, on a log scale.
170    return construct(
171        n,
172        Math.max(
173            ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY,
174            IntMath.sqrt(n, RoundingMode.CEILING)),
175        elements);
176  }

178  /**
179   * Constructs an {@code ImmutableSet} from the first {@code n} elements of the specified array. If
180   * {@code k} is the size of the returned {@code ImmutableSet}, then the unique elements of {@code
181   * elements} will be in the first {@code k} positions, and {@code elements[i] == null} for {@code
182   * k <= i < n}.
183   *
184   * <p>This may modify {@code elements}. Additionally, if {@code n == elements.length} and {@code
185   * elements} contains no duplicates, {@code elements} may be used without copying in the returned
186   * {@code ImmutableSet}, in which case it may no longer be modified.
187   *
188   * <p>{@code elements} may contain only values of type {@code E}.
189   *
190   * @throws NullPointerException if any of the first {@code n} elements of {@code elements} is null
191   */
192  private static <E> ImmutableSet<E> construct(int n, int expectedSize, Object... elements) {
193    switch (n) {
194      case 0:
195        return of();
196      case 1:
197        @SuppressWarnings("unchecked") // safe; elements contains only E's
198        E elem = (E) elements[0];
199        return of(elem);
200      default:
201        SetBuilderImpl<E> builder = new RegularSetBuilderImpl<E>(expectedSize);
202        for (int i = 0; i < n; i++) {
203          @SuppressWarnings("unchecked")
204          E e = (E) checkNotNull(elements[i]);
205          builder = builder.add(e);
206        }
207        return builder.review().build();
208    }
209  }

211  /**
212   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
213   * each appears first in the source collection.
214   *
215   * <p><b>Performance note:</b> This method will sometimes recognize that the actual copy operation
216   * is unnecessary; for example, {@code copyOf(copyOf(anArrayList))} will copy the data only once.
217   * This reduces the expense of habitually making defensive copies at API boundaries. However, the
218   * precise conditions for skipping the copy operation are undefined.
219   *
220   * @throws NullPointerException if any of {@code elements} is null
221   * @since 7.0 (source-compatible since 2.0)
222   */
223  public static <E> ImmutableSet<E> copyOf(Collection<? extends E> elements) {
224    /*
225     * TODO(lowasser): consider checking for ImmutableAsList here
226     * TODO(lowasser): consider checking for Multiset here
227     */
228    // Don't refer to ImmutableSortedSet by name so it won't pull in all that code
229    if (elements instanceof ImmutableSet && !(elements instanceof SortedSet)) {
230      @SuppressWarnings("unchecked") // all supported methods are covariant
231      ImmutableSet<E> set = (ImmutableSet<E>) elements;
232      if (!set.isPartialView()) {
233        return set;
234      }
235    } else if (elements instanceof EnumSet) {
236      return copyOfEnumSet((EnumSet) elements);
237    }
238    Object[] array = elements.toArray();
239    if (elements instanceof Set) {
240      // assume probably no duplicates (though it might be using different equality semantics)
241      return construct(array.length, array.length, array);
242    } else {
243      return constructUnknownDuplication(array.length, array);
244    }
245  }

247  /**
248   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
249   * each appears first in the source iterable. This method iterates over {@code elements} only
250   * once.
251   *
252   * <p><b>Performance note:</b> This method will sometimes recognize that the actual copy operation
253   * is unnecessary; for example, {@code copyOf(copyOf(anArrayList))} should copy the data only
254   * once. This reduces the expense of habitually making defensive copies at API boundaries.
255   * However, the precise conditions for skipping the copy operation are undefined.
256   *
257   * @throws NullPointerException if any of {@code elements} is null
258   */
259  public static <E> ImmutableSet<E> copyOf(Iterable<? extends E> elements) {
260    return (elements instanceof Collection)
261        ? copyOf((Collection<? extends E>) elements)
262        : copyOf(elements.iterator());
263  }

265  /**
266   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
267   * each appears first in the source iterator.
268   *
269   * @throws NullPointerException if any of {@code elements} is null
270   */
271  public static <E> ImmutableSet<E> copyOf(Iterator<? extends E> elements) {
272    // We special-case for 0 or 1 elements, but anything further is madness.
273    if (!elements.hasNext()) {
274      return of();
275    }
276    E first = elements.next();
277    if (!elements.hasNext()) {
278      return of(first);
279    } else {
280      return new ImmutableSet.Builder<E>().add(first).addAll(elements).build();
281    }
282  }

284  /**
285   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
286   * each appears first in the source array.
287   *
288   * @throws NullPointerException if any of {@code elements} is null
289   * @since 3.0
290   */
291  public static <E> ImmutableSet<E> copyOf(E[] elements) {
292    switch (elements.length) {
293      case 0:
294        return of();
295      case 1:
296        return of(elements[0]);
297      default:
298        return constructUnknownDuplication(elements.length, elements.clone());
299    }
300  }

445  /**
446   * A builder for creating {@code ImmutableSet} instances. Example:
447   *
448   * <pre>{@code
449   * static final ImmutableSet<Color> GOOGLE_COLORS =
450   *     ImmutableSet.<Color>builder()
451   *         .addAll(WEBSAFE_COLORS)
452   *         .add(new Color(0, 191, 255))
453   *         .build();
454   * }</pre>
455   *
456   * <p>Elements appear in the resulting set in the same order they were first added to the builder.
457   *
458   * <p>Building does not change the state of the builder, so it is still possible to add more
459   * elements and to build again.
460   *
461   * @since 2.0
462   */
463  public static class Builder<E> extends ImmutableCollection.Builder<E> {
464    /*
465     * `impl` is null only for instances of the subclass, ImmutableSortedSet.Builder. That subclass
466     * overrides all the methods that access it here. Thus, all the methods here can safely assume
467     * that this field is non-null.
468     */
469    @CheckForNull private SetBuilderImpl<E> impl;
470    boolean forceCopy;
471
472    public Builder() {
473      this(0);
474    }
475
476    Builder(int capacity) {
477      if (capacity > 0) {
478        impl = new RegularSetBuilderImpl<E>(capacity);
479      } else {
480        impl = EmptySetBuilderImpl.instance();
481      }
482    }
483
484    Builder(@SuppressWarnings("unused") boolean subclass) {
485      this.impl = null; // unused
486    }
487
488    @VisibleForTesting
489    void forceJdk() {
490      requireNonNull(impl); // see the comment on the field
491      this.impl = new JdkBackedSetBuilderImpl<E>(impl);
492    }
493
494    final void copyIfNecessary() {
495      if (forceCopy) {
496        copy();
497        forceCopy = false;
498      }
499    }
500
501    void copy() {
502      requireNonNull(impl); // see the comment on the field
503      impl = impl.copy();
504    }
505
506    @Override
507    @CanIgnoreReturnValue
508    public Builder<E> add(E element) {
509      requireNonNull(impl); // see the comment on the field
510      checkNotNull(element);
511      copyIfNecessary();
512      impl = impl.add(element);
513      return this;
514    }
515
516    @Override
517    @CanIgnoreReturnValue
518    public Builder<E> add(E... elements) {
519      super.add(elements);
520      return this;
521    }
522
523    /**
524     * Adds each element of {@code elements} to the {@code ImmutableSet}, ignoring duplicate
525     * elements (only the first duplicate element is added).
526     *
527     * @param elements the elements to add
528     * @return this {@code Builder} object
529     * @throws NullPointerException if {@code elements} is null or contains a null element
530     */
531    @Override
532    @CanIgnoreReturnValue
533    public Builder<E> addAll(Iterable<? extends E> elements) {
534      super.addAll(elements);
535      return this;
536    }
537
538    @Override
539    @CanIgnoreReturnValue
540    public Builder<E> addAll(Iterator<? extends E> elements) {
541      super.addAll(elements);
542      return this;
543    }
544
545    Builder<E> combine(Builder<E> other) {
546      requireNonNull(impl);
547      requireNonNull(other.impl);
548      /*
549       * For discussion of requireNonNull, see the comment on the field.
550       *
551       * (And I don't believe there's any situation in which we call x.combine(y) when x is a plain
552       * ImmutableSet.Builder but y is an ImmutableSortedSet.Builder (or vice versa). Certainly
553       * ImmutableSortedSet.Builder.combine() is written as if its argument will never be a plain
554       * ImmutableSet.Builder: It casts immediately to ImmutableSortedSet.Builder.)
555       */
556      copyIfNecessary();
557      this.impl = this.impl.combine(other.impl);
558      return this;
559    }
560
561    @Override
562    public ImmutableSet<E> build() {
563      requireNonNull(impl); // see the comment on the field
564      forceCopy = true;
565      impl = impl.review();
566      return impl.build();
567    }
568  }

570  /** Swappable internal implementation of an ImmutableSet.Builder. */
571  private abstract static class SetBuilderImpl<E> {
572    // The first `distinct` elements are non-null.
573    // Since we can never access null elements, we don't mark this nullable.
574    E[] dedupedElements;
575    int distinct;
576
577    @SuppressWarnings("unchecked")
578    SetBuilderImpl(int expectedCapacity) {
579      this.dedupedElements = (E[]) new Object[expectedCapacity];
580      this.distinct = 0;
581    }
582
583    /** Initializes this SetBuilderImpl with a copy of the deduped elements array from toCopy. */
584    SetBuilderImpl(SetBuilderImpl<E> toCopy) {
585      this.dedupedElements = Arrays.copyOf(toCopy.dedupedElements, toCopy.dedupedElements.length);
586      this.distinct = toCopy.distinct;
587    }
588
589    /**
590     * Resizes internal data structures if necessary to store the specified number of distinct
591     * elements.
592     */
593    private void ensureCapacity(int minCapacity) {
594      if (minCapacity > dedupedElements.length) {
595        int newCapacity =
596            ImmutableCollection.Builder.expandedCapacity(dedupedElements.length, minCapacity);
597        dedupedElements = Arrays.copyOf(dedupedElements, newCapacity);
598      }
599    }
600
601    /** Adds e to the insertion-order array of deduplicated elements. Calls ensureCapacity. */
602    final void addDedupedElement(E e) {
603      ensureCapacity(distinct + 1);
604      dedupedElements[distinct++] = e;
605    }
606
607    /**
608     * Adds e to this SetBuilderImpl, returning the updated result. Only use the returned
609     * SetBuilderImpl, since we may switch implementations if e.g. hash flooding is detected.
610     */
611    abstract SetBuilderImpl<E> add(E e);
612
613    /** Adds all the elements from the specified SetBuilderImpl to this SetBuilderImpl. */
614    final SetBuilderImpl<E> combine(SetBuilderImpl<E> other) {
615      SetBuilderImpl<E> result = this;
616      for (int i = 0; i < other.distinct; i++) {
617        /*
618         * requireNonNull is safe because we ensure that the first `distinct` elements have been
619         * populated.
620         */
621        result = result.add(requireNonNull(other.dedupedElements[i]));
622      }
623      return result;
624    }
625
626    /**
627     * Creates a new copy of this SetBuilderImpl. Modifications to that SetBuilderImpl will not
628     * affect this SetBuilderImpl or sets constructed from this SetBuilderImpl via build().
629     */
630    abstract SetBuilderImpl<E> copy();
631
632    /**
633     * Call this before build(). Does a final check on the internal data structures, e.g. shrinking
634     * unnecessarily large structures or detecting previously unnoticed hash flooding.
635     */
636    SetBuilderImpl<E> review() {
637      return this;
638    }
639
640    abstract ImmutableSet<E> build();
641  }

680  /**
681   * Returns an array size suitable for the backing array of a hash table that uses open addressing
682   * with linear probing in its implementation. The returned size is the smallest power of two that
683   * can hold setSize elements with the desired load factor. Always returns at least setSize + 2.
684   */
685  // TODO(cpovirk): Move to Hashing or something, since it's used elsewhere in the Android version.
686  static int chooseTableSize(int setSize) {
687    setSize = Math.max(setSize, 2);
688    // Correct the size for open addressing to match desired load factor.
689    if (setSize < CUTOFF) {
690      // Round up to the next highest power of 2.
691      int tableSize = Integer.highestOneBit(setSize - 1) << 1;
692      while (tableSize * DESIRED_LOAD_FACTOR < setSize) {
693        tableSize <<= 1;
694      }
695      return tableSize;
696    }
697
698    // The table can't be completely full or we'll get infinite reprobes
699    checkArgument(setSize < MAX_TABLE_SIZE, "collection too large");
700    return MAX_TABLE_SIZE;
701  }

703  /**
704   * Default implementation of the guts of ImmutableSet.Builder, creating an open-addressed hash
705   * table and deduplicating elements as they come, so it only allocates O(max(distinct,
706   * expectedCapacity)) rather than O(calls to add).
707   *
708   * <p>This implementation attempts to detect hash flooding, and if it's identified, falls back to
709   * JdkBackedSetBuilderImpl.
710   */
711  private static final class RegularSetBuilderImpl<E> extends SetBuilderImpl<E> {
712    private @Nullable Object[] hashTable;
713    private int maxRunBeforeFallback;
714    private int expandTableThreshold;
715    private int hashCode;
716
717    RegularSetBuilderImpl(int expectedCapacity) {
718      super(expectedCapacity);
719      int tableSize = chooseTableSize(expectedCapacity);
720      this.hashTable = new Object[tableSize];
721      this.maxRunBeforeFallback = maxRunBeforeFallback(tableSize);
722      this.expandTableThreshold = (int) (DESIRED_LOAD_FACTOR * tableSize);
723    }
724
725    RegularSetBuilderImpl(RegularSetBuilderImpl<E> toCopy) {
726      super(toCopy);
727      this.hashTable = toCopy.hashTable.clone();
728      this.maxRunBeforeFallback = toCopy.maxRunBeforeFallback;
729      this.expandTableThreshold = toCopy.expandTableThreshold;
730      this.hashCode = toCopy.hashCode;
731    }
732
733    @Override
734    SetBuilderImpl<E> add(E e) {
735      checkNotNull(e);
736      int eHash = e.hashCode();
737      int i0 = Hashing.smear(eHash);
738      int mask = hashTable.length - 1;
739      for (int i = i0; i - i0 < maxRunBeforeFallback; i++) {
740        int index = i & mask;
741        Object tableEntry = hashTable[index];
742        if (tableEntry == null) {
743          addDedupedElement(e);
744          hashTable[index] = e;
745          hashCode += eHash;
746          ensureTableCapacity(distinct); // rebuilds table if necessary
747          return this;
748        } else if (tableEntry.equals(e)) { // not a new element, ignore
749          return this;
750        }
751      }
752      // we fell out of the loop due to a long run; fall back to JDK impl
753      return new JdkBackedSetBuilderImpl<E>(this).add(e);
754    }
755
756    @Override
757    SetBuilderImpl<E> copy() {
758      return new RegularSetBuilderImpl<E>(this);
759    }
760
761    @Override
762    SetBuilderImpl<E> review() {
763      int targetTableSize = chooseTableSize(distinct);
764      if (targetTableSize * 2 < hashTable.length) {
765        hashTable = rebuildHashTable(targetTableSize, dedupedElements, distinct);
766        maxRunBeforeFallback = maxRunBeforeFallback(targetTableSize);
767        expandTableThreshold = (int) (DESIRED_LOAD_FACTOR * targetTableSize);
768      }
769      return hashFloodingDetected(hashTable) ? new JdkBackedSetBuilderImpl<E>(this) : this;
770    }
771
772    @Override
773    ImmutableSet<E> build() {
774      switch (distinct) {
775        case 0:
776          return of();
777        case 1:
778          /*
779           * requireNonNull is safe because we ensure that the first `distinct` elements have been
780           * populated.
781           */
782          return of(requireNonNull(dedupedElements[0]));
783        default:
784          /*
785           * The suppression is safe because we ensure that the first `distinct` elements have been
786           * populated.
787           */
788          @SuppressWarnings("nullness")
789          Object[] elements =
790              (distinct == dedupedElements.length)
791                  ? dedupedElements
792                  : Arrays.copyOf(dedupedElements, distinct);
793          return new RegularImmutableSet<E>(elements, hashCode, hashTable, hashTable.length - 1);
794      }
795    }
796
797    /** Builds a new open-addressed hash table from the first n objects in elements. */
798    static @Nullable Object[] rebuildHashTable(int newTableSize, Object[] elements, int n) {
799      @Nullable Object[] hashTable = new @Nullable Object[newTableSize];
800      int mask = hashTable.length - 1;
801      for (int i = 0; i < n; i++) {
802        // requireNonNull is safe because we ensure that the first n elements have been populated.
803        Object e = requireNonNull(elements[i]);
804        int j0 = Hashing.smear(e.hashCode());
805        for (int j = j0; ; j++) {
806          int index = j & mask;
807          if (hashTable[index] == null) {
808            hashTable[index] = e;
809            break;
810          }
811        }
812      }
813      return hashTable;
814    }
815
816    void ensureTableCapacity(int minCapacity) {
817      if (minCapacity > expandTableThreshold && hashTable.length < MAX_TABLE_SIZE) {
818        int newTableSize = hashTable.length * 2;
819        hashTable = rebuildHashTable(newTableSize, dedupedElements, distinct);
820        maxRunBeforeFallback = maxRunBeforeFallback(newTableSize);
821        expandTableThreshold = (int) (DESIRED_LOAD_FACTOR * newTableSize);
822      }
823    }
824
825    /**
826     * We attempt to detect deliberate hash flooding attempts. If one is detected, we fall back to a
827     * wrapper around j.u.HashSet, which has built in flooding protection. MAX_RUN_MULTIPLIER was
828     * determined experimentally to match our desired probability of false positives.
829     */
830    // NB: yes, this is surprisingly high, but that's what the experiments said was necessary
831    // Raising this number slows the worst-case contains behavior, speeds up hashFloodingDetected,
832    // and reduces the false-positive probability.
833    static final int MAX_RUN_MULTIPLIER = 13;
834
835    /**
836     * Checks the whole hash table for poor hash distribution. Takes O(n) in the worst case, O(n /
837     * log n) on average.
838     *
839     * <p>The online hash flooding detecting in RegularSetBuilderImpl.add can detect e.g. many
840     * exactly matching hash codes, which would cause construction to take O(n^2), but can't detect
841     * e.g. hash codes adversarially designed to go into ascending table locations, which keeps
842     * construction O(n) (as desired) but then can have O(n) queries later.
843     *
844     * <p>If this returns false, then no query can take more than O(log n).
845     *
846     * <p>Note that for a RegularImmutableSet with elements with truly random hash codes, contains
847     * operations take expected O(1) time but with high probability take O(log n) for at least some
848     * element. (https://en.wikipedia.org/wiki/Linear_probing#Analysis)
849     *
850     * <p>This method may return {@code true} even on truly random input, but {@code
851     * ImmutableSetTest} tests that the probability of that is low.
852     */
853    static boolean hashFloodingDetected(@Nullable Object[] hashTable) {
854      int maxRunBeforeFallback = maxRunBeforeFallback(hashTable.length);
855      int mask = hashTable.length - 1;
856
857      // Invariant: all elements at indices in [knownRunStart, knownRunEnd) are nonnull.
858      // If knownRunStart == knownRunEnd, this is vacuously true.
859      // When knownRunEnd exceeds hashTable.length, it "wraps", detecting runs around the end
860      // of the table.
861      int knownRunStart = 0;
862      int knownRunEnd = 0;
863
864      outerLoop:
865      while (knownRunStart < hashTable.length) {
866        if (knownRunStart == knownRunEnd && hashTable[knownRunStart] == null) {
867          if (hashTable[(knownRunStart + maxRunBeforeFallback - 1) & mask] == null) {
868            // There are only maxRunBeforeFallback - 1 elements between here and there,
869            // so even if they were all nonnull, we wouldn't detect a hash flood.  Therefore,
870            // we can skip them all.
871            knownRunStart += maxRunBeforeFallback;
872          } else {
873            knownRunStart++; // the only case in which maxRunEnd doesn't increase by mRBF
874            // happens about f * (1-f) for f = DESIRED_LOAD_FACTOR, so around 21% of the time
875          }
876          knownRunEnd = knownRunStart;
877        } else {
878          for (int j = knownRunStart + maxRunBeforeFallback - 1; j >= knownRunEnd; j--) {
879            if (hashTable[j & mask] == null) {
880              knownRunEnd = knownRunStart + maxRunBeforeFallback;
881              knownRunStart = j + 1;
882              continue outerLoop;
883            }
884          }
885          return true;
886        }
887      }
888      return false;
889    }
890
891    /**
892     * If more than this many consecutive positions are filled in a table of the specified size,
893     * report probable hash flooding. ({@link #hashFloodingDetected} may also report hash flooding
894     * if fewer consecutive positions are filled; see that method for details.)
895     */
896    static int maxRunBeforeFallback(int tableSize) {
897      return MAX_RUN_MULTIPLIER * IntMath.log2(tableSize, RoundingMode.UNNECESSARY);
898    }
899  }

901  /**
902   * SetBuilderImpl version that uses a JDK HashSet, which has built in hash flooding protection.
903   */
904  private static final class JdkBackedSetBuilderImpl<E> extends SetBuilderImpl<E> {
905    private final Set<Object> delegate;
906
907    JdkBackedSetBuilderImpl(SetBuilderImpl<E> toCopy) {
908      super(toCopy); // initializes dedupedElements and distinct
909      delegate = Sets.newHashSetWithExpectedSize(distinct);
910      for (int i = 0; i < distinct; i++) {
911        /*
912         * requireNonNull is safe because we ensure that the first `distinct` elements have been
913         * populated.
914         */
915        delegate.add(requireNonNull(dedupedElements[i]));
916      }
917    }
918
919    @Override
920    SetBuilderImpl<E> add(E e) {
921      checkNotNull(e);
922      if (delegate.add(e)) {
923        addDedupedElement(e);
924      }
925      return this;
926    }
927
928    @Override
929    SetBuilderImpl<E> copy() {
930      return new JdkBackedSetBuilderImpl<>(this);
931    }
932
933    @Override
934    ImmutableSet<E> build() {
935      switch (distinct) {
936        case 0:
937          return of();
938        case 1:
939          /*
940           * requireNonNull is safe because we ensure that the first `distinct` elements have been
941           * populated.
942           */
943          return of(requireNonNull(dedupedElements[0]));
944        default:
945          return new JdkBackedImmutableSet<E>(
946              delegate, ImmutableList.asImmutableList(dedupedElements, distinct));
947      }
948    }
