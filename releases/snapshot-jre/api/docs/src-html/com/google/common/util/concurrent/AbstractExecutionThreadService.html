015package com.google.common.util.concurrent;
016
017import static com.google.common.util.concurrent.Platform.restoreInterruptIfIsInterruptedException;
018
019import com.google.common.annotations.Beta;
020import com.google.common.annotations.GwtIncompatible;
021import com.google.common.base.Supplier;
022import com.google.errorprone.annotations.CanIgnoreReturnValue;
023import java.time.Duration;
024import java.util.concurrent.Executor;
025import java.util.concurrent.TimeUnit;
026import java.util.concurrent.TimeoutException;
027import java.util.logging.Level;
028import java.util.logging.Logger;
029
030/**
031 * Base class for services that can implement {@link #startUp}, {@link #run} and {@link #shutDown}
032 * methods. This class uses a single thread to execute the service; consider {@link AbstractService}
033 * if you would like to manage any threading manually.
034 *
035 * @author Jesse Wilson
036 * @since 1.0
037 */
038@GwtIncompatible
039@ElementTypesAreNonnullByDefault
040public abstract class AbstractExecutionThreadService implements Service {
041  private static final Logger logger =
042      Logger.getLogger(AbstractExecutionThreadService.class.getName());
043
044  /* use AbstractService for state management */
045  private final Service delegate =
046      new AbstractService() {
047        @Override
048        protected final void doStart() {
049          Executor executor =
050              MoreExecutors.renamingDecorator(
051                  executor(),
052                  new Supplier<String>() {
053                    @Override
054                    public String get() {
055                      return serviceName();
056                    }
057                  });
058          executor.execute(
059              new Runnable() {
060                @Override
061                public void run() {
062                  try {
063                    startUp();
064                    notifyStarted();
065                    // If stopAsync() is called while starting we may be in the STOPPING state in
066                    // which case we should skip right down to shutdown.
067                    if (isRunning()) {
068                      try {
069                        AbstractExecutionThreadService.this.run();
070                      } catch (Throwable t) {
071                        restoreInterruptIfIsInterruptedException(t);
072                        try {
073                          shutDown();
074                        } catch (Exception ignored) {
075                          restoreInterruptIfIsInterruptedException(ignored);
076                          // TODO(lukes): if guava ever moves to java7, this would be a good
077                          // candidate for a suppressed exception, or maybe we could generalize
078                          // Closer.Suppressor
079                          logger.log(
080                              Level.WARNING,
081                              "Error while attempting to shut down the service after failure.",
082                              ignored);
083                        }
084                        notifyFailed(t);
085                        return;
086                      }
087                    }
088
089                    shutDown();
090                    notifyStopped();
091                  } catch (Throwable t) {
092                    restoreInterruptIfIsInterruptedException(t);
093                    notifyFailed(t);
094                  }
095                }
096              });
097        }
098
099        @Override
100        protected void doStop() {
101          triggerShutdown();
102        }
103
104        @Override
105        public String toString() {
106          return AbstractExecutionThreadService.this.toString();
107        }
108      };
109
110  /** Constructor for use by subclasses. */
111  protected AbstractExecutionThreadService() {}
112
113  /**
114   * Start the service. This method is invoked on the execution thread.
115   *
116   * <p>By default this method does nothing.
117   */
118  protected void startUp() throws Exception {}
119
120  /**
121   * Run the service. This method is invoked on the execution thread. Implementations must respond
122   * to stop requests. You could poll for lifecycle changes in a work loop:
123   *
124   * <pre>
125   *   public void run() {
126   *     while ({@link #isRunning()}) {
127   *       // perform a unit of work
128   *     }
129   *   }
130   * </pre>
131   *
132   * <p>...or you could respond to stop requests by implementing {@link #triggerShutdown()}, which
133   * should cause {@link #run()} to return.
134   */
135  protected abstract void run() throws Exception;
136
137  /**
138   * Stop the service. This method is invoked on the execution thread.
139   *
140   * <p>By default this method does nothing.
141   */
142  // TODO: consider supporting a TearDownTestCase-like API
143  protected void shutDown() throws Exception {}
144
145  /**
146   * Invoked to request the service to stop.
147   *
148   * <p>By default this method does nothing.
149   *
150   * <p>Currently, this method is invoked while holding a lock. If an implementation of this method
151   * blocks, it can prevent this service from changing state. If you need to performing a blocking
152   * operation in order to trigger shutdown, consider instead registering a listener and
153   * implementing {@code stopping}. Note, however, that {@code stopping} does not run at exactly the
154   * same times as {@code triggerShutdown}.
155   */
156  @Beta
157  protected void triggerShutdown() {}
158
159  /**
160   * Returns the {@link Executor} that will be used to run this service. Subclasses may override
161   * this method to use a custom {@link Executor}, which may configure its worker thread with a
162   * specific name, thread group or priority. The returned executor's {@link
163   * Executor#execute(Runnable) execute()} method is called when this service is started, and should
164   * return promptly.
165   *
166   * <p>The default implementation returns a new {@link Executor} that sets the name of its threads
167   * to the string returned by {@link #serviceName}
168   */
169  protected Executor executor() {
170    return new Executor() {
171      @Override
172      public void execute(Runnable command) {
173        MoreExecutors.newThread(serviceName(), command).start();
174      }
175    };
176  }
177
178  @Override
179  public String toString() {
180    return serviceName() + " [" + state() + "]";
181  }
182
183  @Override
184  public final boolean isRunning() {
185    return delegate.isRunning();
186  }
187
188  @Override
189  public final State state() {
190    return delegate.state();
191  }
192
193  /** @since 13.0 */
194  @Override
195  public final void addListener(Listener listener, Executor executor) {
196    delegate.addListener(listener, executor);
197  }
198
199  /** @since 14.0 */
200  @Override
201  public final Throwable failureCause() {
202    return delegate.failureCause();
203  }
204
205  /** @since 15.0 */
206  @CanIgnoreReturnValue
207  @Override
208  public final Service startAsync() {
209    delegate.startAsync();
210    return this;
211  }
212
213  /** @since 15.0 */
214  @CanIgnoreReturnValue
215  @Override
216  public final Service stopAsync() {
217    delegate.stopAsync();
218    return this;
219  }
220
221  /** @since 15.0 */
222  @Override
223  public final void awaitRunning() {
224    delegate.awaitRunning();
225  }
226
227  /** @since 28.0 */
228  @Override
229  public final void awaitRunning(Duration timeout) throws TimeoutException {
230    Service.super.awaitRunning(timeout);
231  }
232
233  /** @since 15.0 */
234  @Override
235  public final void awaitRunning(long timeout, TimeUnit unit) throws TimeoutException {
236    delegate.awaitRunning(timeout, unit);
237  }
238
239  /** @since 15.0 */
240  @Override
241  public final void awaitTerminated() {
242    delegate.awaitTerminated();
243  }
244
245  /** @since 28.0 */
246  @Override
247  public final void awaitTerminated(Duration timeout) throws TimeoutException {
248    Service.super.awaitTerminated(timeout);
249  }
250
251  /** @since 15.0 */
252  @Override
253  public final void awaitTerminated(long timeout, TimeUnit unit) throws TimeoutException {
254    delegate.awaitTerminated(timeout, unit);
255  }
256
257  /**
258   * Returns the name of this service. {@link AbstractExecutionThreadService} may include the name
259   * in debugging output.
260   *
261   * <p>Subclasses may override this method.
262   *
263   * @since 14.0 (present in 10.0 as getServiceName)
264   */
265  protected String serviceName() {
266    return getClass().getSimpleName();
267  }
268}
