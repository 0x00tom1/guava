125  private static final class FileByteSource extends
126      ByteSource
127  {
128
129    private final File file;
130
131    private FileByteSource(File file) {
132      this.file = checkNotNull(file);
133    }
134
135    @Override
136    public FileInputStream openStream() throws IOException {
137      return new FileInputStream(file);
138    }
139
140    @Override
141    public Optional<Long> sizeIfKnown() {
142      if (file.isFile()) {
143        return Optional.of(file.length());
144      } else {
145        return Optional.absent();
146      }
147    }
148
149    @Override
150    public long size() throws IOException {
151      if (!file.isFile()) {
152        throw new FileNotFoundException(file.toString());
153      }
154      return file.length();
155    }
156
157    @Override
158    public byte[] read() throws IOException {
159      Closer closer = Closer.create();
160      try {
161        FileInputStream in = closer.register(openStream());
162        return ByteStreams.toByteArray(in, in.getChannel().size());
163      } catch (Throwable e) {
164        throw closer.rethrow(e);
165      } finally {
166        closer.close();
167      }
168    }
169
170    @Override
171    public String toString() {
172      return "Files.asByteSource(" + file + ")";
173    }
174  }

373  /**
374   * Returns true if the given files exist, are not directories, and contain the same bytes.
375   *
376   * @throws IOException if an I/O error occurs
377   */
378  public static boolean equal(File file1, File file2) throws IOException {
379    checkNotNull(file1);
380    checkNotNull(file2);
381    if (file1 == file2 || file1.equals(file2)) {
382      return true;
383    }
384
385    /*
386     * Some operating systems may return zero as the length for files denoting system-dependent
387     * entities such as devices or pipes, in which case we must fall back on comparing the bytes
388     * directly.
389     */
390    long len1 = file1.length();
391    long len2 = file2.length();
392    if (len1 != 0 && len2 != 0 && len1 != len2) {
393      return false;
394    }
395    return asByteSource(file1).contentEquals(asByteSource(file2));
396  }

398  /**
399   * Atomically creates a new directory somewhere beneath the system's temporary directory (as
400   * defined by the {@code java.io.tmpdir} system property), and returns its name.
401   *
402   * <p>Use this method instead of {@link File#createTempFile(String, String)} when you wish to
403   * create a directory, not a regular file. A common pitfall is to call {@code createTempFile},
404   * delete the file and create a directory in its place, but this leads a race condition which can
405   * be exploited to create security vulnerabilities, especially when executable files are to be
406   * written into the directory.
407   *
408   * <p>Depending on the environment that this code is run in, the system temporary directory (and
409   * thus the directory this method creates) may be more visible that a program would like - files
410   * written to this directory may be read or overwritten by hostile programs running on the same
411   * machine.
412   *
413   * <p>This method assumes that the temporary volume is writable, has free inodes and free blocks,
414   * and that it will not be called thousands of times per second.
415   *
416   * <p><b>{@link java.nio.file.Path} equivalent:</b> {@link
417   * java.nio.file.Files#createTempDirectory}.
418   *
419   * @return the newly-created directory
420   * @throws IllegalStateException if the directory could not be created
421   * @deprecated For Android users, see the <a
422   *     href="https://developer.android.com/training/data-storage" target="_blank">Data and File
423   *     Storage overview</a> to select an appropriate temporary directory (perhaps {@code
424   *     context.getCacheDir()}). For developers on Java 7 or later, use {@link
425   *     java.nio.file.Files#createTempDirectory}, transforming it to a {@link File} using {@link
426   *     java.nio.file.Path#toFile() toFile()} if needed.
427   */
428  @Beta
429  @Deprecated
430  @J2ObjCIncompatible
431  public static File createTempDir() {
432    File baseDir = new File(System.getProperty("java.io.tmpdir"));
433    @SuppressWarnings("GoodTime") // reading system time without TimeSource
434    String baseName = System.currentTimeMillis() + "-";
435
436    for (int counter = 0; counter < TEMP_DIR_ATTEMPTS; counter++) {
437      File tempDir = new File(baseDir, baseName + counter);
438      if (tempDir.mkdir()) {
439        return tempDir;
440      }
441    }
442    throw new IllegalStateException(
443        "Failed to create directory within "
444            + TEMP_DIR_ATTEMPTS
445            + " attempts (tried "
446            + baseName
447            + "0 to "
448            + baseName
449            + (TEMP_DIR_ATTEMPTS - 1)
450            + ')');
451  }

465    }

468  /**
469   * Creates any necessary but nonexistent parent directories of the specified file. Note that if
470   * this operation fails it may have succeeded in creating some (but not all) of the necessary
471   * parent directories.
472   *
473   * @throws IOException if an I/O error occurs, or if any necessary but nonexistent parent
474   *     directories of the specified file could not be created.
475   * @since 4.0
476   */
477  public static void createParentDirs(File file) throws IOException {
478    checkNotNull(file);
479    File parent = file.getCanonicalFile().getParentFile();
480    if (parent == null) {
481      /*
482       * The given directory is a filesystem root. All zero of its ancestors exist. This doesn't
483       * mean that the root itself exists -- consider x:\ on a Windows machine without such a drive
484       * -- or even that the caller can create it, but this method makes no such guarantees even for
485       * non-root files.
486       */
487      return;
488    }
489    parent.mkdirs();
490    if (!parent.isDirectory()) {
491      throw new IOException("Unable to create parent directories of " + file);
492    }
493  }

495  /**
496   * Moves a file from one path to another. This method can rename a file and/or move it to a
497   * different directory. In either case {@code to} must be the target path for the file itself; not
498   * just the new name for the file or the path to the new parent directory.
499   *
500   * <p><b>{@link java.nio.file.Path} equivalent:</b> {@link java.nio.file.Files#move}.
501   *
502   * @param from the source file
503   * @param to the destination file
504   * @throws IOException if an I/O error occurs
505   * @throws IllegalArgumentException if {@code from.equals(to)}
506   */
507  public static void move(File from, File to) throws IOException {
508    checkNotNull(from);
509    checkNotNull(to);
510    checkArgument(!from.equals(to), "Source %s and destination %s must be different", from, to);
511
512    if (!from.renameTo(to)) {
513      copy(from, to);
514      if (!from.delete()) {
515        if (!to.delete()) {
516          throw new IOException("Unable to delete " + to);
517        }
518        throw new IOException("Unable to delete " + from);
519      }
520    }
521  }

688  /**
689   * Maps a file in to memory as per {@link FileChannel#map(java.nio.channels.FileChannel.MapMode,
690   * long, long)} using the requested {@link MapMode}.
691   *
692   * <p>Files are mapped from offset 0 to {@code size}.
693   *
694   * <p>If the mode is {@link MapMode#READ_WRITE} and the file does not exist, it will be created
695   * with the requested {@code size}. Thus this method is useful for creating memory mapped files
696   * which do not yet exist.
697   *
698   * <p>This only works for files ≤ {@link Integer#MAX_VALUE} bytes.
699   *
700   * @param file the file to map
701   * @param mode the mode to use when mapping {@code file}
702   * @return a buffer reflecting {@code file}
703   * @throws IOException if an I/O error occurs
704   * @see FileChannel#map(MapMode, long, long)
705   * @since 2.0
706   */
707  public static MappedByteBuffer map(File file, MapMode mode, long size) throws IOException {
708    checkArgument(size >= 0, "size (%s) may not be negative", size);
709    return mapInternal(file, mode, size);
710  }
711
712  private static MappedByteBuffer mapInternal(File file, MapMode mode, long size)
713      throws IOException {
714    checkNotNull(file);
715    checkNotNull(mode);
716
717    Closer closer = Closer.create();
718    try {
719      RandomAccessFile raf =
720          closer.register(new RandomAccessFile(file, mode == MapMode.READ_ONLY ? "r" : "rw"));
721      FileChannel channel = closer.register(raf.getChannel());
722      return channel.map(mode, 0, size == -1 ? channel.size() : size);
723    } catch (Throwable e) {
724      throw closer.rethrow(e);
725    } finally {
726      closer.close();
727    }
728  }

730  /**
731   * Returns the lexically cleaned form of the path name, <i>usually</i> (but not always) equivalent
732   * to the original. The following heuristics are used:
733   *
734   * <ul>
735   *   <li>empty string becomes .
736   *   <li>. stays as .
737   *   <li>fold out ./
738   *   <li>fold out ../ when possible
739   *   <li>collapse multiple slashes
740   *   <li>delete trailing slashes (unless the path is just "/")
741   * </ul>
742   *
743   * <p>These heuristics do not always match the behavior of the filesystem. In particular, consider
744   * the path {@code a/../b}, which {@code simplifyPath} will change to {@code b}. If {@code a} is a
745   * symlink to {@code x}, {@code a/../b} may refer to a sibling of {@code x}, rather than the
746   * sibling of {@code a} referred to by {@code b}.
747   *
748   * @since 11.0
749   */
750  public static String simplifyPath(String pathname) {
751    checkNotNull(pathname);
752    if (pathname.length() == 0) {
753      return ".";
754    }
755
756    // split the path apart
757    Iterable<String> components = Splitter.on('/').omitEmptyStrings().split(pathname);
758    List<String> path = new ArrayList<>();
759
760    // resolve ., .., and //
761    for (String component : components) {
762      switch (component) {
763        case ".":
764          continue;
765        case "..":
766          if (path.size() > 0 && !path.get(path.size() - 1).equals("..")) {
767            path.remove(path.size() - 1);
768          } else {
769            path.add("..");
770          }
771          break;
772        default:
773          path.add(component);
774          break;
775      }
776    }
777
778    // put it back together
779    String result = Joiner.on('/').join(path);
780    if (pathname.charAt(0) == '/') {
781      result = "/" + result;
782    }
783
784    while (result.startsWith("/../")) {
785      result = result.substring(3);
786    }
787    if (result.equals("/..")) {
788      result = "/";
789    } else if ("".equals(result)) {
790      result = ".";
791    }
792
793    return result;
794  }

796  /**
797   * Returns the <a href="http://en.wikipedia.org/wiki/Filename_extension">file extension</a> for
798   * the given file name, or the empty string if the file has no extension. The result does not
799   * include the '{@code .}'.
800   *
801   * <p><b>Note:</b> This method simply returns everything after the last '{@code .}' in the file's
802   * name as determined by {@link File#getName}. It does not account for any filesystem-specific
803   * behavior that the {@link File} API does not already account for. For example, on NTFS it will
804   * report {@code "txt"} as the extension for the filename {@code "foo.exe:.txt"} even though NTFS
805   * will drop the {@code ":.txt"} part of the name when the file is actually created on the
806   * filesystem due to NTFS's <a href="https://goo.gl/vTpJi4">Alternate Data Streams</a>.
807   *
808   * @since 11.0
809   */
810  public static String getFileExtension(String fullName) {
811    checkNotNull(fullName);
812    String fileName = new File(fullName).getName();
813    int dotIndex = fileName.lastIndexOf('.');
814    return (dotIndex == -1) ? "" : fileName.substring(dotIndex + 1);
815  }

817  /**
818   * Returns the file name without its <a
819   * href="http://en.wikipedia.org/wiki/Filename_extension">file extension</a> or path. This is
820   * similar to the {@code basename} unix command. The result does not include the '{@code .}'.
821   *
822   * @param file The name of the file to trim the extension from. This can be either a fully
823   *     qualified file name (including a path) or just a file name.
824   * @return The file name without its path or extension.
825   * @since 14.0
826   */
827  public static String getNameWithoutExtension(String file) {
828    checkNotNull(file);
829    String fileName = new File(file).getName();
830    int dotIndex = fileName.lastIndexOf('.');
831    return (dotIndex == -1) ? fileName : fileName.substring(0, dotIndex);
832  }

834  /**
835   * Returns a {@link Traverser} instance for the file and directory tree. The returned traverser
836   * starts from a {@link File} and will return all files and directories it encounters.
837   *
838   * <p><b>Warning:</b> {@code File} provides no support for symbolic links, and as such there is no
839   * way to ensure that a symbolic link to a directory is not followed when traversing the tree. In
840   * this case, iterables created by this traverser could contain files that are outside of the
841   * given directory or even be infinite if there is a symbolic link loop.
842   *
843   * <p>If available, consider using {@link MoreFiles#fileTraverser()} instead. It behaves the same
844   * except that it doesn't follow symbolic links and returns {@code Path} instances.
845   *
846   * <p>If the {@link File} passed to one of the {@link Traverser} methods does not exist or is not
847   * a directory, no exception will be thrown and the returned {@link Iterable} will contain a
848   * single element: that file.
849   *
850   * <p>Example: {@code Files.fileTraverser().depthFirstPreOrder(new File("/"))} may return files
851   * with the following paths: {@code ["/", "/etc", "/etc/config.txt", "/etc/fonts", "/home",
852   * "/home/alice", ...]}
853   *
854   * @since 23.5
855   */
856  public static Traverser<File> fileTraverser() {
857    return Traverser.forTree(FILE_TREE);
858  }
859
860  private static final SuccessorsFunction<File> FILE_TREE =
861      new SuccessorsFunction<File>() {
862        @Override
863        public Iterable<File> successors(File file) {
864          // check isDirectory() just because it may be faster than listFiles() on a non-directory
865          if (file.isDirectory()) {
866            File[] files = file.listFiles();
867            if (files != null) {
868              return Collections.unmodifiableList(Arrays.asList(files));
869            }
870          }
871
872          return ImmutableList.of();
873        }
874      };
