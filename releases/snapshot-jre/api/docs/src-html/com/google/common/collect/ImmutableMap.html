053/**
054 * A {@link Map} whose contents will never change, with many other important properties detailed at
055 * {@link ImmutableCollection}.
056 *
057 * <p>See the Guava User Guide article on <a href=
058 * "https://github.com/google/guava/wiki/ImmutableCollectionsExplained"> immutable collections</a>.
059 *
060 * @author Jesse Wilson
061 * @author Kevin Bourrillion
062 * @since 2.0

378  /**
379   * A builder for creating immutable map instances, especially {@code public static final} maps
380   * ("constant maps"). Example:
381   *
382   * <pre>{@code
383   * static final ImmutableMap<String, Integer> WORD_TO_INT =
384   *     new ImmutableMap.Builder<String, Integer>()
385   *         .put("one", 1)
386   *         .put("two", 2)
387   *         .put("three", 3)
388   *         .buildOrThrow();
389   * }</pre>
390   *
391   * <p>For <i>small</i> immutable maps, the {@code ImmutableMap.of()} methods are even more
392   * convenient.
393   *
394   * <p>By default, a {@code Builder} will generate maps that iterate over entries in the order they
395   * were inserted into the builder, equivalently to {@code LinkedHashMap}. For example, in the
396   * above example, {@code WORD_TO_INT.entrySet()} is guaranteed to iterate over the entries in the
397   * order {@code "one"=1, "two"=2, "three"=3}, and {@code keySet()} and {@code values()} respect
398   * the same order. If you want a different order, consider using {@link ImmutableSortedMap} to
399   * sort by keys, or call {@link #orderEntriesByValue(Comparator)}, which changes this builder to
400   * sort entries by value.
401   *
402   * <p>Builder instances can be reused - it is safe to call {@link #buildOrThrow} multiple times to
403   * build multiple maps in series. Each map is a superset of the maps created before it.
404   *
405   * @since 2.0
406   */
407  @DoNotMock
408  public static class Builder<K, V> {
409    @CheckForNull Comparator<? super V> valueComparator;
410    @Nullable Entry<K, V>[] entries;
411    int size;
412    boolean entriesUsed;
413
414    /**
415     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
416     * ImmutableMap#builder}.
417     */
418    public Builder() {
419      this(ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY);
420    }
421
422    @SuppressWarnings({"unchecked", "rawtypes"})
423    Builder(int initialCapacity) {
424      this.entries = new @Nullable Entry[initialCapacity];
425      this.size = 0;
426      this.entriesUsed = false;
427    }
428
429    private void ensureCapacity(int minCapacity) {
430      if (minCapacity > entries.length) {
431        entries =
432            Arrays.copyOf(
433                entries, ImmutableCollection.Builder.expandedCapacity(entries.length, minCapacity));
434        entriesUsed = false;
435      }
436    }
437
438    /**
439     * Associates {@code key} with {@code value} in the built map. Duplicate keys are not allowed,
440     * and will cause {@link #build} to fail.
441     */
442    @CanIgnoreReturnValue
443    public Builder<K, V> put(K key, V value) {
444      ensureCapacity(size + 1);
445      Entry<K, V> entry = entryOf(key, value);
446      // don't inline this: we want to fail atomically if key or value is null
447      entries[size++] = entry;
448      return this;
449    }
450
451    /**
452     * Adds the given {@code entry} to the map, making it immutable if necessary. Duplicate keys are
453     * not allowed, and will cause {@link #build} to fail.
454     *
455     * @since 11.0
456     */
457    @CanIgnoreReturnValue
458    public Builder<K, V> put(Entry<? extends K, ? extends V> entry) {
459      return put(entry.getKey(), entry.getValue());
460    }
461
462    /**
463     * Associates all of the given map's keys and values in the built map. Duplicate keys are not
464     * allowed, and will cause {@link #build} to fail.
465     *
466     * @throws NullPointerException if any key or value in {@code map} is null
467     */
468    @CanIgnoreReturnValue
469    public Builder<K, V> putAll(Map<? extends K, ? extends V> map) {
470      return putAll(map.entrySet());
471    }
472
473    /**
474     * Adds all of the given entries to the built map. Duplicate keys are not allowed, and will
475     * cause {@link #build} to fail.
476     *
477     * @throws NullPointerException if any key, value, or entry is null
478     * @since 19.0
479     */
480    @CanIgnoreReturnValue
481    @Beta
482    public Builder<K, V> putAll(Iterable<? extends Entry<? extends K, ? extends V>> entries) {
483      if (entries instanceof Collection) {
484        ensureCapacity(size + ((Collection<?>) entries).size());
485      }
486      for (Entry<? extends K, ? extends V> entry : entries) {
487        put(entry);
488      }
489      return this;
490    }
491
492    /**
493     * Configures this {@code Builder} to order entries by value according to the specified
494     * comparator.
495     *
496     * <p>The sort order is stable, that is, if two entries have values that compare as equivalent,
497     * the entry that was inserted first will be first in the built map's iteration order.
498     *
499     * @throws IllegalStateException if this method was already called
500     * @since 19.0
501     */
502    @CanIgnoreReturnValue
503    @Beta
504    public Builder<K, V> orderEntriesByValue(Comparator<? super V> valueComparator) {
505      checkState(this.valueComparator == null, "valueComparator was already set");
506      this.valueComparator = checkNotNull(valueComparator, "valueComparator");
507      return this;
508    }
509
510    @CanIgnoreReturnValue
511    Builder<K, V> combine(Builder<K, V> other) {
512      checkNotNull(other);
513      ensureCapacity(this.size + other.size);
514      System.arraycopy(other.entries, 0, this.entries, this.size, other.size);
515      this.size += other.size;
516      return this;
517    }

524    /**
525     * Returns a newly-created immutable map. The iteration order of the returned map is the order
526     * in which entries were inserted into the builder, unless {@link #orderEntriesByValue} was
527     * called, in which case entries are sorted by value.
528     *
529     * <p>Prefer the equivalent method {@link #buildOrThrow()} to make it explicit that the method
530     * will throw an exception if there are duplicate keys. The {@code build()} method will soon be
531     * deprecated.
532     *
533     * @throws IllegalArgumentException if duplicate keys were added
534     */
535    public ImmutableMap<K, V> build() {
536      return buildOrThrow();
537    }
538
539    /**
540     * Returns a newly-created immutable map, or throws an exception if any key was added more than
541     * once. The iteration order of the returned map is the order in which entries were inserted
542     * into the builder, unless {@link #orderEntriesByValue} was called, in which case entries are
543     * sorted by value.
544     *
545     * @throws IllegalArgumentException if duplicate keys were added
546     * @since 31.0
547     */
548    public ImmutableMap<K, V> buildOrThrow() {
549      /*
550       * If entries is full, or if hash flooding is detected, then this implementation may end up
551       * using the entries array directly and writing over the entry objects with non-terminal
552       * entries, but this is safe; if this Builder is used further, it will grow the entries array
553       * (so it can't affect the original array), and future build() calls will always copy any
554       * entry objects that cannot be safely reused.
555       */
556      if (valueComparator != null) {
557        if (entriesUsed) {
558          entries = Arrays.copyOf(entries, size);
559        }
560        Arrays.sort(
561            entries, 0, size, Ordering.from(valueComparator).onResultOf(Maps.<V>valueFunction()));
562      }
563      switch (size) {
564        case 0:
565          return of();
566        case 1:
567          // requireNonNull is safe because the first `size` elements have been filled in.
568          Entry<K, V> onlyEntry = requireNonNull(entries[0]);
569          return of(onlyEntry.getKey(), onlyEntry.getValue());
570        default:
571          entriesUsed = true;
572          return RegularImmutableMap.fromEntryArray(size, entries);
573      }
574    }
575
576    @VisibleForTesting // only for testing JDK backed implementation
577    ImmutableMap<K, V> buildJdkBacked() {
578      checkState(
579          valueComparator == null, "buildJdkBacked is only for testing; can't use valueComparator");
580      switch (size) {
581        case 0:
582          return of();
583        case 1:
584          // requireNonNull is safe because the first `size` elements have been filled in.
585          Entry<K, V> onlyEntry = requireNonNull(entries[0]);
586          return of(onlyEntry.getKey(), onlyEntry.getValue());
587        default:
588          entriesUsed = true;
589          return JdkBackedImmutableMap.create(size, entries);
590      }
591    }
592  }

594  /**
595   * Returns an immutable map containing the same entries as {@code map}. The returned map iterates
596   * over entries in the same order as the {@code entrySet} of the original map. If {@code map}
597   * somehow contains entries with duplicate keys (for example, if it is a {@code SortedMap} whose
598   * comparator is not <i>consistent with equals</i>), the results of this method are undefined.
599   *
600   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
601   * safe to do so. The exact circumstances under which a copy will or will not be performed are
602   * undocumented and subject to change.
603   *
604   * @throws NullPointerException if any key or value in {@code map} is null
605   */
606  public static <K, V> ImmutableMap<K, V> copyOf(Map<? extends K, ? extends V> map) {
607    if ((map instanceof ImmutableMap) && !(map instanceof SortedMap)) {
608      @SuppressWarnings("unchecked") // safe since map is not writable
609      ImmutableMap<K, V> kvMap = (ImmutableMap<K, V>) map;
610      if (!kvMap.isPartialView()) {
611        return kvMap;
612      }
613    } else if (map instanceof EnumMap) {
614      @SuppressWarnings("unchecked") // safe since map is not writable
615      ImmutableMap<K, V> kvMap = (ImmutableMap<K, V>) copyOfEnumMap((EnumMap<?, ?>) map);
616      return kvMap;
617    }
618    return copyOf(map.entrySet());
619  }

621  /**
622   * Returns an immutable map containing the specified entries. The returned map iterates over
623   * entries in the same order as the original iterable.
624   *
625   * @throws NullPointerException if any key, value, or entry is null
626   * @throws IllegalArgumentException if two entries have the same key
627   * @since 19.0
628   */
629  @Beta
630  public static <K, V> ImmutableMap<K, V> copyOf(
631      Iterable<? extends Entry<? extends K, ? extends V>> entries) {
632    @SuppressWarnings("unchecked") // we'll only be using getKey and getValue, which are covariant
633    Entry<K, V>[] entryArray = (Entry<K, V>[]) Iterables.toArray(entries, EMPTY_ENTRY_ARRAY);
634    switch (entryArray.length) {
635      case 0:
636        return of();
637      case 1:
638        // requireNonNull is safe because the first `size` elements have been filled in.
639        Entry<K, V> onlyEntry = requireNonNull(entryArray[0]);
640        return of(onlyEntry.getKey(), onlyEntry.getValue());
641      default:
642        /*
643         * The current implementation will end up using entryArray directly, though it will write
644         * over the (arbitrary, potentially mutable) Entry objects actually stored in entryArray.

646        return RegularImmutableMap.fromEntries(entryArray);

1096  /**
1097   * Serialized type for all ImmutableMap instances. It captures the logical contents and they are
1098   * reconstructed using public factory methods. This ensures that the implementation types remain
1099   * as implementation details.
1100   */
1101  static class SerializedForm<K, V> implements Serializable {
1102    // This object retains references to collections returned by keySet() and value(). This saves
1103    // bytes when the both the map and its keySet or value collection are written to the same
1104    // instance of ObjectOutputStream.
1105
1106    // TODO(b/160980469): remove support for the old serialization format after some time
1107    private static final boolean USE_LEGACY_SERIALIZATION = true;
1108
1109    private final Object keys;
1110    private final Object values;
1111
1112    SerializedForm(ImmutableMap<K, V> map) {
1113      if (USE_LEGACY_SERIALIZATION) {
1114        Object[] keys = new Object[map.size()];
1115        Object[] values = new Object[map.size()];
1116        int i = 0;
1117        // "extends Object" works around https://github.com/typetools/checker-framework/issues/3013
1118        for (Entry<? extends Object, ? extends Object> entry : map.entrySet()) {
1119          keys[i] = entry.getKey();
1120          values[i] = entry.getValue();
1121          i++;
1122        }
1123        this.keys = keys;
1124        this.values = values;
1125        return;
1126      }
1127      this.keys = map.keySet();
1128      this.values = map.values();
1129    }
1130
1131    @SuppressWarnings("unchecked")
1132    final Object readResolve() {
1133      if (!(this.keys instanceof ImmutableSet)) {
1134        return legacyReadResolve();
1135      }
1136
1137      ImmutableSet<K> keySet = (ImmutableSet<K>) this.keys;
1138      ImmutableCollection<V> values = (ImmutableCollection<V>) this.values;
1139
1140      Builder<K, V> builder = makeBuilder(keySet.size());
1141
1142      UnmodifiableIterator<K> keyIter = keySet.iterator();
1143      UnmodifiableIterator<V> valueIter = values.iterator();
1144
1145      while (keyIter.hasNext()) {
1146        builder.put(keyIter.next(), valueIter.next());
1147      }
1148
1149      return builder.build();
1150    }
1151
1152    @SuppressWarnings("unchecked")
1153    final Object legacyReadResolve() {
1154      K[] keys = (K[]) this.keys;
1155      V[] values = (V[]) this.values;
1156
1157      Builder<K, V> builder = makeBuilder(keys.length);
1158
1159      for (int i = 0; i < keys.length; i++) {
1160        builder.put(keys[i], values[i]);
1161      }
1162      return builder.build();
1163    }
1164
1165    /**
1166     * Returns a builder that builds the unserialized type. Subclasses should override this method.
1167     */
1168    Builder<K, V> makeBuilder(int size) {
1169      return new Builder<>(size);
1170    }
1171
1172    private static final long serialVersionUID = 0;
1173  }

1175  /**
1176   * Returns a serializable form of this object. Non-public subclasses should not override this
1177   * method. Publicly-accessible subclasses must override this method and should return a subclass
1178   * of SerializedForm whose readResolve() method returns objects of the subclass type.
1179   */
1180  Object writeReplace() {
1181    return new SerializedForm<>(this);
