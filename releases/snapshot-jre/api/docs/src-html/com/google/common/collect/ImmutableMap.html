054/**
055 * A {@link Map} whose contents will never change, with many other important properties detailed at
056 * {@link ImmutableCollection}.
057 *
058 * <p>See the Guava User Guide article on <a href=
059 * "https://github.com/google/guava/wiki/ImmutableCollectionsExplained"> immutable collections</a>.
060 *
061 * @author Jesse Wilson
062 * @author Kevin Bourrillion
063 * @since 2.0
064 */
065@DoNotMock("Use ImmutableMap.of or another implementation")
066@GwtCompatible(serializable = true, emulated = true)
067@SuppressWarnings("serial") // we're overriding default serialization
068@ElementTypesAreNonnullByDefault
069public abstract class ImmutableMap<K, V> implements Map<K, V>, Serializable {

072   * Returns a {@link Collector} that accumulates elements into an {@code ImmutableMap} whose keys
073   * and values are the result of applying the provided mapping functions to the input elements.
074   * Entries appear in the result {@code ImmutableMap} in encounter order.
075   *
076   * <p>If the mapped keys contain duplicates (according to {@link Object#equals(Object)}, an {@code
077   * IllegalArgumentException} is thrown when the collection operation is performed. (This differs
078   * from the {@code Collector} returned by {@link Collectors#toMap(Function, Function)}, which
079   * throws an {@code IllegalStateException}.)
080   *
081   * @since 21.0
082   */
083  public static <T extends @Nullable Object, K, V>
084      Collector<T, ?, ImmutableMap<K, V>> toImmutableMap(
085          Function<? super T, ? extends K> keyFunction,
086          Function<? super T, ? extends V> valueFunction) {
087    return CollectCollectors.toImmutableMap(keyFunction, valueFunction);
088  }

221  /**
222   * A builder for creating immutable map instances, especially {@code public static final} maps
223   * ("constant maps"). Example:
224   *
225   * <pre>{@code
226   * static final ImmutableMap<String, Integer> WORD_TO_INT =
227   *     new ImmutableMap.Builder<String, Integer>()
228   *         .put("one", 1)
229   *         .put("two", 2)
230   *         .put("three", 3)
231   *         .build();
232   * }</pre>
233   *
234   * <p>For <i>small</i> immutable maps, the {@code ImmutableMap.of()} methods are even more
235   * convenient.
236   *
237   * <p>By default, a {@code Builder} will generate maps that iterate over entries in the order they
238   * were inserted into the builder, equivalently to {@code LinkedHashMap}. For example, in the
239   * above example, {@code WORD_TO_INT.entrySet()} is guaranteed to iterate over the entries in the
240   * order {@code "one"=1, "two"=2, "three"=3}, and {@code keySet()} and {@code values()} respect
241   * the same order. If you want a different order, consider using {@link ImmutableSortedMap} to
242   * sort by keys, or call {@link #orderEntriesByValue(Comparator)}, which changes this builder to
243   * sort entries by value.
244   *
245   * <p>Builder instances can be reused - it is safe to call {@link #build} multiple times to build
246   * multiple maps in series. Each map is a superset of the maps created before it.
247   *
248   * @since 2.0
249   */
250  @DoNotMock
251  public static class Builder<K, V> {
252    @CheckForNull Comparator<? super V> valueComparator;
253    @Nullable Entry<K, V>[] entries;
254    int size;
255    boolean entriesUsed;
256
257    /**
258     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
259     * ImmutableMap#builder}.
260     */
261    public Builder() {
262      this(ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY);
263    }
264
265    @SuppressWarnings({"unchecked", "rawtypes"})
266    Builder(int initialCapacity) {
267      this.entries = new @Nullable Entry[initialCapacity];
268      this.size = 0;
269      this.entriesUsed = false;
270    }
271
272    private void ensureCapacity(int minCapacity) {
273      if (minCapacity > entries.length) {
274        entries =
275            Arrays.copyOf(
276                entries, ImmutableCollection.Builder.expandedCapacity(entries.length, minCapacity));
277        entriesUsed = false;
278      }
279    }
280
281    /**
282     * Associates {@code key} with {@code value} in the built map. Duplicate keys are not allowed,
283     * and will cause {@link #build} to fail.
284     */
285    @CanIgnoreReturnValue
286    public Builder<K, V> put(K key, V value) {
287      ensureCapacity(size + 1);
288      Entry<K, V> entry = entryOf(key, value);
289      // don't inline this: we want to fail atomically if key or value is null
290      entries[size++] = entry;
291      return this;
292    }
293
294    /**
295     * Adds the given {@code entry} to the map, making it immutable if necessary. Duplicate keys are
296     * not allowed, and will cause {@link #build} to fail.
297     *
298     * @since 11.0
299     */
300    @CanIgnoreReturnValue
301    public Builder<K, V> put(Entry<? extends K, ? extends V> entry) {
302      return put(entry.getKey(), entry.getValue());
303    }
304
305    /**
306     * Associates all of the given map's keys and values in the built map. Duplicate keys are not
307     * allowed, and will cause {@link #build} to fail.
308     *
309     * @throws NullPointerException if any key or value in {@code map} is null
310     */
311    @CanIgnoreReturnValue
312    public Builder<K, V> putAll(Map<? extends K, ? extends V> map) {
313      return putAll(map.entrySet());
314    }
315
316    /**
317     * Adds all of the given entries to the built map. Duplicate keys are not allowed, and will
318     * cause {@link #build} to fail.
319     *
320     * @throws NullPointerException if any key, value, or entry is null
321     * @since 19.0
322     */
323    @CanIgnoreReturnValue
324    @Beta
325    public Builder<K, V> putAll(Iterable<? extends Entry<? extends K, ? extends V>> entries) {
326      if (entries instanceof Collection) {
327        ensureCapacity(size + ((Collection<?>) entries).size());
328      }
329      for (Entry<? extends K, ? extends V> entry : entries) {
330        put(entry);
331      }
332      return this;
333    }
334
335    /**
336     * Configures this {@code Builder} to order entries by value according to the specified
337     * comparator.
338     *
339     * <p>The sort order is stable, that is, if two entries have values that compare as equivalent,
340     * the entry that was inserted first will be first in the built map's iteration order.
341     *
342     * @throws IllegalStateException if this method was already called
343     * @since 19.0
344     */
345    @CanIgnoreReturnValue
346    @Beta
347    public Builder<K, V> orderEntriesByValue(Comparator<? super V> valueComparator) {
348      checkState(this.valueComparator == null, "valueComparator was already set");
349      this.valueComparator = checkNotNull(valueComparator, "valueComparator");
350      return this;
351    }
352
353    @CanIgnoreReturnValue
354    Builder<K, V> combine(Builder<K, V> other) {
355      checkNotNull(other);
356      ensureCapacity(this.size + other.size);
357      System.arraycopy(other.entries, 0, this.entries, this.size, other.size);
358      this.size += other.size;
359      return this;
360    }

367    /**
368     * Returns a newly-created immutable map. The iteration order of the returned map is the order
369     * in which entries were inserted into the builder, unless {@link #orderEntriesByValue} was
370     * called, in which case entries are sorted by value.
371     *
372     * @throws IllegalArgumentException if duplicate keys were added
373     */
374    public ImmutableMap<K, V> build() {
375      /*
376       * If entries is full, or if hash flooding is detected, then this implementation may end up
377       * using the entries array directly and writing over the entry objects with non-terminal
378       * entries, but this is safe; if this Builder is used further, it will grow the entries array
379       * (so it can't affect the original array), and future build() calls will always copy any
380       * entry objects that cannot be safely reused.
381       */
382      if (valueComparator != null) {
383        if (entriesUsed) {
384          entries = Arrays.copyOf(entries, size);
385        }
386        Arrays.sort(
387            entries, 0, size, Ordering.from(valueComparator).onResultOf(Maps.<V>valueFunction()));
388      }
389      switch (size) {
390        case 0:
391          return of();
392        case 1:
393          // requireNonNull is safe because the first `size` elements have been filled in.
394          Entry<K, V> onlyEntry = requireNonNull(entries[0]);
395          return of(onlyEntry.getKey(), onlyEntry.getValue());
396        default:
397          entriesUsed = true;
398          return RegularImmutableMap.fromEntryArray(size, entries);
399      }
400    }
401
402    @VisibleForTesting // only for testing JDK backed implementation
403    ImmutableMap<K, V> buildJdkBacked() {
404      checkState(
405          valueComparator == null, "buildJdkBacked is only for testing; can't use valueComparator");
406      switch (size) {
407        case 0:
408          return of();
409        case 1:
410          // requireNonNull is safe because the first `size` elements have been filled in.
411          Entry<K, V> onlyEntry = requireNonNull(entries[0]);
412          return of(onlyEntry.getKey(), onlyEntry.getValue());
413        default:
414          entriesUsed = true;
415          return JdkBackedImmutableMap.create(size, entries);
416      }
417    }
418  }

420  /**
421   * Returns an immutable map containing the same entries as {@code map}. The returned map iterates
422   * over entries in the same order as the {@code entrySet} of the original map. If {@code map}
423   * somehow contains entries with duplicate keys (for example, if it is a {@code SortedMap} whose
424   * comparator is not <i>consistent with equals</i>), the results of this method are undefined.
425   *
426   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
427   * safe to do so. The exact circumstances under which a copy will or will not be performed are
428   * undocumented and subject to change.
429   *
430   * @throws NullPointerException if any key or value in {@code map} is null
431   */
432  public static <K, V> ImmutableMap<K, V> copyOf(Map<? extends K, ? extends V> map) {
433    if ((map instanceof ImmutableMap) && !(map instanceof SortedMap)) {
434      @SuppressWarnings("unchecked") // safe since map is not writable
435      ImmutableMap<K, V> kvMap = (ImmutableMap<K, V>) map;
436      if (!kvMap.isPartialView()) {
437        return kvMap;
438      }
439    } else if (map instanceof EnumMap) {
440      @SuppressWarnings("unchecked") // safe since map is not writable
441      ImmutableMap<K, V> kvMap = (ImmutableMap<K, V>) copyOfEnumMap((EnumMap<?, ?>) map);
442      return kvMap;
443    }
444    return copyOf(map.entrySet());
445  }

447  /**
448   * Returns an immutable map containing the specified entries. The returned map iterates over
449   * entries in the same order as the original iterable.
450   *
451   * @throws NullPointerException if any key, value, or entry is null
452   * @throws IllegalArgumentException if two entries have the same key
453   * @since 19.0
454   */
455  @Beta
456  public static <K, V> ImmutableMap<K, V> copyOf(
457      Iterable<? extends Entry<? extends K, ? extends V>> entries) {
458    @SuppressWarnings("unchecked") // we'll only be using getKey and getValue, which are covariant
459    Entry<K, V>[] entryArray = (Entry<K, V>[]) Iterables.toArray(entries, EMPTY_ENTRY_ARRAY);
460    switch (entryArray.length) {
461      case 0:
462        return of();
463      case 1:
464        // requireNonNull is safe because the first `size` elements have been filled in.
465        Entry<K, V> onlyEntry = requireNonNull(entryArray[0]);
466        return of(onlyEntry.getKey(), onlyEntry.getValue());
467      default:
468        /*
469         * The current implementation will end up using entryArray directly, though it will write
470         * over the (arbitrary, potentially mutable) Entry objects actually stored in entryArray.
471         */
472        return RegularImmutableMap.fromEntries(entryArray);
473    }
474  }

476  private static <K extends Enum<K>, V> ImmutableMap<K, V> copyOfEnumMap(
477      EnumMap<K, ? extends V> original) {
478    EnumMap<K, V> copy = new EnumMap<>(original);
479    for (Entry<K, V> entry : copy.entrySet()) {
480      checkEntryNotNull(entry.getKey(), entry.getValue());
481    }
482    return ImmutableEnumMap.asImmutable(copy);
483  }

487  abstract static class IteratorBasedImmutableMap<K, V> extends ImmutableMap<K, V> {
488    abstract UnmodifiableIterator<Entry<K, V>> entryIterator();
489
490    Spliterator<Entry<K, V>> entrySpliterator() {
491      return Spliterators.spliterator(
492          entryIterator(),
493          size(),
494          Spliterator.DISTINCT | Spliterator.NONNULL | Spliterator.IMMUTABLE | Spliterator.ORDERED);
495    }
496
497    @Override
498    ImmutableSet<K> createKeySet() {
499      return new ImmutableMapKeySet<>(this);
500    }
501
502    @Override
503    ImmutableSet<Entry<K, V>> createEntrySet() {
504      class EntrySetImpl extends ImmutableMapEntrySet<K, V> {
505        @Override
506        ImmutableMap<K, V> map() {
507          return IteratorBasedImmutableMap.this;
508        }
509
510        @Override
511        public UnmodifiableIterator<Entry<K, V>> iterator() {
512          return entryIterator();
513        }
514      }
515      return new EntrySetImpl();
516    }
517
518    @Override
519    ImmutableCollection<V> createValues() {
520      return new ImmutableMapValues<>(this);
521    }
522  }

922  /**
923   * Serialized type for all ImmutableMap instances. It captures the logical contents and they are
924   * reconstructed using public factory methods. This ensures that the implementation types remain
925   * as implementation details.
926   */
927  static class SerializedForm<K, V> implements Serializable {
928    // This object retains references to collections returned by keySet() and value(). This saves
929    // bytes when the both the map and its keySet or value collection are written to the same
930    // instance of ObjectOutputStream.
931
932    // TODO(b/160980469): remove support for the old serialization format after some time
933    private static final boolean USE_LEGACY_SERIALIZATION = true;
934
935    private final Object keys;
936    private final Object values;
937
938    SerializedForm(ImmutableMap<K, V> map) {
939      if (USE_LEGACY_SERIALIZATION) {
940        Object[] keys = new Object[map.size()];
941        Object[] values = new Object[map.size()];
942        int i = 0;
943        // "extends Object" works around https://github.com/typetools/checker-framework/issues/3013
944        for (Entry<? extends Object, ? extends Object> entry : map.entrySet()) {
945          keys[i] = entry.getKey();
946          values[i] = entry.getValue();
947          i++;
948        }
949        this.keys = keys;
950        this.values = values;
951        return;
952      }
953      this.keys = map.keySet();
954      this.values = map.values();
955    }
956
957    @SuppressWarnings("unchecked")
958    final Object readResolve() {
959      if (!(this.keys instanceof ImmutableSet)) {
960        return legacyReadResolve();
961      }
962
963      ImmutableSet<K> keySet = (ImmutableSet<K>) this.keys;
964      ImmutableCollection<V> values = (ImmutableCollection<V>) this.values;
965
966      Builder<K, V> builder = makeBuilder(keySet.size());
967
968      UnmodifiableIterator<K> keyIter = keySet.iterator();
969      UnmodifiableIterator<V> valueIter = values.iterator();
970
971      while (keyIter.hasNext()) {
972        builder.put(keyIter.next(), valueIter.next());
973      }
974
975      return builder.build();
976    }
977
978    @SuppressWarnings("unchecked")
979    final Object legacyReadResolve() {
980      K[] keys = (K[]) this.keys;
981      V[] values = (V[]) this.values;
982
983      Builder<K, V> builder = makeBuilder(keys.length);
984
985      for (int i = 0; i < keys.length; i++) {
986        builder.put(keys[i], values[i]);
987      }
988      return builder.build();
989    }
990
991    /**
992     * Returns a builder that builds the unserialized type. Subclasses should override this method.
993     */
994    Builder<K, V> makeBuilder(int size) {
995      return new Builder<>(size);
996    }
997
998    private static final long serialVersionUID = 0;
999  }

1001  /**
1002   * Returns a serializable form of this object. Non-public subclasses should not override this
1003   * method. Publicly-accessible subclasses must override this method and should return a subclass
1004   * of SerializedForm whose readResolve() method returns objects of the subclass type.
1005   */
1006  Object writeReplace() {
1007    return new SerializedForm<>(this);
