031/**
032 * A {@link BiMap} whose contents will never change, with many other important properties detailed
033 * at {@link ImmutableCollection}.
034 *
035 * @author Jared Levy
036 * @since 2.0
037 */
038@GwtCompatible(serializable = true, emulated = true)
039@ElementTypesAreNonnullByDefault
040public abstract class ImmutableBiMap<K, V> extends ImmutableMap<K, V> implements BiMap<K, V> {
041
042  /**
043   * Returns the empty bimap.
044   *
045   * <p><b>Performance note:</b> the instance returned is a singleton.
046   */
047  // Casting to any type is safe because the set will never hold any elements.
048  @SuppressWarnings("unchecked")
049  public static <K, V> ImmutableBiMap<K, V> of() {
050    return (ImmutableBiMap<K, V>) RegularImmutableBiMap.EMPTY;
051  }
052
053  /** Returns an immutable bimap containing a single entry. */
054  public static <K, V> ImmutableBiMap<K, V> of(K k1, V v1) {
055    checkEntryNotNull(k1, v1);
056    return new RegularImmutableBiMap<>(new Object[] {k1, v1}, 1);
057  }
058
059  /**
060   * Returns an immutable map containing the given entries, in order.
061   *
062   * @throws IllegalArgumentException if duplicate keys or values are added
063   */
064  public static <K, V> ImmutableBiMap<K, V> of(K k1, V v1, K k2, V v2) {
065    checkEntryNotNull(k1, v1);
066    checkEntryNotNull(k2, v2);
067    return new RegularImmutableBiMap<K, V>(new Object[] {k1, v1, k2, v2}, 2);
068  }
069
070  /**
071   * Returns an immutable map containing the given entries, in order.
072   *
073   * @throws IllegalArgumentException if duplicate keys or values are added
074   */
075  public static <K, V> ImmutableBiMap<K, V> of(K k1, V v1, K k2, V v2, K k3, V v3) {
076    checkEntryNotNull(k1, v1);
077    checkEntryNotNull(k2, v2);
078    checkEntryNotNull(k3, v3);
079    return new RegularImmutableBiMap<K, V>(new Object[] {k1, v1, k2, v2, k3, v3}, 3);
080  }

082  /**
083   * Returns an immutable map containing the given entries, in order.
084   *
085   * @throws IllegalArgumentException if duplicate keys or values are added
086   */
087  public static <K, V> ImmutableBiMap<K, V> of(K k1, V v1, K k2, V v2, K k3, V v3, K k4, V v4) {
088    checkEntryNotNull(k1, v1);
089    checkEntryNotNull(k2, v2);
090    checkEntryNotNull(k3, v3);
091    checkEntryNotNull(k4, v4);
092    return new RegularImmutableBiMap<K, V>(new Object[] {k1, v1, k2, v2, k3, v3, k4, v4}, 4);
093  }

096   * Returns an immutable map containing the given entries, in order.
097   *
098   * @throws IllegalArgumentException if duplicate keys or values are added
099   */
100  public static <K, V> ImmutableBiMap<K, V> of(
101      K k1, V v1, K k2, V v2, K k3, V v3, K k4, V v4, K k5, V v5) {
102    checkEntryNotNull(k1, v1);
103    checkEntryNotNull(k2, v2);
104    checkEntryNotNull(k3, v3);
105    checkEntryNotNull(k4, v4);
106    checkEntryNotNull(k5, v5);
107    return new RegularImmutableBiMap<K, V>(
108        new Object[] {k1, v1, k2, v2, k3, v3, k4, v4, k5, v5}, 5);
109  }

113  /**
114   * Returns a new builder. The generated builder is equivalent to the builder created by the {@link
115   * Builder} constructor.
116   */
117  public static <K, V> Builder<K, V> builder() {
118    return new Builder<>();
119  }

121  /**
122   * Returns a new builder, expecting the specified number of entries to be added.
123   *
124   * <p>If {@code expectedSize} is exactly the number of entries added to the builder before {@link
125   * Builder#build} is called, the builder is likely to perform better than an unsized {@link
126   * #builder()} would have.
127   *
128   * <p>It is not specified if any performance benefits apply if {@code expectedSize} is close to,
129   * but not exactly, the number of entries added to the builder.
130   *
131   * @since 23.1
132   */
133  @Beta
134  public static <K, V> Builder<K, V> builderWithExpectedSize(int expectedSize) {
135    checkNonnegative(expectedSize, "expectedSize");
136    return new Builder<>(expectedSize);
137  }
138
139  /**
140   * A builder for creating immutable bimap instances, especially {@code public static final} bimaps
141   * ("constant bimaps"). Example:
142   *
143   * <pre>{@code
144   * static final ImmutableBiMap<String, Integer> WORD_TO_INT =
145   *     new ImmutableBiMap.Builder<String, Integer>()
146   *         .put("one", 1)
147   *         .put("two", 2)
148   *         .put("three", 3)
149   *         .build();
150   * }</pre>
151   *
152   * <p>For <i>small</i> immutable bimaps, the {@code ImmutableBiMap.of()} methods are even more
153   * convenient.
154   *
155   * <p>By default, a {@code Builder} will generate bimaps that iterate over entries in the order
156   * they were inserted into the builder. For example, in the above example, {@code
157   * WORD_TO_INT.entrySet()} is guaranteed to iterate over the entries in the order {@code "one"=1,
158   * "two"=2, "three"=3}, and {@code keySet()} and {@code values()} respect the same order. If you
159   * want a different order, consider using {@link #orderEntriesByValue(Comparator)}, which changes
160   * this builder to sort entries by value.
161   *
162   * <p>Builder instances can be reused - it is safe to call {@link #build} multiple times to build
163   * multiple bimaps in series. Each bimap is a superset of the bimaps created before it.
164   *
165   * @since 2.0
166   */
167  public static final class Builder<K, V> extends ImmutableMap.Builder<K, V> {
168    /**
169     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
170     * ImmutableBiMap#builder}.
171     */
172    public Builder() {
173      super();
174    }
175
176    Builder(int size) {
177      super(size);
178    }
179
180    /**
181     * Associates {@code key} with {@code value} in the built bimap. Duplicate keys or values are
182     * not allowed, and will cause {@link #build} to fail.
183     */
184    @CanIgnoreReturnValue
185    @Override
186    public Builder<K, V> put(K key, V value) {
187      super.put(key, value);
188      return this;
189    }
190
191    /**
192     * Adds the given {@code entry} to the bimap. Duplicate keys or values are not allowed, and will
193     * cause {@link #build} to fail.
194     *
195     * @since 19.0
196     */
197    @CanIgnoreReturnValue
198    @Override
199    public Builder<K, V> put(Entry<? extends K, ? extends V> entry) {
200      super.put(entry);
201      return this;
202    }
203
204    /**
205     * Associates all of the given map's keys and values in the built bimap. Duplicate keys or
206     * values are not allowed, and will cause {@link #build} to fail.
207     *
208     * @throws NullPointerException if any key or value in {@code map} is null
209     */
210    @CanIgnoreReturnValue
211    @Override
212    public Builder<K, V> putAll(Map<? extends K, ? extends V> map) {
213      super.putAll(map);
214      return this;
215    }
216
217    /**
218     * Adds all of the given entries to the built bimap. Duplicate keys or values are not allowed,
219     * and will cause {@link #build} to fail.
220     *
221     * @throws NullPointerException if any key, value, or entry is null
222     * @since 19.0
223     */
224    @CanIgnoreReturnValue
225    @Beta
226    @Override
227    public Builder<K, V> putAll(Iterable<? extends Entry<? extends K, ? extends V>> entries) {
228      super.putAll(entries);
229      return this;
230    }
231
232    /**
233     * Configures this {@code Builder} to order entries by value according to the specified
234     * comparator.
235     *
236     * <p>The sort order is stable, that is, if two entries have values that compare as equivalent,
237     * the entry that was inserted first will be first in the built map's iteration order.
238     *
239     * @throws IllegalStateException if this method was already called
240     * @since 19.0
241     */
242    @CanIgnoreReturnValue
243    @Beta
244    @Override
245    public Builder<K, V> orderEntriesByValue(Comparator<? super V> valueComparator) {
246      super.orderEntriesByValue(valueComparator);
247      return this;
248    }
249
250    @Override
251    @CanIgnoreReturnValue
252    Builder<K, V> combine(ImmutableMap.Builder<K, V> builder) {
253      super.combine(builder);
254      return this;
255    }
256
257    /**
258     * Returns a newly-created immutable bimap. The iteration order of the returned bimap is the
259     * order in which entries were inserted into the builder, unless {@link #orderEntriesByValue}
260     * was called, in which case entries are sorted by value.
261     *
262     * @throws IllegalArgumentException if duplicate keys or values were added
263     */
264    @Override
265    public ImmutableBiMap<K, V> build() {
266      if (size == 0) {
267        return of();
268      }
269      sortEntries();
270      entriesUsed = true;
271      return new RegularImmutableBiMap<K, V>(alternatingKeysAndValues, size);
272    }
273  }
274
275  /**
276   * Returns an immutable bimap containing the same entries as {@code map}. If {@code map} somehow
277   * contains entries with duplicate keys (for example, if it is a {@code SortedMap} whose
278   * comparator is not <i>consistent with equals</i>), the results of this method are undefined.
279   *
280   * <p>The returned {@code BiMap} iterates over entries in the same order as the {@code entrySet}
281   * of the original map.
282   *
283   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
284   * safe to do so. The exact circumstances under which a copy will or will not be performed are
285   * undocumented and subject to change.
286   *
287   * @throws IllegalArgumentException if two keys have the same value or two values have the same
288   *     key
289   * @throws NullPointerException if any key or value in {@code map} is null
290   */
291  public static <K, V> ImmutableBiMap<K, V> copyOf(Map<? extends K, ? extends V> map) {
292    if (map instanceof ImmutableBiMap) {
293      @SuppressWarnings("unchecked") // safe since map is not writable
294      ImmutableBiMap<K, V> bimap = (ImmutableBiMap<K, V>) map;
295      // TODO(lowasser): if we need to make a copy of a BiMap because the
296      // forward map is a view, don't make a copy of the non-view delegate map
297      if (!bimap.isPartialView()) {
298        return bimap;
299      }
300    }
301    return copyOf(map.entrySet());
302  }
303
304  /**
305   * Returns an immutable bimap containing the given entries. The returned bimap iterates over
306   * entries in the same order as the original iterable.
307   *
308   * @throws IllegalArgumentException if two keys have the same value or two values have the same
309   *     key
310   * @throws NullPointerException if any key, value, or entry is null
311   * @since 19.0
312   */
313  @Beta
314  public static <K, V> ImmutableBiMap<K, V> copyOf(
315      Iterable<? extends Entry<? extends K, ? extends V>> entries) {
316    int estimatedSize =
317        (entries instanceof Collection)
318            ? ((Collection<?>) entries).size()
319            : ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY;
320    return new Builder<K, V>(estimatedSize).putAll(entries).build();
321  }

339    return inverse().keySet();

362  /**
363   * Serialized type for all ImmutableBiMap instances. It captures the logical contents and they are
364   * reconstructed using public factory methods. This ensures that the implementation types remain
365   * as implementation details.
366   *
367   * <p>Since the bimap is immutable, ImmutableBiMap doesn't require special logic for keeping the
368   * bimap and its inverse in sync during serialization, the way AbstractBiMap does.
369   */
370  private static class SerializedForm<K, V> extends ImmutableMap.SerializedForm<K, V> {
371    SerializedForm(ImmutableBiMap<K, V> bimap) {
372      super(bimap);
373    }
374
375    @Override
376    Builder<K, V> makeBuilder(int size) {
377      return new Builder<>(size);
378    }
379
380    private static final long serialVersionUID = 0;
381  }
382
383  @Override
384  Object writeReplace() {
385    return new SerializedForm<>(this);
386  }
