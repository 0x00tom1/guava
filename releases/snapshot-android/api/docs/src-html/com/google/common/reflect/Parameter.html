015package com.google.common.reflect;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018import static java.util.Objects.requireNonNull;
019
020import com.google.common.annotations.Beta;
021import com.google.common.collect.FluentIterable;
022import com.google.common.collect.ImmutableList;
023import com.google.errorprone.annotations.DoNotCall;
024import java.lang.annotation.Annotation;
025import java.lang.reflect.AnnotatedElement;
026import java.lang.reflect.AnnotatedType;
027import javax.annotation.CheckForNull;
028import org.checkerframework.checker.nullness.qual.Nullable;
029
030/**
031 * Represents a method or constructor parameter.
032 *
033 * <p><b>Note:</b> Since Java 8 introduced {@link java.lang.reflect.Parameter} to represent method
034 * and constructor parameters, this class is no longer necessary. We intend to deprecate it in a
035 * future version.
036 *
037 * @author Ben Yu
038 * @since 14.0
039 */
040@Beta
041@ElementTypesAreNonnullByDefault
042public final class Parameter implements AnnotatedElement {
043
044  private final Invokable<?, ?> declaration;
045  private final int position;
046  private final TypeToken<?> type;
047  private final ImmutableList<Annotation> annotations;
048
049  /**
050   * An {@link AnnotatedType} instance, or {@code null} under Android VMs (possible only when using
051   * the Android flavor of Guava). The field is declared with a type of {@code Object} to avoid
052   * compatibility problems on Android VMs. The corresponding accessor method, however, can have the
053   * more specific return type as long as users are careful to guard calls to it with version checks
054   * or reflection: Android VMs ignore the types of elements that aren't used.
055   */
056  private final @Nullable Object annotatedType;
057
058  Parameter(
059      Invokable<?, ?> declaration,
060      int position,
061      TypeToken<?> type,
062      Annotation[] annotations,
063      @Nullable Object annotatedType) {
064    this.declaration = declaration;
065    this.position = position;
066    this.type = type;
067    this.annotations = ImmutableList.copyOf(annotations);
068    this.annotatedType = annotatedType;
069  }
070
071  /** Returns the type of the parameter. */
072  public TypeToken<?> getType() {
073    return type;
074  }
075
076  /** Returns the {@link Invokable} that declares this parameter. */
077  public Invokable<?, ?> getDeclaringInvokable() {
078    return declaration;
079  }
080
081  @Override
082  public boolean isAnnotationPresent(Class<? extends Annotation> annotationType) {
083    return getAnnotation(annotationType) != null;
084  }
085
086  @Override
087  @CheckForNull
088  public <A extends Annotation> A getAnnotation(Class<A> annotationType) {
089    checkNotNull(annotationType);
090    for (Annotation annotation : annotations) {
091      if (annotationType.isInstance(annotation)) {
092        return annotationType.cast(annotation);
093      }
094    }
095    return null;
096  }
097
098  @Override
099  public Annotation[] getAnnotations() {
100    return getDeclaredAnnotations();
101  }
102
103  /**
104   * @since 18.0
105   */
106  @Override
107  public <A extends Annotation> A[] getAnnotationsByType(Class<A> annotationType) {
108    return getDeclaredAnnotationsByType(annotationType);
109  }
110
111  /** @since 18.0 */
112  @Override
113  public Annotation[] getDeclaredAnnotations() {
114    return annotations.toArray(new Annotation[0]);
115  }
116
117  /**
118   * @since 18.0
119   */
120  @Override
121  @CheckForNull
122  public <A extends Annotation> A getDeclaredAnnotation(Class<A> annotationType) {
123    checkNotNull(annotationType);
124    return FluentIterable.from(annotations).filter(annotationType).first().orNull();
125  }
126
127  /**
128   * @since 18.0
129   */
130  @Override
131  public <A extends Annotation> A[] getDeclaredAnnotationsByType(Class<A> annotationType) {
132    @Nullable
133    A[] result = FluentIterable.from(annotations).filter(annotationType).toArray(annotationType);
134    @SuppressWarnings("nullness") // safe because the input list contains no nulls
135    A[] cast = (A[]) result;
136    return cast;
137  }
138
139  /**
140   * Returns the {@link AnnotatedType} of the parameter.
141   *
142   * <p>This method will fail if run under an Android VM.
143   *
144   * @since NEXT for guava-android (available since 25.1 in guava-jre)
145   * @deprecated This method does not work under Android VMs. It is safe to use from guava-jre, but
146   *     this copy in guava-android is not safe to use.
147   */
148  @SuppressWarnings({"Java7ApiChecker", "AndroidJdkLibsChecker"})
149  @Deprecated
150  @DoNotCall("fails under Android VMs; do not use from guava-android")
151  @IgnoreJRERequirement
152  public AnnotatedType getAnnotatedType() {
153    return requireNonNull((AnnotatedType) annotatedType);
154  }
155
156  @Override
157  public boolean equals(@CheckForNull Object obj) {
158    if (obj instanceof Parameter) {
159      Parameter that = (Parameter) obj;
160      return position == that.position && declaration.equals(that.declaration);
161    }
162    return false;
163  }
164
165  @Override
166  public int hashCode() {
167    return position;
168  }
169
170  @Override
171  public String toString() {
172    return type + " arg" + position;
173  }
174}
