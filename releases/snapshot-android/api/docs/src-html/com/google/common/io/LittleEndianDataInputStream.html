015package com.google.common.io;
016
017import com.google.common.annotations.Beta;
018import com.google.common.annotations.GwtIncompatible;
019import com.google.common.annotations.J2ktIncompatible;
020import com.google.common.base.Preconditions;
021import com.google.common.primitives.Ints;
022import com.google.common.primitives.Longs;
023import com.google.errorprone.annotations.CanIgnoreReturnValue;
024import com.google.errorprone.annotations.DoNotCall;
025import java.io.DataInput;
026import java.io.DataInputStream;
027import java.io.EOFException;
028import java.io.FilterInputStream;
029import java.io.IOException;
030import java.io.InputStream;
031
032/**
033 * An implementation of {@link DataInput} that uses little-endian byte ordering for reading {@code
034 * short}, {@code int}, {@code float}, {@code double}, and {@code long} values.
035 *
036 * <p><b>Note:</b> This class intentionally violates the specification of its supertype {@code
037 * DataInput}, which explicitly requires big-endian byte order.
038 *
039 * @author Chris Nokleberg
040 * @author Keith Bottner
041 * @since 8.0
042 */
043@Beta
044@J2ktIncompatible
045@GwtIncompatible
046@ElementTypesAreNonnullByDefault
047public final class LittleEndianDataInputStream extends FilterInputStream implements DataInput {
048
049  /**
050   * Creates a {@code LittleEndianDataInputStream} that wraps the given stream.
051   *
052   * @param in the stream to delegate to
053   */
054  public LittleEndianDataInputStream(InputStream in) {
055    super(Preconditions.checkNotNull(in));
056  }
057
058  /** This method will throw an {@link UnsupportedOperationException}. */
059  @CanIgnoreReturnValue // to skip a line
060  @Override
061  @DoNotCall("Always throws UnsupportedOperationException")
062  public String readLine() {
063    throw new UnsupportedOperationException("readLine is not supported");
064  }
065
066  @Override
067  public void readFully(byte[] b) throws IOException {
068    ByteStreams.readFully(this, b);
069  }
070
071  @Override
072  public void readFully(byte[] b, int off, int len) throws IOException {
073    ByteStreams.readFully(this, b, off, len);
074  }
075
076  @Override
077  public int skipBytes(int n) throws IOException {
078    return (int) in.skip(n);
079  }
080
081  @CanIgnoreReturnValue // to skip a byte
082  @Override
083  public int readUnsignedByte() throws IOException {
084    int b1 = in.read();
085    if (0 > b1) {
086      throw new EOFException();
087    }
088
089    return b1;
090  }
091
092  /**
093   * Reads an unsigned {@code short} as specified by {@link DataInputStream#readUnsignedShort()},
094   * except using little-endian byte order.
095   *
096   * @return the next two bytes of the input stream, interpreted as an unsigned 16-bit integer in
097   *     little-endian byte order
098   * @throws IOException if an I/O error occurs
099   */
100  @CanIgnoreReturnValue // to skip some bytes
101  @Override
102  public int readUnsignedShort() throws IOException {
103    byte b1 = readAndCheckByte();
104    byte b2 = readAndCheckByte();
105
106    return Ints.fromBytes((byte) 0, (byte) 0, b2, b1);
107  }
108
109  /**
110   * Reads an integer as specified by {@link DataInputStream#readInt()}, except using little-endian
111   * byte order.
112   *
113   * @return the next four bytes of the input stream, interpreted as an {@code int} in little-endian
114   *     byte order
115   * @throws IOException if an I/O error occurs
116   */
117  @CanIgnoreReturnValue // to skip some bytes
118  @Override
119  public int readInt() throws IOException {
120    byte b1 = readAndCheckByte();
121    byte b2 = readAndCheckByte();
122    byte b3 = readAndCheckByte();
123    byte b4 = readAndCheckByte();
124
125    return Ints.fromBytes(b4, b3, b2, b1);
126  }
127
128  /**
129   * Reads a {@code long} as specified by {@link DataInputStream#readLong()}, except using
130   * little-endian byte order.
131   *
132   * @return the next eight bytes of the input stream, interpreted as a {@code long} in
133   *     little-endian byte order
134   * @throws IOException if an I/O error occurs
135   */
136  @CanIgnoreReturnValue // to skip some bytes
137  @Override
138  public long readLong() throws IOException {
139    byte b1 = readAndCheckByte();
140    byte b2 = readAndCheckByte();
141    byte b3 = readAndCheckByte();
142    byte b4 = readAndCheckByte();
143    byte b5 = readAndCheckByte();
144    byte b6 = readAndCheckByte();
145    byte b7 = readAndCheckByte();
146    byte b8 = readAndCheckByte();
147
148    return Longs.fromBytes(b8, b7, b6, b5, b4, b3, b2, b1);
149  }
150
151  /**
152   * Reads a {@code float} as specified by {@link DataInputStream#readFloat()}, except using
153   * little-endian byte order.
154   *
155   * @return the next four bytes of the input stream, interpreted as a {@code float} in
156   *     little-endian byte order
157   * @throws IOException if an I/O error occurs
158   */
159  @CanIgnoreReturnValue // to skip some bytes
160  @Override
161  public float readFloat() throws IOException {
162    return Float.intBitsToFloat(readInt());
163  }
164
165  /**
166   * Reads a {@code double} as specified by {@link DataInputStream#readDouble()}, except using
167   * little-endian byte order.
168   *
169   * @return the next eight bytes of the input stream, interpreted as a {@code double} in
170   *     little-endian byte order
171   * @throws IOException if an I/O error occurs
172   */
173  @CanIgnoreReturnValue // to skip some bytes
174  @Override
175  public double readDouble() throws IOException {
176    return Double.longBitsToDouble(readLong());
177  }
178
179  @CanIgnoreReturnValue // to skip a field
180  @Override
181  public String readUTF() throws IOException {
182    return new DataInputStream(in).readUTF();
183  }
184
185  /**
186   * Reads a {@code short} as specified by {@link DataInputStream#readShort()}, except using
187   * little-endian byte order.
188   *
189   * @return the next two bytes of the input stream, interpreted as a {@code short} in little-endian
190   *     byte order.
191   * @throws IOException if an I/O error occurs.
192   */
193  @CanIgnoreReturnValue // to skip some bytes
194  @Override
195  public short readShort() throws IOException {
196    return (short) readUnsignedShort();
197  }
198
199  /**
200   * Reads a char as specified by {@link DataInputStream#readChar()}, except using little-endian
201   * byte order.
202   *
203   * @return the next two bytes of the input stream, interpreted as a {@code char} in little-endian
204   *     byte order
205   * @throws IOException if an I/O error occurs
206   */
207  @CanIgnoreReturnValue // to skip some bytes
208  @Override
209  public char readChar() throws IOException {
210    return (char) readUnsignedShort();
211  }
212
213  @CanIgnoreReturnValue // to skip a byte
214  @Override
215  public byte readByte() throws IOException {
216    return (byte) readUnsignedByte();
217  }
218
219  @CanIgnoreReturnValue // to skip a byte
220  @Override
221  public boolean readBoolean() throws IOException {
222    return readUnsignedByte() != 0;
223  }
224
225  /**
226   * Reads a byte from the input stream checking that the end of file (EOF) has not been
227   * encountered.
228   *
229   * @return byte read from input
230   * @throws IOException if an error is encountered while reading
231   * @throws EOFException if the end of file (EOF) is encountered.
232   */
233  private byte readAndCheckByte() throws IOException, EOFException {
234    int b1 = in.read();
235
236    if (-1 == b1) {
237      throw new EOFException();
238    }
239
240    return (byte) b1;
241  }
242}
