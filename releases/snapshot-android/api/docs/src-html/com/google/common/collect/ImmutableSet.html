040/**
041 * A {@link Set} whose contents will never change, with many other important properties detailed at
042 * {@link ImmutableCollection}.
043 *
044 * @since 2.0
045 */
046@GwtCompatible(serializable = true, emulated = true)
047@SuppressWarnings("serial") // we're overriding default serialization
048public abstract class ImmutableSet<E> extends ImmutableCollection<E> implements Set<E> {

105  /**
106   * Returns an immutable set containing the given elements, minus duplicates, in the order each was
107   * first specified. That is, if multiple elements are {@linkplain Object#equals equal}, all except
108   * the first are ignored.
109   *
110   * <p>The array {@code others} must not be longer than {@code Integer.MAX_VALUE - 6}.
111   *
112   * @since 3.0 (source-compatible since 2.0)
113   */
114  @SafeVarargs // For Eclipse. For internal javac we have disabled this pointless type of warning.
115  public static <E> ImmutableSet<E> of(E e1, E e2, E e3, E e4, E e5, E e6, E... others) {
116    checkArgument(
117        others.length <= Integer.MAX_VALUE - 6, "the total number of elements must fit in an int");
118    final int paramCount = 6;
119    Object[] elements = new Object[paramCount + others.length];
120    elements[0] = e1;
121    elements[1] = e2;
122    elements[2] = e3;
123    elements[3] = e4;
124    elements[4] = e5;
125    elements[5] = e6;
126    System.arraycopy(others, 0, elements, paramCount, others.length);
127    return construct(elements.length, elements);
128  }

130  /**
131   * Constructs an {@code ImmutableSet} from the first {@code n} elements of the specified array. If
132   * {@code k} is the size of the returned {@code ImmutableSet}, then the unique elements of {@code
133   * elements} will be in the first {@code k} positions, and {@code elements[i] == null} for {@code
134   * k <= i < n}.
135   *
136   * <p>After this method returns, {@code elements} will contain no duplicates, but {@code elements}
137   * may be the real array backing the returned set, so do not modify it further.
138   *
139   * <p>{@code elements} may contain only values of type {@code E}.
140   *
141   * @throws NullPointerException if any of the first {@code n} elements of {@code elements} is null
142   */
143  private static <E> ImmutableSet<E> construct(int n, Object... elements) {
144    switch (n) {
145      case 0:
146        return of();
147      case 1:
148        @SuppressWarnings("unchecked") // safe; elements contains only E's
149        E elem = (E) elements[0];
150        return of(elem);
151      default:
152        // continue below to handle the general case
153    }
154    int tableSize = chooseTableSize(n);
155    Object[] table = new Object[tableSize];
156    int mask = tableSize - 1;
157    int hashCode = 0;
158    int uniques = 0;
159    for (int i = 0; i < n; i++) {
160      Object element = checkElementNotNull(elements[i], i);
161      int hash = element.hashCode();
162      for (int j = Hashing.smear(hash); ; j++) {
163        int index = j & mask;
164        Object value = table[index];
165        if (value == null) {
166          // Came to an empty slot. Put the element here.
167          elements[uniques++] = element;
168          table[index] = element;
169          hashCode += hash;
170          break;
171        } else if (value.equals(element)) {
172          break;
173        }
174      }
175    }
176    Arrays.fill(elements, uniques, n, null);
177    if (uniques == 1) {
178      // There is only one element or elements are all duplicates
179      @SuppressWarnings("unchecked") // we are careful to only pass in E
180      E element = (E) elements[0];
181      return new SingletonImmutableSet<E>(element, hashCode);
182    } else if (chooseTableSize(uniques) < tableSize / 2) {
183      // Resize the table when the array includes too many duplicates.
184      return construct(uniques, elements);
185    } else {
186      Object[] uniqueElements =
187          shouldTrim(uniques, elements.length) ? Arrays.copyOf(elements, uniques) : elements;
188      return new RegularImmutableSet<E>(uniqueElements, hashCode, table, mask, uniques);
189    }
190  }
191
192  private static boolean shouldTrim(int actualUnique, int expectedUnique) {
193    return actualUnique < (expectedUnique >> 1) + (expectedUnique >> 2);
194  }

196  // We use power-of-2 tables, and this is the highest int that's a power of 2
197  static final int MAX_TABLE_SIZE = Ints.MAX_POWER_OF_TWO;
198
199  // Represents how tightly we can pack things, as a maximum.
200  private static final double DESIRED_LOAD_FACTOR = 0.7;
201
202  // If the set has this many elements, it will "max out" the table size
203  private static final int CUTOFF = (int) (MAX_TABLE_SIZE * DESIRED_LOAD_FACTOR);
204
205  /**
206   * Returns an array size suitable for the backing array of a hash table that uses open addressing
207   * with linear probing in its implementation. The returned size is the smallest power of two that
208   * can hold setSize elements with the desired load factor. Always returns at least setSize + 2.
209   */
210  @VisibleForTesting
211  static int chooseTableSize(int setSize) {
212    setSize = Math.max(setSize, 2);
213    // Correct the size for open addressing to match desired load factor.
214    if (setSize < CUTOFF) {
215      // Round up to the next highest power of 2.
216      int tableSize = Integer.highestOneBit(setSize - 1) << 1;
217      while (tableSize * DESIRED_LOAD_FACTOR < setSize) {
218        tableSize <<= 1;
219      }
220      return tableSize;
221    }
222
223    // The table can't be completely full or we'll get infinite reprobes
224    checkArgument(setSize < MAX_TABLE_SIZE, "collection too large");
225    return MAX_TABLE_SIZE;
226  }

228  /**
229   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
230   * each appears first in the source collection.
231   *
232   * <p><b>Performance note:</b> This method will sometimes recognize that the actual copy operation
233   * is unnecessary; for example, {@code copyOf(copyOf(anArrayList))} will copy the data only once.
234   * This reduces the expense of habitually making defensive copies at API boundaries. However, the
235   * precise conditions for skipping the copy operation are undefined.
236   *
237   * @throws NullPointerException if any of {@code elements} is null
238   * @since 7.0 (source-compatible since 2.0)
239   */
240  public static <E> ImmutableSet<E> copyOf(Collection<? extends E> elements) {
241    /*
242     * TODO(lowasser): consider checking for ImmutableAsList here
243     * TODO(lowasser): consider checking for Multiset here
244     */
245    // Don't refer to ImmutableSortedSet by name so it won't pull in all that code
246    if (elements instanceof ImmutableSet && !(elements instanceof SortedSet)) {
247      @SuppressWarnings("unchecked") // all supported methods are covariant
248      ImmutableSet<E> set = (ImmutableSet<E>) elements;
249      if (!set.isPartialView()) {
250        return set;
251      }
252    }
253    Object[] array = elements.toArray();
254    return construct(array.length, array);
255  }

257  /**
258   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
259   * each appears first in the source iterable. This method iterates over {@code elements} only
260   * once.
261   *
262   * <p><b>Performance note:</b> This method will sometimes recognize that the actual copy operation
263   * is unnecessary; for example, {@code copyOf(copyOf(anArrayList))} should copy the data only
264   * once. This reduces the expense of habitually making defensive copies at API boundaries.
265   * However, the precise conditions for skipping the copy operation are undefined.
266   *
267   * @throws NullPointerException if any of {@code elements} is null
268   */
269  public static <E> ImmutableSet<E> copyOf(Iterable<? extends E> elements) {
270    return (elements instanceof Collection)
271        ? copyOf((Collection<? extends E>) elements)
272        : copyOf(elements.iterator());
273  }

275  /**
276   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
277   * each appears first in the source iterator.
278   *
279   * @throws NullPointerException if any of {@code elements} is null
280   */
281  public static <E> ImmutableSet<E> copyOf(Iterator<? extends E> elements) {
282    // We special-case for 0 or 1 elements, but anything further is madness.
283    if (!elements.hasNext()) {
284      return of();
285    }
286    E first = elements.next();
287    if (!elements.hasNext()) {
288      return of(first);
289    } else {
290      return new ImmutableSet.Builder<E>().add(first).addAll(elements).build();
291    }
292  }

294  /**
295   * Returns an immutable set containing each of {@code elements}, minus duplicates, in the order
296   * each appears first in the source array.
297   *
298   * @throws NullPointerException if any of {@code elements} is null
299   * @since 3.0
300   */
301  public static <E> ImmutableSet<E> copyOf(E[] elements) {
302    switch (elements.length) {
303      case 0:
304        return of();
305      case 1:
306        return of(elements[0]);
307      default:
308        return construct(elements.length, elements.clone());
309    }
310  }

319  @Override
320  public boolean equals(@NullableDecl Object object) {
321    if (object == this) {
322      return true;
323    }
324    if (object instanceof ImmutableSet
325        && isHashCodeFast()
326        && ((ImmutableSet<?>) object).isHashCodeFast()
327        && hashCode() != object.hashCode()) {
328      return false;
329    }
330    return Sets.equalsImpl(this, object);
331  }

355  /*
356   * This class is used to serialize all ImmutableSet instances, except for
357   * ImmutableEnumSet/ImmutableSortedSet, regardless of implementation type. It
358   * captures their "logical contents" and they are reconstructed using public
359   * static factories. This is necessary to ensure that the existence of a

366      this.elements = elements;

370      return copyOf(elements);
371    }

373    private static final long serialVersionUID = 0;
374  }

407  /**
408   * A builder for creating {@code ImmutableSet} instances. Example:
409   *
410   * <pre>{@code
411   * static final ImmutableSet<Color> GOOGLE_COLORS =
412   *     ImmutableSet.<Color>builder()
413   *         .addAll(WEBSAFE_COLORS)
414   *         .add(new Color(0, 191, 255))
415   *         .build();
416   * }</pre>
417   *
418   * <p>Elements appear in the resulting set in the same order they were first added to the builder.
419   *
420   * <p>Building does not change the state of the builder, so it is still possible to add more
421   * elements and to build again.
422   *
423   * @since 2.0
424   */
425  public static class Builder<E> extends ImmutableCollection.ArrayBasedBuilder<E> {
426    @NullableDecl @VisibleForTesting Object[] hashTable;
427    private int hashCode;
428
429    /**
430     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
431     * ImmutableSet#builder}.
432     */
433    public Builder() {
434      super(DEFAULT_INITIAL_CAPACITY);
435    }
436
437    Builder(int capacity) {
438      super(capacity);
439      this.hashTable = new Object[chooseTableSize(capacity)];
440    }
441
442    /**
443     * Adds {@code element} to the {@code ImmutableSet}. If the {@code ImmutableSet} already
444     * contains {@code element}, then {@code add} has no effect (only the previously added element
445     * is retained).
446     *
447     * @param element the element to add
448     * @return this {@code Builder} object
449     * @throws NullPointerException if {@code element} is null
450     */
451    @CanIgnoreReturnValue
452    @Override
453    public Builder<E> add(E element) {
454      checkNotNull(element);
455      if (hashTable != null && chooseTableSize(size) <= hashTable.length) {
456        addDeduping(element);
457        return this;
458      } else {
459        hashTable = null;
460        super.add(element);
461        return this;
462      }
463    }
464
465    /**
466     * Adds each element of {@code elements} to the {@code ImmutableSet}, ignoring duplicate
467     * elements (only the first duplicate element is added).
468     *
469     * @param elements the elements to add
470     * @return this {@code Builder} object
471     * @throws NullPointerException if {@code elements} is null or contains a null element
472     */
473    @Override
474    @CanIgnoreReturnValue
475    public Builder<E> add(E... elements) {
476      if (hashTable != null) {
477        for (E e : elements) {
478          add(e);
479        }
480      } else {
481        super.add(elements);
482      }
483      return this;
484    }
485
486    private void addDeduping(E element) {
487      int mask = hashTable.length - 1;
488      int hash = element.hashCode();
489      for (int i = Hashing.smear(hash); ; i++) {
490        i &= mask;
491        Object previous = hashTable[i];
492        if (previous == null) {
493          hashTable[i] = element;
494          hashCode += hash;
495          super.add(element);
496          return;
497        } else if (previous.equals(element)) {
498          return;
499        }
500      }
501    }
502
503    /**
504     * Adds each element of {@code elements} to the {@code ImmutableSet}, ignoring duplicate
505     * elements (only the first duplicate element is added).
506     *
507     * @param elements the {@code Iterable} to add to the {@code ImmutableSet}
508     * @return this {@code Builder} object
509     * @throws NullPointerException if {@code elements} is null or contains a null element
510     */
511    @CanIgnoreReturnValue
512    @Override
513    public Builder<E> addAll(Iterable<? extends E> elements) {
514      checkNotNull(elements);
515      if (hashTable != null) {
516        for (E e : elements) {
517          add(e);
518        }
519      } else {
520        super.addAll(elements);
521      }
522      return this;
523    }
524
525    /**
526     * Adds each element of {@code elements} to the {@code ImmutableSet}, ignoring duplicate
527     * elements (only the first duplicate element is added).
528     *
529     * @param elements the elements to add to the {@code ImmutableSet}
530     * @return this {@code Builder} object
531     * @throws NullPointerException if {@code elements} is null or contains a null element
532     */
533    @CanIgnoreReturnValue
534    @Override
535    public Builder<E> addAll(Iterator<? extends E> elements) {
536      checkNotNull(elements);
537      while (elements.hasNext()) {
538        add(elements.next());
539      }
540      return this;
541    }
542
543    @CanIgnoreReturnValue
544    @SuppressWarnings("unchecked") // ArrayBasedBuilder stores its elements as Object.
545    Builder<E> combine(Builder<E> other) {
546      if (hashTable != null) {
547        for (int i = 0; i < other.size; ++i) {
548          add((E) other.contents[i]);
549        }
550      } else {
551        addAll(other.contents, other.size);
552      }
553      return this;
554    }
555
556    /**
557     * Returns a newly-created {@code ImmutableSet} based on the contents of the {@code Builder}.
558     */
559    @SuppressWarnings("unchecked")
560    @Override
561    public ImmutableSet<E> build() {
562      switch (size) {
563        case 0:
564          return of();
565        case 1:
566          return (ImmutableSet<E>) of(contents[0]);
567        default:
568          ImmutableSet<E> result;
569          if (hashTable != null && chooseTableSize(size) == hashTable.length) {
570            Object[] uniqueElements =
571                shouldTrim(size, contents.length) ? Arrays.copyOf(contents, size) : contents;
572            result =
573                new RegularImmutableSet<E>(
574                    uniqueElements, hashCode, hashTable, hashTable.length - 1, size);
575          } else {
576            result = construct(size, contents);
577            // construct has the side effect of deduping contents, so we update size
578            // accordingly.
579            size = result.size();
580          }
581          forceCopy = true;
582          hashTable = null;
583          return result;
584      }
585    }
586  }
