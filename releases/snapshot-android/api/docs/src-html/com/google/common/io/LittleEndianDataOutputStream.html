015package com.google.common.io;
016
017import com.google.common.annotations.Beta;
018import com.google.common.annotations.GwtIncompatible;
019import com.google.common.annotations.J2ktIncompatible;
020import com.google.common.base.Preconditions;
021import com.google.common.primitives.Longs;
022import java.io.DataOutput;
023import java.io.DataOutputStream;
024import java.io.FilterOutputStream;
025import java.io.IOException;
026import java.io.OutputStream;
027
028/**
029 * An implementation of {@link DataOutput} that uses little-endian byte ordering for writing {@code
030 * char}, {@code short}, {@code int}, {@code float}, {@code double}, and {@code long} values.
031 *
032 * <p><b>Note:</b> This class intentionally violates the specification of its supertype {@code
033 * DataOutput}, which explicitly requires big-endian byte order.
034 *
035 * @author Chris Nokleberg
036 * @author Keith Bottner
037 * @since 8.0
038 */
039@Beta
040@J2ktIncompatible
041@GwtIncompatible
042@ElementTypesAreNonnullByDefault
043public final class LittleEndianDataOutputStream extends FilterOutputStream implements DataOutput {
044
045  /**
046   * Creates a {@code LittleEndianDataOutputStream} that wraps the given stream.
047   *
048   * @param out the stream to delegate to
049   */
050  public LittleEndianDataOutputStream(OutputStream out) {
051    super(new DataOutputStream(Preconditions.checkNotNull(out)));
052  }
053
054  @Override
055  public void write(byte[] b, int off, int len) throws IOException {
056    // Override slow FilterOutputStream impl
057    out.write(b, off, len);
058  }
059
060  @Override
061  public void writeBoolean(boolean v) throws IOException {
062    ((DataOutputStream) out).writeBoolean(v);
063  }
064
065  @Override
066  public void writeByte(int v) throws IOException {
067    ((DataOutputStream) out).writeByte(v);
068  }
069
070  /**
071   * @deprecated The semantics of {@code writeBytes(String s)} are considered dangerous. Please use
072   *     {@link #writeUTF(String s)}, {@link #writeChars(String s)} or another write method instead.
073   */
074  @Deprecated
075  @Override
076  public void writeBytes(String s) throws IOException {
077    ((DataOutputStream) out).writeBytes(s);
078  }
079
080  /**
081   * Writes a char as specified by {@link DataOutputStream#writeChar(int)}, except using
082   * little-endian byte order.
083   *
084   * @throws IOException if an I/O error occurs
085   */
086  @Override
087  public void writeChar(int v) throws IOException {
088    writeShort(v);
089  }
090
091  /**
092   * Writes a {@code String} as specified by {@link DataOutputStream#writeChars(String)}, except
093   * each character is written using little-endian byte order.
094   *
095   * @throws IOException if an I/O error occurs
096   */
097  @Override
098  public void writeChars(String s) throws IOException {
099    for (int i = 0; i < s.length(); i++) {
100      writeChar(s.charAt(i));
101    }
102  }
103
104  /**
105   * Writes a {@code double} as specified by {@link DataOutputStream#writeDouble(double)}, except
106   * using little-endian byte order.
107   *
108   * @throws IOException if an I/O error occurs
109   */
110  @Override
111  public void writeDouble(double v) throws IOException {
112    writeLong(Double.doubleToLongBits(v));
113  }
114
115  /**
116   * Writes a {@code float} as specified by {@link DataOutputStream#writeFloat(float)}, except using
117   * little-endian byte order.
118   *
119   * @throws IOException if an I/O error occurs
120   */
121  @Override
122  public void writeFloat(float v) throws IOException {
123    writeInt(Float.floatToIntBits(v));
124  }
125
126  /**
127   * Writes an {@code int} as specified by {@link DataOutputStream#writeInt(int)}, except using
128   * little-endian byte order.
129   *
130   * @throws IOException if an I/O error occurs
131   */
132  @Override
133  public void writeInt(int v) throws IOException {
134    out.write(0xFF & v);
135    out.write(0xFF & (v >> 8));
136    out.write(0xFF & (v >> 16));
137    out.write(0xFF & (v >> 24));
138  }
139
140  /**
141   * Writes a {@code long} as specified by {@link DataOutputStream#writeLong(long)}, except using
142   * little-endian byte order.
143   *
144   * @throws IOException if an I/O error occurs
145   */
146  @Override
147  public void writeLong(long v) throws IOException {
148    byte[] bytes = Longs.toByteArray(Long.reverseBytes(v));
149    write(bytes, 0, bytes.length);
150  }
151
152  /**
153   * Writes a {@code short} as specified by {@link DataOutputStream#writeShort(int)}, except using
154   * little-endian byte order.
155   *
156   * @throws IOException if an I/O error occurs
157   */
158  @Override
159  public void writeShort(int v) throws IOException {
160    out.write(0xFF & v);
161    out.write(0xFF & (v >> 8));
162  }
163
164  @Override
165  public void writeUTF(String str) throws IOException {
166    ((DataOutputStream) out).writeUTF(str);
167  }
168
169  // Overriding close() because FilterOutputStream's close() method pre-JDK8 has bad behavior:
170  // it silently ignores any exception thrown by flush(). Instead, just close the delegate stream.
171  // It should flush itself if necessary.
172  @Override
173  public void close() throws IOException {
174    out.close();
175  }
176}
