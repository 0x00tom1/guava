259  /**
260   * Drains the queue as {@link BlockingQueue#drainTo(Collection, int)}, but if the requested {@code
261   * numElements} elements are not available, it will wait for them up to the specified timeout.
262   *
263   * @param q the blocking queue to be drained
264   * @param buffer where to add the transferred elements
265   * @param numElements the number of elements to be waited for
266   * @param timeout how long to wait before giving up, in units of {@code unit}
267   * @param unit a {@code TimeUnit} determining how to interpret the timeout parameter
268   * @return the number of elements transferred
269   * @throws InterruptedException if interrupted while waiting
270   */
271  @Beta
272  @CanIgnoreReturnValue
273  @GwtIncompatible // BlockingQueue
274  @SuppressWarnings("GoodTime") // should accept a java.time.Duration
275  public static <E> int drain(
276      BlockingQueue<E> q,
277      Collection<? super E> buffer,
278      int numElements,
279      long timeout,
280      TimeUnit unit)
281      throws InterruptedException {
282    Preconditions.checkNotNull(buffer);
283    /*
284     * This code performs one System.nanoTime() more than necessary, and in return, the time to
285     * execute Queue#drainTo is not added *on top* of waiting for the timeout (which could make
286     * the timeout arbitrarily inaccurate, given a queue that is slow to drain).
287     */
288    long deadline = System.nanoTime() + unit.toNanos(timeout);
289    int added = 0;
290    while (added < numElements) {
291      // we could rely solely on #poll, but #drainTo might be more efficient when there are multiple
292      // elements already available (e.g. LinkedBlockingQueue#drainTo locks only once)
293      added += q.drainTo(buffer, numElements - added);
294      if (added < numElements) { // not enough elements immediately available; will have to poll
295        E e = q.poll(deadline - System.nanoTime(), TimeUnit.NANOSECONDS);
296        if (e == null) {
297          break; // we already waited enough, and there are no more elements in sight
298        }
299        buffer.add(e);
300        added++;
301      }
302    }
303    return added;
304  }

306  /**
307   * Drains the queue as {@linkplain #drain(BlockingQueue, Collection, int, long, TimeUnit)}, but
308   * with a different behavior in case it is interrupted while waiting. In that case, the operation
309   * will continue as usual, and in the end the thread's interruption status will be set (no {@code
310   * InterruptedException} is thrown).
311   *
312   * @param q the blocking queue to be drained
313   * @param buffer where to add the transferred elements
314   * @param numElements the number of elements to be waited for
315   * @param timeout how long to wait before giving up, in units of {@code unit}
316   * @param unit a {@code TimeUnit} determining how to interpret the timeout parameter
317   * @return the number of elements transferred
318   */
319  @Beta
320  @CanIgnoreReturnValue
321  @GwtIncompatible // BlockingQueue
322  @SuppressWarnings("GoodTime") // should accept a java.time.Duration
323  public static <E> int drainUninterruptibly(
324      BlockingQueue<E> q,
325      Collection<? super E> buffer,
326      int numElements,
327      long timeout,
328      TimeUnit unit) {
329    Preconditions.checkNotNull(buffer);
330    long deadline = System.nanoTime() + unit.toNanos(timeout);
331    int added = 0;
332    boolean interrupted = false;
333    try {
334      while (added < numElements) {
335        // we could rely solely on #poll, but #drainTo might be more efficient when there are
336        // multiple elements already available (e.g. LinkedBlockingQueue#drainTo locks only once)
337        added += q.drainTo(buffer, numElements - added);
338        if (added < numElements) { // not enough elements immediately available; will have to poll
339          E e; // written exactly once, by a successful (uninterrupted) invocation of #poll
340          while (true) {
341            try {
342              e = q.poll(deadline - System.nanoTime(), TimeUnit.NANOSECONDS);
343              break;
344            } catch (InterruptedException ex) {
345              interrupted = true; // note interruption and retry
346            }
347          }
348          if (e == null) {
349            break; // we already waited enough, and there are no more elements in sight
350          }
351          buffer.add(e);
352          added++;
353        }
354      }
355    } finally {
356      if (interrupted) {
357        Thread.currentThread().interrupt();
358      }
359    }
360    return added;
361  }

363  /**
364   * Returns a synchronized (thread-safe) queue backed by the specified queue. In order to guarantee
365   * serial access, it is critical that <b>all</b> access to the backing queue is accomplished
366   * through the returned queue.
367   *
368   * <p>It is imperative that the user manually synchronize on the returned queue when accessing the
369   * queue's iterator:
370   *
371   * <pre>{@code
372   * Queue<E> queue = Queues.synchronizedQueue(MinMaxPriorityQueue.<E>create());
373   * ...
374   * queue.add(element);  // Needn't be in synchronized block
375   * ...
376   * synchronized (queue) {  // Must synchronize on queue!
377   *   Iterator<E> i = queue.iterator(); // Must be in synchronized block
378   *   while (i.hasNext()) {
379   *     foo(i.next());
380   *   }
381   * }
382   * }</pre>
383   *
384   * <p>Failure to follow this advice may result in non-deterministic behavior.
385   *
386   * <p>The returned queue will be serializable if the specified queue is serializable.
387   *
388   * @param queue the queue to be wrapped in a synchronized view
389   * @return a synchronized view of the specified queue
390   * @since 14.0
391   */
392  public static <E> Queue<E> synchronizedQueue(Queue<E> queue) {
393    return Synchronized.queue(queue, null);
394  }
