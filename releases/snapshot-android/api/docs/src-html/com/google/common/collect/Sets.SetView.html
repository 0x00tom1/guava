108  /**
109   * Returns an immutable set instance containing the given enum elements. Internally, the returned
110   * set will be backed by an {@link EnumSet}.
111   *
112   * <p>The iteration order of the returned set follows the enum's iteration order, not the order in
113   * which the elements appear in the given collection.
114   *
115   * @param elements the elements, all of the same {@code enum} type, that the set should contain
116   * @return an immutable set containing those elements, minus duplicates
117   */
118  // http://code.google.com/p/google-web-toolkit/issues/detail?id=3028
119  @GwtCompatible(serializable = true)
120  public static <E extends Enum<E>> ImmutableSet<E> immutableEnumSet(Iterable<E> elements) {
121    if (elements instanceof ImmutableEnumSet) {
122      return (ImmutableEnumSet<E>) elements;
123    } else if (elements instanceof Collection) {
124      Collection<E> collection = (Collection<E>) elements;
125      if (collection.isEmpty()) {
126        return ImmutableSet.of();
127      } else {
128        return ImmutableEnumSet.asImmutable(EnumSet.copyOf(collection));
129      }
130    } else {
131      Iterator<E> itr = elements.iterator();
132      if (itr.hasNext()) {
133        EnumSet<E> enumSet = EnumSet.of(itr.next());
134        Iterators.addAll(enumSet, itr);
135        return ImmutableEnumSet.asImmutable(enumSet);
136      } else {
137        return ImmutableSet.of();
138      }
139    }
140  }

541   * An unmodifiable view of a set which may be backed by other sets; this view will change as the
542   * backing sets do. Contains methods to copy the data into a new set which will then remain
543   * stable. There is usually no reason to retain a reference of type {@code SetView}; typically,
544   * you either use it as a plain {@link Set}, or immediately invoke {@link #immutableCopy} or
545   * {@link #copyInto} and forget the {@code SetView} itself.
546   *
547   * @since 2.0
548   */
549  public abstract static class SetView<E extends @Nullable Object> extends AbstractSet<E> {
550    private SetView() {} // no subclasses but our own
551
552    /**
553     * Returns an immutable copy of the current contents of this set view. Does not support null
554     * elements.
555     *
556     * <p><b>Warning:</b> this may have unexpected results if a backing set of this view uses a
557     * nonstandard notion of equivalence, for example if it is a {@link TreeSet} using a comparator
558     * that is inconsistent with {@link Object#equals(Object)}.
559     */
560    @SuppressWarnings("nullness") // Unsafe, but we can't fix it now.
561    public ImmutableSet<@NonNull E> immutableCopy() {
562      return ImmutableSet.copyOf((SetView<@NonNull E>) this);
563    }
564
565    /**
566     * Copies the current contents of this set view into an existing set. This method has equivalent
567     * behavior to {@code set.addAll(this)}, assuming that all the sets involved are based on the
568     * same notion of equivalence.
569     *
570     * @return a reference to {@code set}, for convenience
571     */
572    // Note: S should logically extend Set<? super E> but can't due to either

1578   * Returns the set of all subsets of {@code set} of size {@code size}. For example, {@code
1579   * combinations(ImmutableSet.of(1, 2, 3), 2)} returns the set {@code {{1, 2}, {1, 3}, {2, 3}}}.
1580   *
1581   * <p>Elements appear in these subsets in the same iteration order as they appeared in the input
1582   * set. The order in which these subsets appear in the outer set is undefined.
1583   *
1584   * <p>The returned set and its constituent sets use {@code equals} to decide whether two elements
1585   * are identical, even if the input set uses a different concept of equivalence.
1586   *
1587   * <p><i>Performance notes:</i> the memory usage of the returned set is only {@code O(n)}. When
1588   * the result set is constructed, the input set is merely copied. Only as the result set is
1589   * iterated are the individual subsets created. Each of these subsets occupies an additional O(n)
1590   * memory but only for as long as the user retains a reference to it. That is, the set returned by
1591   * {@code combinations} does not retain the individual subsets.
1592   *
1593   * @param set the set of elements to take combinations of
1594   * @param size the number of elements per combination
1595   * @return the set of all combinations of {@code size} elements from {@code set}
1596   * @throws IllegalArgumentException if {@code size} is not between 0 and {@code set.size()}
1597   *     inclusive
1598   * @throws NullPointerException if {@code set} is or contains {@code null}
1599   * @since 23.0
1600   */
1601  @Beta
1602  public static <E> Set<Set<E>> combinations(Set<E> set, final int size) {
1603    final ImmutableMap<E, Integer> index = Maps.indexMap(set);
1604    checkNonnegative(size, "size");
1605    checkArgument(size <= index.size(), "size (%s) must be <= set.size() (%s)", size, index.size());
1606    if (size == 0) {
1607      return ImmutableSet.<Set<E>>of(ImmutableSet.<E>of());
1608    } else if (size == index.size()) {
1609      return ImmutableSet.<Set<E>>of(index.keySet());
1610    }
1611    return new AbstractSet<Set<E>>() {
1612      @Override
1613      public boolean contains(@CheckForNull Object o) {
1614        if (o instanceof Set) {
1615          Set<?> s = (Set<?>) o;
1616          return s.size() == size && index.keySet().containsAll(s);
1617        }
1618        return false;
1619      }
1620
1621      @Override
1622      public Iterator<Set<E>> iterator() {
1623        return new AbstractIterator<Set<E>>() {
1624          final BitSet bits = new BitSet(index.size());
1625
1626          @Override
1627          @CheckForNull
1628          protected Set<E> computeNext() {
1629            if (bits.isEmpty()) {
1630              bits.set(0, size);
1631            } else {
1632              int firstSetBit = bits.nextSetBit(0);
1633              int bitToFlip = bits.nextClearBit(firstSetBit);
1634
1635              if (bitToFlip == index.size()) {
1636                return endOfData();
1637              }
1638              /*
1639               * The current set in sorted order looks like
1640               * {firstSetBit, firstSetBit + 1, ..., bitToFlip - 1, ...}
1641               * where it does *not* contain bitToFlip.
1642               *
1643               * The next combination is
1644               *
1645               * {0, 1, ..., bitToFlip - firstSetBit - 2, bitToFlip, ...}
1646               *
1647               * This is lexicographically next if you look at the combinations in descending order
1648               * e.g. {2, 1, 0}, {3, 1, 0}, {3, 2, 0}, {3, 2, 1}, {4, 1, 0}...
1649               */
1650
1651              bits.set(0, bitToFlip - firstSetBit - 1);
1652              bits.clear(bitToFlip - firstSetBit - 1, bitToFlip);
1653              bits.set(bitToFlip);
1654            }
1655            final BitSet copy = (BitSet) bits.clone();
1656            return new AbstractSet<E>() {
1657              @Override
1658              public boolean contains(@CheckForNull Object o) {
1659                Integer i = index.get(o);
1660                return i != null && copy.get(i);
1661              }
1662
1663              @Override
1664              public Iterator<E> iterator() {
1665                return new AbstractIterator<E>() {
1666                  int i = -1;
1667
1668                  @Override
1669                  @CheckForNull
1670                  protected E computeNext() {
1671                    i = copy.nextSetBit(i + 1);
1672                    if (i == -1) {
1673                      return endOfData();
1674                    }
1675                    return index.keySet().asList().get(i);
1676                  }
1677                };
1678              }
1679
1680              @Override
1681              public int size() {
1682                return size;
1683              }
1684            };
1685          }
1686        };
1687      }
1688
1689      @Override
1690      public int size() {
1691        return IntMath.binomial(index.size(), size);
1692      }
1693
1694      @Override
1695      public String toString() {
1696        return "Sets.combinations(" + index.keySet() + ", " + size + ")";
1697      }
1698    };
1699  }

1902  static boolean removeAllImpl(Set<?> set, Collection<?> collection) {
1903    checkNotNull(collection); // for GWT
1904    if (collection instanceof Multiset) {
1905      collection = ((Multiset<?>) collection).elementSet();
1906    }
1907    /*
1908     * AbstractSet.removeAll(List) has quadratic behavior if the list size
1909     * is just more than the set's size.  We augment the test by
1910     * assuming that sets have fast contains() performance, and other
1911     * collections don't.  See
1912     * http://code.google.com/p/guava-libraries/issues/detail?id=1013
1913     */
1914    if (collection instanceof Set && collection.size() > set.size()) {
1915      return Iterators.removeAll(set.iterator(), collection);
1916    } else {
1917      return removeAllImpl(set, collection.iterator());
1918    }

2064  /**
2065   * Returns a view of the portion of {@code set} whose elements are contained by {@code range}.
2066   *
2067   * <p>This method delegates to the appropriate methods of {@link NavigableSet} (namely {@link
2068   * NavigableSet#subSet(Object, boolean, Object, boolean) subSet()}, {@link
2069   * NavigableSet#tailSet(Object, boolean) tailSet()}, and {@link NavigableSet#headSet(Object,
2070   * boolean) headSet()}) to actually construct the view. Consult these methods for a full
2071   * description of the returned view's behavior.
2072   *
2073   * <p><b>Warning:</b> {@code Range}s always represent a range of values using the values' natural
2074   * ordering. {@code NavigableSet} on the other hand can specify a custom ordering via a {@link
2075   * Comparator}, which can violate the natural ordering. Using this method (or in general using
2076   * {@code Range}) with unnaturally-ordered sets can lead to unexpected and undefined behavior.
2077   *
2078   * @since 20.0
2079   */
2080  @Beta
2081  @GwtIncompatible // NavigableSet
2082  public static <K extends Comparable<? super K>> NavigableSet<K> subSet(
2083      NavigableSet<K> set, Range<K> range) {
2084    if (set.comparator() != null
2085        && set.comparator() != Ordering.natural()
2086        && range.hasLowerBound()
2087        && range.hasUpperBound()) {
2088      checkArgument(
2089          set.comparator().compare(range.lowerEndpoint(), range.upperEndpoint()) <= 0,
2090          "set is using a custom comparator which is inconsistent with the natural ordering.");
2091    }
2092    if (range.hasLowerBound() && range.hasUpperBound()) {
2093      return set.subSet(
2094          range.lowerEndpoint(),
2095          range.lowerBoundType() == BoundType.CLOSED,
2096          range.upperEndpoint(),
2097          range.upperBoundType() == BoundType.CLOSED);
2098    } else if (range.hasLowerBound()) {
2099      return set.tailSet(range.lowerEndpoint(), range.lowerBoundType() == BoundType.CLOSED);
2100    } else if (range.hasUpperBound()) {
2101      return set.headSet(range.upperEndpoint(), range.upperBoundType() == BoundType.CLOSED);
2102    }
2103    return checkNotNull(set);
2104  }
