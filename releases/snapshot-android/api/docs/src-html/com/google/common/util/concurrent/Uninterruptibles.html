034/**
035 * Utilities for treating interruptible operations as uninterruptible. In all cases, if a thread is
036 * interrupted during such a call, the call continues to block until the result is available or the
037 * timeout elapses, and only then re-interrupts the thread.
038 *
039 * @author Anthony Zana
040 * @since 10.0
041 */
042@GwtCompatible(emulated = true)
043public final class Uninterruptibles {
044

046  // methods is identical, save for method being invoked.

048  /** Invokes {@code latch.}{@link CountDownLatch#await() await()} uninterruptibly. */
049  @GwtIncompatible // concurrency
050  public static void awaitUninterruptibly(CountDownLatch latch) {
051    boolean interrupted = false;
052    try {
053      while (true) {
054        try {
055          latch.await();
056          return;
057        } catch (InterruptedException e) {
058          interrupted = true;
059        }
060      }
061    } finally {
062      if (interrupted) {
063        Thread.currentThread().interrupt();
064      }
065    }
066  }
067
068  /**
069   * Invokes {@code latch.}{@link CountDownLatch#await(long, TimeUnit) await(timeout, unit)}
070   * uninterruptibly.
071   */
072  @CanIgnoreReturnValue // TODO(cpovirk): Consider being more strict.
073  @GwtIncompatible // concurrency
074  @SuppressWarnings("GoodTime") // should accept a java.time.Duration
075  public static boolean awaitUninterruptibly(CountDownLatch latch, long timeout, TimeUnit unit) {
076    boolean interrupted = false;
077    try {
078      long remainingNanos = unit.toNanos(timeout);
079      long end = System.nanoTime() + remainingNanos;
080
081      while (true) {
082        try {
083          // CountDownLatch treats negative timeouts just like zero.
084          return latch.await(remainingNanos, NANOSECONDS);
085        } catch (InterruptedException e) {
086          interrupted = true;
087          remainingNanos = end - System.nanoTime();
088        }
089      }
090    } finally {
091      if (interrupted) {
092        Thread.currentThread().interrupt();
093      }
094    }
095  }

175  /**
176   * Invokes {@code future.}{@link Future#get() get()} uninterruptibly.
177   *
178   * <p>Similar methods:
179   *
180   * <ul>
181   *   <li>To retrieve a result from a {@code Future} that is already done, use {@link
182   *       Futures#getDone Futures.getDone}.
183   *   <li>To treat {@link InterruptedException} uniformly with other exceptions, use {@link
184   *       Futures#getChecked(Future, Class) Futures.getChecked}.
185   *   <li>To get uninterruptibility and remove checked exceptions, use {@link
186   *       Futures#getUnchecked}.
187   * </ul>
188   *
189   * @throws ExecutionException if the computation threw an exception
190   * @throws CancellationException if the computation was cancelled
191   */
192  @CanIgnoreReturnValue
193  public static <V> V getUninterruptibly(Future<V> future) throws ExecutionException {
194    boolean interrupted = false;
195    try {
196      while (true) {
197        try {
198          return future.get();
199        } catch (InterruptedException e) {
200          interrupted = true;
201        }
202      }
203    } finally {
204      if (interrupted) {
205        Thread.currentThread().interrupt();
206      }
207    }
208  }
