189  /**
190   * A builder for creating immutable map instances, especially {@code public static final} maps
191   * ("constant maps"). Example:
192   *
193   * <pre>{@code
194   * static final ImmutableMap<String, Integer> WORD_TO_INT =
195   *     new ImmutableMap.Builder<String, Integer>()
196   *         .put("one", 1)
197   *         .put("two", 2)
198   *         .put("three", 3)
199   *         .build();
200   * }</pre>
201   *
202   * <p>For <i>small</i> immutable maps, the {@code ImmutableMap.of()} methods are even more
203   * convenient.
204   *
205   * <p>By default, a {@code Builder} will generate maps that iterate over entries in the order they
206   * were inserted into the builder, equivalently to {@code LinkedHashMap}. For example, in the
207   * above example, {@code WORD_TO_INT.entrySet()} is guaranteed to iterate over the entries in the
208   * order {@code "one"=1, "two"=2, "three"=3}, and {@code keySet()} and {@code values()} respect
209   * the same order. If you want a different order, consider using {@link ImmutableSortedMap} to
210   * sort by keys, or call {@link #orderEntriesByValue(Comparator)}, which changes this builder to
211   * sort entries by value.
212   *
213   * <p>Builder instances can be reused - it is safe to call {@link #build} multiple times to build
214   * multiple maps in series. Each map is a superset of the maps created before it.
215   *
216   * @since 2.0
217   */
218  @DoNotMock
219  public static class Builder<K, V> {
220    @CheckForNull Comparator<? super V> valueComparator;
221    @Nullable Object[] alternatingKeysAndValues;
222    int size;
223    boolean entriesUsed;
224
225    /**
226     * Creates a new builder. The returned builder is equivalent to the builder generated by {@link
227     * ImmutableMap#builder}.
228     */
229    public Builder() {
230      this(ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY);
231    }
232
233    @SuppressWarnings({"unchecked", "rawtypes"})
234    Builder(int initialCapacity) {
235      this.alternatingKeysAndValues = new @Nullable Object[2 * initialCapacity];
236      this.size = 0;
237      this.entriesUsed = false;
238    }
239
240    private void ensureCapacity(int minCapacity) {
241      if (minCapacity * 2 > alternatingKeysAndValues.length) {
242        alternatingKeysAndValues =
243            Arrays.copyOf(
244                alternatingKeysAndValues,
245                ImmutableCollection.Builder.expandedCapacity(
246                    alternatingKeysAndValues.length, minCapacity * 2));
247        entriesUsed = false;
248      }
249    }
250
251    /**
252     * Associates {@code key} with {@code value} in the built map. Duplicate keys are not allowed,
253     * and will cause {@link #build} to fail.
254     */
255    @CanIgnoreReturnValue
256    public Builder<K, V> put(K key, V value) {
257      ensureCapacity(size + 1);
258      checkEntryNotNull(key, value);
259      alternatingKeysAndValues[2 * size] = key;
260      alternatingKeysAndValues[2 * size + 1] = value;
261      size++;
262      return this;
263    }
264
265    /**
266     * Adds the given {@code entry} to the map, making it immutable if necessary. Duplicate keys are
267     * not allowed, and will cause {@link #build} to fail.
268     *
269     * @since 11.0
270     */
271    @CanIgnoreReturnValue
272    public Builder<K, V> put(Entry<? extends K, ? extends V> entry) {
273      return put(entry.getKey(), entry.getValue());
274    }
275
276    /**
277     * Associates all of the given map's keys and values in the built map. Duplicate keys are not
278     * allowed, and will cause {@link #build} to fail.
279     *
280     * @throws NullPointerException if any key or value in {@code map} is null
281     */
282    @CanIgnoreReturnValue
283    public Builder<K, V> putAll(Map<? extends K, ? extends V> map) {
284      return putAll(map.entrySet());
285    }
286
287    /**
288     * Adds all of the given entries to the built map. Duplicate keys are not allowed, and will
289     * cause {@link #build} to fail.
290     *
291     * @throws NullPointerException if any key, value, or entry is null
292     * @since 19.0
293     */
294    @CanIgnoreReturnValue
295    @Beta
296    public Builder<K, V> putAll(Iterable<? extends Entry<? extends K, ? extends V>> entries) {
297      if (entries instanceof Collection) {
298        ensureCapacity(size + ((Collection<?>) entries).size());
299      }
300      for (Entry<? extends K, ? extends V> entry : entries) {
301        put(entry);
302      }
303      return this;
304    }
305
306    /**
307     * Configures this {@code Builder} to order entries by value according to the specified
308     * comparator.
309     *
310     * <p>The sort order is stable, that is, if two entries have values that compare as equivalent,
311     * the entry that was inserted first will be first in the built map's iteration order.
312     *
313     * @throws IllegalStateException if this method was already called
314     * @since 19.0
315     */
316    @CanIgnoreReturnValue
317    @Beta
318    public Builder<K, V> orderEntriesByValue(Comparator<? super V> valueComparator) {
319      checkState(this.valueComparator == null, "valueComparator was already set");
320      this.valueComparator = checkNotNull(valueComparator, "valueComparator");
321      return this;
322    }
323
324    @CanIgnoreReturnValue
325    Builder<K, V> combine(Builder<K, V> other) {
326      checkNotNull(other);
327      ensureCapacity(this.size + other.size);
328      System.arraycopy(
329          other.alternatingKeysAndValues,
330          0,
331          this.alternatingKeysAndValues,
332          this.size * 2,
333          other.size * 2);
334      this.size += other.size;
335      return this;
336    }
337
338    /*
339     * TODO(kevinb): Should build() and the ImmutableBiMap & ImmutableSortedMap
340     * versions throw an IllegalStateException instead?
341     */
342
343    /**
344     * Returns a newly-created immutable map. The iteration order of the returned map is the order
345     * in which entries were inserted into the builder, unless {@link #orderEntriesByValue} was
346     * called, in which case entries are sorted by value.
347     *
348     * @throws IllegalArgumentException if duplicate keys were added
349     */
350    @SuppressWarnings("unchecked")
351    public ImmutableMap<K, V> build() {
352      /*
353       * If entries is full, then this implementation may end up using the entries array
354       * directly and writing over the entry objects with non-terminal entries, but this is
355       * safe; if this Builder is used further, it will grow the entries array (so it can't
356       * affect the original array), and future build() calls will always copy any entry
357       * objects that cannot be safely reused.
358       */
359      sortEntries();
360      entriesUsed = true;
361      return RegularImmutableMap.create(size, alternatingKeysAndValues);
362    }
363
364    void sortEntries() {
365      if (valueComparator != null) {
366        if (entriesUsed) {
367          alternatingKeysAndValues = Arrays.copyOf(alternatingKeysAndValues, 2 * size);
368        }
369        Entry<K, V>[] entries = new Entry[size];
370        for (int i = 0; i < size; i++) {
371          // requireNonNull is safe because the first `2*size` elements have been filled in.
372          entries[i] =
373              new AbstractMap.SimpleImmutableEntry<K, V>(
374                  (K) requireNonNull(alternatingKeysAndValues[2 * i]),
375                  (V) requireNonNull(alternatingKeysAndValues[2 * i + 1]));
376        }
377        Arrays.sort(
378            entries, 0, size, Ordering.from(valueComparator).onResultOf(Maps.<V>valueFunction()));
379        for (int i = 0; i < size; i++) {
380          alternatingKeysAndValues[2 * i] = entries[i].getKey();
381          alternatingKeysAndValues[2 * i + 1] = entries[i].getValue();
382        }
383      }
384    }
385  }

387  /**
388   * Returns an immutable map containing the same entries as {@code map}. The returned map iterates
389   * over entries in the same order as the {@code entrySet} of the original map. If {@code map}
390   * somehow contains entries with duplicate keys (for example, if it is a {@code SortedMap} whose
391   * comparator is not <i>consistent with equals</i>), the results of this method are undefined.
392   *
393   * <p>Despite the method name, this method attempts to avoid actually copying the data when it is
394   * safe to do so. The exact circumstances under which a copy will or will not be performed are
395   * undocumented and subject to change.
396   *
397   * @throws NullPointerException if any key or value in {@code map} is null
398   */
399  public static <K, V> ImmutableMap<K, V> copyOf(Map<? extends K, ? extends V> map) {
400    if ((map instanceof ImmutableMap) && !(map instanceof SortedMap)) {
401      @SuppressWarnings("unchecked") // safe since map is not writable
402      ImmutableMap<K, V> kvMap = (ImmutableMap<K, V>) map;
403      if (!kvMap.isPartialView()) {
404        return kvMap;
405      }
406    }
407    return copyOf(map.entrySet());
408  }

410  /**
411   * Returns an immutable map containing the specified entries. The returned map iterates over
412   * entries in the same order as the original iterable.
413   *
414   * @throws NullPointerException if any key, value, or entry is null
415   * @throws IllegalArgumentException if two entries have the same key
416   * @since 19.0
417   */
418  @Beta
419  public static <K, V> ImmutableMap<K, V> copyOf(
420      Iterable<? extends Entry<? extends K, ? extends V>> entries) {
421    int initialCapacity =
422        (entries instanceof Collection)
423            ? ((Collection<?>) entries).size()
424            : ImmutableCollection.Builder.DEFAULT_INITIAL_CAPACITY;
425    ImmutableMap.Builder<K, V> builder = new ImmutableMap.Builder<K, V>(initialCapacity);
426    builder.putAll(entries);
427    return builder.build();
428  }

739  /**
740   * Serialized type for all ImmutableMap instances. It captures the logical contents and they are
741   * reconstructed using public factory methods. This ensures that the implementation types remain
742   * as implementation details.
743   */
744  static class SerializedForm<K, V> implements Serializable {
745    // This object retains references to collections returned by keySet() and value(). This saves
746    // bytes when the both the map and its keySet or value collection are written to the same
747    // instance of ObjectOutputStream.
748
749    // TODO(b/160980469): remove support for the old serialization format after some time
750    private static final boolean USE_LEGACY_SERIALIZATION = true;
751
752    private final Object keys;
753    private final Object values;
754
755    SerializedForm(ImmutableMap<K, V> map) {
756      if (USE_LEGACY_SERIALIZATION) {
757        Object[] keys = new Object[map.size()];
758        Object[] values = new Object[map.size()];
759        int i = 0;
760        // "extends Object" works around https://github.com/typetools/checker-framework/issues/3013
761        for (Entry<? extends Object, ? extends Object> entry : map.entrySet()) {
762          keys[i] = entry.getKey();
763          values[i] = entry.getValue();
764          i++;
765        }
766        this.keys = keys;
767        this.values = values;
768        return;
769      }
770      this.keys = map.keySet();
771      this.values = map.values();
772    }
773
774    @SuppressWarnings("unchecked")
775    final Object readResolve() {
776      if (!(this.keys instanceof ImmutableSet)) {
777        return legacyReadResolve();
778      }
779
780      ImmutableSet<K> keySet = (ImmutableSet<K>) this.keys;
781      ImmutableCollection<V> values = (ImmutableCollection<V>) this.values;
782
783      Builder<K, V> builder = makeBuilder(keySet.size());
784
785      UnmodifiableIterator<K> keyIter = keySet.iterator();
786      UnmodifiableIterator<V> valueIter = values.iterator();
787
788      while (keyIter.hasNext()) {
789        builder.put(keyIter.next(), valueIter.next());
790      }
791
792      return builder.build();
793    }
794
795    @SuppressWarnings("unchecked")
796    final Object legacyReadResolve() {
797      K[] keys = (K[]) this.keys;
798      V[] values = (V[]) this.values;
799
800      Builder<K, V> builder = makeBuilder(keys.length);
801
802      for (int i = 0; i < keys.length; i++) {
803        builder.put(keys[i], values[i]);
804      }
805      return builder.build();
806    }
807
808    /**
809     * Returns a builder that builds the unserialized type. Subclasses should override this method.
810     */
811    Builder<K, V> makeBuilder(int size) {
812      return new Builder<>(size);
813    }
814
815    private static final long serialVersionUID = 0;
816  }
817
818  /**
819   * Returns a serializable form of this object. Non-public subclasses should not override this
820   * method. Publicly-accessible subclasses must override this method and should return a subclass
821   * of SerializedForm whose readResolve() method returns objects of the subclass type.
822   */
823  Object writeReplace() {
824    return new SerializedForm<>(this);
825  }
