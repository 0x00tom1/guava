030package com.azul.tooling.in;
031
032import java.net.URL;
033import java.util.jar.JarFile;
034
035import static java.lang.System.currentTimeMillis;
036
037@Model("com.azul.tooling.JarLoadEventModel")
038public final class JarLoadEvent implements Tooling.ToolingEvent {
039    private static final boolean isEnabled;
040
041    static {
042        isEnabled = Tooling.isEventTypeEnabled(JarLoadEvent.class);
043    }
044
045    private JarLoadEvent(URL url, JarFile jar) {
046        this.url = url;
047        this.jar = jar;
048    }
049
050    public static JarLoadEvent jarLoadEvent(URL url, JarFile jar) {
051        return new JarLoadEvent(url, jar);
052    }
053
054    public static boolean isEnabled() {
055        return isEnabled;
056    }
057
058    public boolean isEventEnabled() {
059        return isEnabled();
060    }
061
062    public final URL url;
063    public final JarFile jar;
064
065    public URL getURL() {
066      return url;
067    }
068
069    public JarFile getJarFile() {
070      return jar;
071    }
072}
