030package com.azul.tooling.in;
031
032import java.net.InetAddress;
033import java.util.List;
034import java.util.Map;
035
036import static java.lang.System.currentTimeMillis;
037
038@Model("com.azul.tooling.AnyConnectionEventModel")
039public final class NetConnectionEvent implements Tooling.ToolingEvent {
040    private static final boolean isEnabled;
041
042    static {
043        isEnabled = Tooling.isEventTypeEnabled(NetConnectionEvent.class);
044    }
045
046    private NetConnectionEvent(String protocol, InetAddress toAddress, int toPort, InetAddress fromAddress,
047                               int fromPort, long connectionInitiatedTime, long connectionEstablishedTime, boolean originating,
048                               Map<String, List<String>> headers, String url) {
049        this.protocol = protocol;
050        this.toAddress = toAddress;
051        this.toPort = toPort;
052        this.fromAddress = fromAddress;
053        this.fromPort = fromPort;
054        this.connectionInitiatedTime = connectionInitiatedTime;
055        this.connectionEstablishedTime = connectionEstablishedTime;
056        this.originating = originating;
057        this.headers = headers;
058        this.url = url;
059    }
060
061    public static NetConnectionEvent udpConnectionEvent(InetAddress toAddress, int toPort, InetAddress fromAddress,
062                                                        int fromPort, long endTimeConnection, boolean originating) {
063        return new NetConnectionEvent("udp", toAddress, toPort, fromAddress, fromPort, -1,
064                endTimeConnection, originating, null, null);
065    }
066
067    public static NetConnectionEvent httpConnectionEvent(String protocol, InetAddress toAddress, int toPort,
068                                                         Map<String, List<String>> headers, String url) {
069        return new NetConnectionEvent(protocol, toAddress, toPort, null, -1,
070                -1, currentTimeMillis(), true, headers, url);
071    }
072
073    public static NetConnectionEvent tcpConnectionEvent(InetAddress toAddress, int toPort, InetAddress fromAddress,
074                              int fromPort, long startTimeConnection, long endTimeConnection, boolean originating) {
075        return new NetConnectionEvent("tcp", toAddress, toPort, fromAddress, fromPort,
076                startTimeConnection, endTimeConnection, originating, null, null);
077    }
078
079    public static boolean isEnabled() {
080        return isEnabled;
081    }
082
083    public boolean isEventEnabled() {
084        return isEnabled();
085    }
086
087    public final String protocol;
088    public final InetAddress toAddress;
089    public final int toPort;
090    public final InetAddress fromAddress;
091    public final int fromPort;
092    public final long connectionInitiatedTime;
093    public final long connectionEstablishedTime;
094    public final Map<String, List<String>> headers;
095    public final String url;
096    public final boolean originating; // true for send/request, false for receive/respond
097}
